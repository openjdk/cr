33 package org.openjdk.jmc.agent;
34 
35 import org.openjdk.jmc.agent.util.TypeUtils;
36 
37 /**
38  * Metadata for a return value to be logged by the agent.
39  */
40 public final class ReturnValue {
41 	private final String name;
42 	private final String fieldName;
43 	private final String description;
44 	private final String contentType;
45 
46 	public ReturnValue(String name, String description, String contentType) {
47 		this.name = name == null ? "Return Value" : name;
48 		this.description = description;
49 		this.contentType = contentType;
50 		this.fieldName = "field" + TypeUtils.deriveIdentifierPart(this.name); //$NON-NLS-1$
51 	}
52 
53 	public String getName() {
54 		return name;
55 	}
56 
57 	public String getDescription() {
58 		return description;
59 	}
60 
61 	public String getContentType() {
62 		return contentType;
63 	}
64 
65 	public String getFieldName() {
66 		return fieldName;
67 	}
68 
69 }
