 73 /** A class that defines all predefined constants and operators
 74  *  as well as special classes such as java.lang.Object, which need
 75  *  to be known to the compiler. All symbols are held in instance
 76  *  fields. This makes it possible to work in multiple concurrent
 77  *  projects, which might use different class files for library classes.
 78  *
 79  *  <p><b>This is NOT part of any supported API.
 80  *  If you write code that depends on this, you do so at your own risk.
 81  *  This code and its internal interfaces are subject to change or
 82  *  deletion without notice.</b>
 83  */

 88     /** Get the symbol table instance. */
 89     public static Symtab instance(Context context) {
 90         Symtab instance = context.get(symtabKey);
 91         if (instance == null)
 92             instance = new Symtab(context);
 93         return instance;
 94     }

261     public VarSymbol getClassField(Type type, Types types) {
262         return classFields.computeIfAbsent(
263             new UniqueType(type, types), k -> {
264                 Type arg = null;
265                 if (type.getTag() == ARRAY || type.getTag() == CLASS)
266                     arg = types.erasure(type);
267                 else if (type.isPrimitiveOrVoid())
268                     arg = types.boxedClass(type).type;
269                 else
270                     throw new AssertionError(type);
271 
272                 Type t = new ClassType(
273                     classType.getEnclosingType(), List.of(arg), classType.tsym);
274                 return new VarSymbol(
275                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
276             });
277     }

307     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
308         final Completer completer = type.tsym.completer;
309         type.tsym.completer = new Completer() {
310             @Override
311             public void complete(Symbol sym) throws CompletionFailure {
312                 try {
313                     completer.complete(sym);
314                 } catch (CompletionFailure e) {
315                     sym.flags_field |= (PUBLIC | INTERFACE);
316                     ((ClassType) sym.type).supertype_field = objectType;
317                 }
318             }
319 
320             @Override
321             public boolean isTerminal() {
322                 return completer.isTerminal();
323             }
324         };
325     }
326 
327     public void synthesizeBoxTypeIfMissing(final Type type) {
328         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
329         final Completer completer = sym.completer;
330         sym.completer = new Completer() {
331             @Override
332             public void complete(Symbol sym) throws CompletionFailure {
333                 try {
334                     completer.complete(sym);
335                 } catch (CompletionFailure e) {
336                     sym.flags_field |= PUBLIC;
337                     ((ClassType) sym.type).supertype_field = objectType;
338                     MethodSymbol boxMethod =
339                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
340                                          new MethodType(List.of(type), sym.type,
341                                 List.nil(), methodClass),
342                             sym);
343                     sym.members().enter(boxMethod);
344                     MethodSymbol unboxMethod =
345                         new MethodSymbol(PUBLIC,
346                             type.tsym.name.append(names.Value), // x.intValue()
347                             new MethodType(List.nil(), type,
348                                 List.nil(), methodClass),
349                             sym);
350                     sym.members().enter(unboxMethod);
351                 }
352             }
353 
354             @Override
355             public boolean isTerminal() {
356                 return completer.isTerminal();
357             }
358         };
359     }

361     // Enter a synthetic class that is used to mark classes in ct.sym.
362     // This class does not have a class file.
363     private Type enterSyntheticAnnotation(String name) {
364         // for now, leave the module null, to prevent problems from synthesizing the
365         // existence of a class in any specific module, including noModule
366         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
367         ClassSymbol sym = (ClassSymbol)type.tsym;
368         sym.completer = Completer.NULL_COMPLETER;
369         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
370         sym.erasure_field = type;
371         sym.members_field = WriteableScope.create(sym);
372         type.typarams_field = List.nil();
373         type.allparams_field = List.nil();
374         type.supertype_field = annotationType;
375         type.interfaces_field = List.nil();
376         return type;
377     }

379     /** Constructor; enters all predefined identifiers and operators
380      *  into symbol table.
381      */
382     protected Symtab(Context context) throws CompletionFailure {
383         context.put(symtabKey, this);
384 
385         names = Names.instance(context);
386 
387         // Create the unknown type
388         unknownType = new UnknownType();
389 
390         messages = JavacMessages.instance(context);
391 
392         MissingInfoHandler missingInfoHandler = MissingInfoHandler.instance(context);
393 
394         Target target = Target.instance(context);
395         rootPackage = new RootPackageSymbol(names.empty, null,
396                                             missingInfoHandler,
397                                             target.runtimeUseNestAccess());
398 
399         // create the basic builtin symbols
400         unnamedModule = new ModuleSymbol(names.empty, null) {
401                 {
402                     directives = List.nil();
403                     exports = List.nil();
404                     provides = List.nil();
405                     uses = List.nil();
406                     ModuleSymbol java_base = enterModule(names.java_base);
407                     com.sun.tools.javac.code.Directive.RequiresDirective d =
408                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
409                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
410                     requires = List.of(d);
411                 }
412                 @Override
413                 public String toString() {
414                     return messages.getLocalizedString("compiler.misc.unnamed.module");
415                 }
416             };
417         addRootPackageFor(unnamedModule);
418         unnamedModule.enclosedPackages = unnamedModule.enclosedPackages.prepend(unnamedModule.unnamedPackage);
419 
420         errModule = new ModuleSymbol(names.empty, null) {
421                 {
422                     directives = List.nil();
423                     exports = List.nil();
424                     provides = List.nil();
425                     uses = List.nil();
426                     ModuleSymbol java_base = enterModule(names.java_base);
427                     com.sun.tools.javac.code.Directive.RequiresDirective d =
428                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
429                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
430                     requires = List.of(d);
431                 }
432             };
433         addRootPackageFor(errModule);
434 
435         noModule = new ModuleSymbol(names.empty, null) {
436             @Override public boolean isNoModule() {
437                 return true;
438             }
439         };
440         addRootPackageFor(noModule);
441 
442         noSymbol = new TypeSymbol(NIL, 0, names.empty, Type.noType, rootPackage) {
443             @Override @DefinedBy(Api.LANGUAGE_MODEL)
444             public <R, P> R accept(ElementVisitor<R, P> v, P p) {
445                 return v.visitUnknown(this, p);
446             }
447         };
448 
449         // create the error symbols
450         errSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.any, null, rootPackage);
451         errType = new ErrorType(errSymbol, Type.noType);
452 
453         unknownSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.fromString("<any?>"), null, rootPackage);
454         unknownSymbol.members_field = new Scope.ErrorScope(unknownSymbol);
455         unknownSymbol.type = unknownType;
456 
457         // initialize builtin types
458         initType(byteType, "byte", "Byte");
459         initType(shortType, "short", "Short");
460         initType(charType, "char", "Character");
461         initType(intType, "int", "Integer");
462         initType(longType, "long", "Long");
463         initType(floatType, "float", "Float");
464         initType(doubleType, "double", "Double");
465         initType(booleanType, "boolean", "Boolean");
466         initType(voidType, "void", "Void");
467         initType(botType, "<nulltype>");
468         initType(errType, errSymbol);
469         initType(unknownType, unknownSymbol);
470 
471         // the builtin class of all arrays
472         arrayClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Array, noSymbol);
473 
474         // VGJ
475         boundClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Bound, noSymbol);
476         boundClass.members_field = new Scope.ErrorScope(boundClass);
477 
478         // the builtin class of all methods
479         methodClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Method, noSymbol);
480         methodClass.members_field = new Scope.ErrorScope(boundClass);
481 
482         // Create class to hold all predefined constants and operations.
483         predefClass = new ClassSymbol(PUBLIC|ACYCLIC, names.empty, rootPackage);
484         WriteableScope scope = WriteableScope.create(predefClass);
485         predefClass.members_field = scope;
486 
487         // Get the initial completer for Symbols from the ClassFinder
488         initialCompleter = ClassFinder.instance(context).getCompleter();
489         rootPackage.members_field = WriteableScope.create(rootPackage);
490 
491         // Enter symbols for basic types.
492         scope.enter(byteType.tsym);
493         scope.enter(shortType.tsym);
494         scope.enter(charType.tsym);
495         scope.enter(intType.tsym);
496         scope.enter(longType.tsym);
497         scope.enter(floatType.tsym);
498         scope.enter(doubleType.tsym);
499         scope.enter(booleanType.tsym);
500         scope.enter(errType.tsym);

545         annotationType = enterClass("java.lang.annotation.Annotation");
546         classLoaderType = enterClass("java.lang.ClassLoader");
547         enumSym = enterClass(java_base, names.java_lang_Enum);

549             new MethodSymbol(PROTECTED|FINAL|HYPOTHETICAL,
550                              names.finalize,
551                              new MethodType(List.nil(), voidType,
552                                             List.nil(), methodClass),
553                              enumSym);

613         arrayClassType.supertype_field = objectType;
614         arrayClassType.interfaces_field = List.of(cloneableType, serializableType);
615         arrayClass.members_field = WriteableScope.create(arrayClass);
616         lengthVar = new VarSymbol(
617             PUBLIC | FINAL,
618             names.length,
619             intType,
620             arrayClass);
621         arrayClass.members().enter(lengthVar);
622         arrayCloneMethod = new MethodSymbol(
623             PUBLIC,
624             names.clone,
625             new MethodType(List.nil(), objectType,
626                            List.nil(), methodClass),
627             arrayClass);
628         arrayClass.members().enter(arrayCloneMethod);
629 
630         if (java_base != noModule)
631             java_base.completer = moduleCompleter::complete; //bootstrap issues
632 
633     }

643     /** Create a new toplevel or member class symbol with given name
644      *  and owner and enter in `classes' unless already there.
645      */
646     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
647         Assert.checkNonNull(msym);
648         Name flatname = TypeSymbol.formFlatName(name, owner);
649         ClassSymbol c = getClass(msym, flatname);
650         if (c == null) {
651             c = defineClass(name, owner);
652             doEnterClass(msym, c);
653         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
654             // reassign fields of classes that might have been loaded with
655             // their flat names.
656             c.owner.members().remove(c);
657             c.name = name;
658             c.owner = owner;
659             c.fullname = ClassSymbol.formFullName(name, owner);
660         }
661         return c;
662     }

669     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
670         Assert.checkNonNull(msym);
671 
672         if (flatName.isEmpty()) {
673             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
674             return msym.unnamedPackage;
675         }
676 
677         if (msym == noModule) {
678             return enterPackage(msym, flatName);
679         }
680 
681         msym.complete();
682 
683         PackageSymbol pack;
684 
685         pack = msym.visiblePackages.get(flatName);
686 
687         if (pack != null)
688             return pack;
689 
690         pack = getPackage(msym, flatName);
691 
692         if (pack != null && pack.exists())
693             return pack;
694 
695         boolean dependsOnUnnamed = msym.requires != null &&
696                                    msym.requires.stream()
697                                                 .map(rd -> rd.module)
698                                                 .anyMatch(mod -> mod == unnamedModule);
699 
700         if (dependsOnUnnamed) {
701             //msyms depends on the unnamed module, for which we generally don't know
702             //the list of packages it "exports" ahead of time. So try to lookup the package in the
703             //current module, and in the unnamed module and see if it exists in one of them
704             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
705 
706             if (unnamedPack != null && unnamedPack.exists()) {
707                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
708                 return unnamedPack;
709             }
710 
711             pack = enterPackage(msym, flatName);
712             pack.complete();
713             if (pack.exists())
714                 return pack;
715 
716             unnamedPack = enterPackage(unnamedModule, flatName);
717             unnamedPack.complete();
718             if (unnamedPack.exists()) {
719                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
720                 return unnamedPack;
721             }
722 
723             return pack;
724         }
725 
726         return enterPackage(msym, flatName);
727     }

767     /** Make a package, given its fully qualified name.
768      */
769     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
770         Assert.checkNonNull(currModule);
771         PackageSymbol p = getPackage(currModule, fullname);
772         if (p == null) {
773             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
774             p = new PackageSymbol(
775                     Convert.shortName(fullname),
776                     enterPackage(currModule, Convert.packagePart(fullname)));
777             p.completer = initialCompleter;
778             p.modle = currModule;
779             doEnterPackage(currModule, p);
780         }
781         return p;
782     }

790         doEnterPackage(module, rootPackage);
791         PackageSymbol unnamedPackage = new PackageSymbol(names.empty, rootPackage) {
792                 @Override

794                     return messages.getLocalizedString("compiler.misc.unnamed.package");

797         unnamedPackage.modle = module;
798         //we cannot use a method reference below, as initialCompleter might be null now
799         unnamedPackage.completer = s -> initialCompleter.complete(s);
800         unnamedPackage.flags_field |= EXISTS;
801         module.unnamedPackage = unnamedPackage;

846             return List.nil();
