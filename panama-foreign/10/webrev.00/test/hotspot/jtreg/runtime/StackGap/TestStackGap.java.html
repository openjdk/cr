24 /*
25  * @test
26  * @summary Linux kernel stack guard should not cause segfaults on x86-32
27  * @modules java.base/jdk.internal.misc
28  * @library /test/lib
29  * @requires os.family == "linux"
30  * @compile T.java
31  * @run main/native TestStackGap
32  */
33 
34 
35 import jdk.test.lib.Utils;
36 import jdk.test.lib.process.ProcessTools;
37 import jdk.test.lib.process.OutputAnalyzer;
38 
39 public class TestStackGap {
40     public static void main(String args[]) throws Exception {
41         ProcessBuilder pb = ProcessTools.createNativeTestProcessBuilder("stack-gap");
42         pb.environment().put("CLASSPATH", Utils.TEST_CLASS_PATH);
43         new OutputAnalyzer(pb.start())
44             .shouldHaveExitValue(0);
45 
46         pb = ProcessTools.createNativeTestProcessBuilder("stack-gap",
47                                                          "-XX:+DisablePrimordialThreadGuardPages");
48         pb.environment().put("CLASSPATH", Utils.TEST_CLASS_PATH);
49         new OutputAnalyzer(pb.start())
50             .shouldHaveExitValue(0);
51     }
52 }
