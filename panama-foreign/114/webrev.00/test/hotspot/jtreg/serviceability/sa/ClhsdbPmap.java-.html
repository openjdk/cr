24 import java.util.HashMap;
25 import java.util.List;
26 import java.util.Map;
27 
28 import jdk.test.lib.apps.LingeredApp;
29 import jtreg.SkippedException;
30 
31 /**
32  * @test
33  * @bug 8190198
34  * @summary Test clhsdb pmap command
35  * @requires vm.hasSA
36  * @library /test/lib
37  * @requires os.family != "mac"
38  * @run main/othervm ClhsdbPmap
39  */
40 
41 public class ClhsdbPmap {
42 
43     public static void main(String[] args) throws Exception {
44         System.out.println("Starting ClhsdbPmap test");
45 
46         LingeredApp theApp = null;
47         try {
48             ClhsdbLauncher test = new ClhsdbLauncher();
49             theApp = LingeredApp.startApp();
50             System.out.println("Started LingeredApp with pid " + theApp.getPid());
51 
52             List<String> cmds = List.of("pmap");
53 
54             Map<String, List<String>> expStrMap = new HashMap<>();
55             expStrMap.put("pmap",
56                     List.of("jvm", "java", "net", "nio", "jimage", "zip"));
57 
58             test.run(theApp.getPid(), cmds, expStrMap, null);
59         } catch (SkippedException se) {
60             throw se;
61         } catch (Exception ex) {
62             throw new RuntimeException("Test ERROR " + ex, ex);
63         } finally {
64             LingeredApp.stopApp(theApp);
65         }
66         System.out.println("Test PASSED");
67     }
68 }
