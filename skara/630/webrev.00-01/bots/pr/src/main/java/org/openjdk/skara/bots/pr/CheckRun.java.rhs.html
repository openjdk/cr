 87     private boolean isTargetBranchAllowed() {
 88         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 89         return matcher.matches();
 90     }

 92     private List<String> allowedTargetBranches() {
 93         return pr.repository()
 94                  .branches()
 95                  .stream()
 96                  .map(HostedBranch::name)
 97                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 98                  .filter(Matcher::matches)
 99                  .map(Matcher::group)
100                  .collect(Collectors.toList());
101     }

103     // For unknown contributors, check that all commits have the same name and email
104     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
105         var author = censusInstance.namespace().get(pr.author().id());
106         if (author != null) {
107             return true;
108         }
109 
110         var names = new HashSet<String>();
111         var emails = new HashSet<String>();
112 
113         for (var commit : commits) {
114             names.add(commit.author().name());
115             emails.add(commit.author().email());
116         }
117 
118         return ((names.size() == 1) && emails.size() == 1);
119     }

121     // Additional bot-specific checks that are not handled by JCheck
122     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
123         var ret = new ArrayList<String>();
124 
125         if (bodyWithoutStatus().isBlank()) {
126             var error = "The pull request body must not be empty.";
127             ret.add(error);
128         }
129 
130         if (!isTargetBranchAllowed()) {
131             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
132                     allowedTargetBranches().stream()
133                     .map(name -> "   - " + name)
134                     .collect(Collectors.joining("\n"));
135             ret.add(error);
136         }
137 
138         var headHash = pr.headHash();
139         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
140 
141         if (!checkCommitAuthor(originalCommits)) {
142             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
143                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
144                     "email addresses. Please amend the commits.";
145             ret.add(error);
146         }
147 
148         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
149             if (labels.contains(blocker.getKey())) {
150                 ret.add(blocker.getValue());
151             }
152         }
153 
154         return ret;
155     }

157     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
158         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
159             checkBuilder.complete(true);
160         } else {
161             checkBuilder.title("Required");
162             var summary = Stream.concat(visitor.messages().stream(), additionalErrors.stream())
163                                 .sorted()
164                                 .map(m -> "- " + m)
165                                 .collect(Collectors.joining("\n"));
166             checkBuilder.summary(summary);
167             for (var annotation : visitor.getAnnotations()) {
168                 checkBuilder.annotation(annotation);
169             }
170             checkBuilder.complete(false);
171         }
172     }

174     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
175         // Additional errors are not allowed
176         if (!additionalErrors.isEmpty()) {
177             newLabels.remove("rfr");
178             return;
179         }
180 
181         // Draft requests are not for review
182         if (pr.isDraft()) {
183             newLabels.remove("rfr");
184             return;
185         }
186 
187         // Check if the visitor found any issues that should be resolved before reviewing
188         if (visitor.isReadyForReview()) {
189             newLabels.add("rfr");
190         } else {
191             newLabels.remove("rfr");
192         }
193     }

195     private String getRole(String username) {
196         var project = censusInstance.project();
197         var version = censusInstance.census().version().format();
198         if (project.isReviewer(username, version)) {
199             return "**Reviewer**";
200         } else if (project.isCommitter(username, version)) {
201             return "Committer";
202         } else if (project.isAuthor(username, version)) {
203             return "Author";
204         } else {
205             return "no project role";
206         }
207     }

209     private String formatReviewer(HostUser reviewer) {
210         var namespace = censusInstance.namespace();
211         var contributor = namespace.get(reviewer.id());
212         if (contributor == null) {
213             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
214         } else {
215             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
216             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
217                     getRole(contributor.username()) + ")";
218         }
219     }

234     private Optional<String> getReviewersList(List<Review> reviews) {
235         var reviewers = reviews.stream()
236                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
237                                .map(review -> {
238                                    var entry = " * " + formatReviewer(review.reviewer());
239                                    if (!review.hash().equals(pr.headHash())) {
240                                        if (ignoreStaleReviews) {
241                                            entry += " üîÑ Re-review required (review applies to " + review.hash() + ")";
242                                        } else {
243                                            entry += " ‚ö†Ô∏è Review applies to " + review.hash();
244                                        }
245                                    }
246                                    return entry;
247                                })
248                                .collect(Collectors.joining("\n"));
249 
250         // Check for manually added reviewers
251         if (!ignoreStaleReviews) {
252             var namespace = censusInstance.namespace();
253             var allReviewers = PullRequestUtils.reviewerNames(reviews, namespace);




254             var additionalEntries = new ArrayList<String>();
255             for (var additional : Reviewers.reviewers(pr.repository().forge().currentUser(), comments)) {
256                 if (!allReviewers.contains(additional)) {
257                     additionalEntries.add(" * " + additional + " - " + getRole(additional) + " ‚ö†Ô∏è Added manually");
258                 }
259             }
260             if (!reviewers.isBlank()) {
261                 reviewers += "\n";
262             }
263             reviewers += String.join("\n", additionalEntries);
264         }
265 
266         if (reviewers.length() > 0) {
267             return Optional.of(reviewers);
268         } else {
269             return Optional.empty();
270         }
271     }

290     private boolean relaxedEquals(String s1, String s2) {
291         return s1.trim()
292                  .replaceAll("\\s+", " ")
293                  .equalsIgnoreCase(s2.trim()
294                                      .replaceAll("\\s+", " "));
295     }

298     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
299         var progressBody = new StringBuilder();
300         progressBody.append("---------\n");
301         progressBody.append("### Progress\n");
302         progressBody.append(getChecksList(visitor));
303 
304         var allAdditionalErrors = Stream.concat(visitor.hiddenMessages().stream(), additionalErrors.stream())
305                                         .sorted()
306                                         .collect(Collectors.toList());
307         if (!allAdditionalErrors.isEmpty()) {
308             progressBody.append("\n\n### Error");
309             if (allAdditionalErrors.size() > 1) {
310                 progressBody.append("s");
311             }
312             progressBody.append("\n");
313             progressBody.append(getAdditionalErrorsList(allAdditionalErrors));
314         }
315 
316         var issue = Issue.fromStringRelaxed(pr.title());
317         var issueProject = workItem.bot.issueProject();
318         if (issueProject != null && issue.isPresent()) {
319             var allIssues = new ArrayList<Issue>();
320             allIssues.add(issue.get());
321             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
322             progressBody.append("\n\n### Issue");
323             if (allIssues.size() > 1) {
324                 progressBody.append("s");
325             }
326             progressBody.append("\n");
327             for (var currentIssue : allIssues) {
328                 progressBody.append(" * ");
329                 if (currentIssue.project().isPresent() && !currentIssue.project().get().equals(issueProject.name())) {
330                     progressBody.append("‚ö†Ô∏è Issue `");
331                     progressBody.append(currentIssue.id());
332                     progressBody.append("` does not belong to the `");
333                     progressBody.append(issueProject.name());
334                     progressBody.append("` project.");
335                 } else {
336                     try {
337                         var iss = issueProject.issue(currentIssue.shortId());
338                         if (iss.isPresent()) {
339                             progressBody.append("[");
340                             progressBody.append(iss.get().id());
341                             progressBody.append("](");
342                             progressBody.append(iss.get().webUrl());
343                             progressBody.append("): ");
344                             progressBody.append(iss.get().title());
345                             if (!relaxedEquals(iss.get().title(), currentIssue.description())) {
346                                 progressBody.append(" ‚ö†Ô∏è Title mismatch between PR and JBS.");
347                             }
348                             progressBody.append("\n");
349                         } else {
350                             progressBody.append("‚ö†Ô∏è Failed to retrieve information on issue `");
351                             progressBody.append(currentIssue.id());
352                             progressBody.append("`.\n");
353                         }
354                     } catch (RuntimeException e) {
355                         progressBody.append("‚ö†Ô∏è Temporary failure when trying to retrieve information on issue `");
356                         progressBody.append(currentIssue.id());
357                         progressBody.append("`.\n");
358                     }
359                 }
360             }
361         }
362 
363         getReviewersList(reviews).ifPresent(reviewers -> {
364             progressBody.append("\n\n### Reviewers\n");
365             progressBody.append(reviewers);
366         });
367 
368         getContributorsList(comments).ifPresent(contributors -> {
369             progressBody.append("\n\n### Contributors\n");
370             progressBody.append(contributors);
371         });
372 
373         progressBody.append("\n\n### Download\n");
374         progressBody.append(checkoutCommands());
375 
376         return progressBody.toString();
377     }

386     private String bodyWithoutStatus() {
387         var description = pr.body();
388         var markerIndex = description.lastIndexOf(progressMarker);
389         return (markerIndex < 0 ?
390                 description :
391                 description.substring(0, markerIndex)).trim();
392     }

394     private String updateStatusMessage(String message) {
395         var description = pr.body();
396         var markerIndex = description.lastIndexOf(progressMarker);
397 
398         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
399             log.info("Progress already up to date");
400             return description;
401         }
402         var originalBody = bodyWithoutStatus();
403         if (originalBody.isBlank()) {
404             originalBody = emptyPrBodyMarker;
405         }
406         var newBody = originalBody + "\n" + progressMarker + "\n" + message;
407 
408         // TODO? Retrieve the body again here to lower the chance of concurrent updates
409         pr.setBody(newBody);
410         return newBody;
411     }

413     private String verdictToString(Review.Verdict verdict) {
414         switch (verdict) {
415             case APPROVED:
416                 return "changes are approved";
417             case DISAPPROVED:
418                 return "more changes needed";
419             case NONE:
420                 return "comment added";
421             default:
422                 throw new RuntimeException("Unknown verdict: " + verdict);
423         }
424     }

446     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
447         var message = new StringBuilder();
448         message.append("@");
449         message.append(pr.author().userName());
450         message.append(" This change now passes all automated pre-integration checks");
451 
452         try {
453             var hasContributingFile =
454                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
455             if (hasContributingFile) {
456                 message.append(". When the change also fulfills all ");
457                 message.append("[project specific requirements](https://github.com/");
458                 message.append(pr.repository().name());
459                 message.append("/blob/");
460                 message.append(pr.targetRef());
461                 message.append("/CONTRIBUTING.md)");
462             }
463         } catch (IOException e) {
464             throw new UncheckedIOException(e);
465         }
466 
467         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
468         message.append("the commit message will be:\n");
469         message.append("```\n");
470         message.append(commitMessage);
471         message.append("\n```\n");
472 
473         message.append("- If you would like to add a summary, use the `/summary` command.\n");
474         message.append("- To credit additional contributors, use the `/contributor` command.\n");
475         message.append("- To add additional solved issues, use the `/issue` command.\n");
476 
477         var divergingCommits = checkablePullRequest.divergingCommits();
478         if (divergingCommits.size() > 0) {
479             message.append("\n");
480             message.append("Since the source branch of this PR was last updated there ");
481             if (divergingCommits.size() == 1) {
482                 message.append("has been 1 commit ");
483             } else {
484                 message.append("have been ");
485                 message.append(divergingCommits.size());
486                 message.append(" commits ");
487             }
488             message.append("pushed to the `");
489             message.append(pr.targetRef());
490             message.append("` branch:\n\n");
491             divergingCommits.stream()
492                             .limit(10)
493                             .forEach(c -> message.append(" * ").append(c.hash().hex()).append(": ").append(c.message().get(0)).append("\n"));
494             if (divergingCommits.size() > 10) {
495                 message.append(" * ... and ").append(divergingCommits.size() - 10).append(" more: ")
496                        .append(pr.repository().webUrl(baseHash.hex(), pr.targetRef())).append("\n");
497             }
498 
499             message.append("\n");
500             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
501             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
502             message.append(pr.targetRef());
503             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
504             message.append("`/integrate ");
505             message.append(pr.targetHash());
506             message.append("`.\n");
507         } else {
508             message.append("\n");
509             message.append("There are currently no new commits on the `");
510             message.append(pr.targetRef());
511             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
512             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
513             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
514             message.append("`/integrate ");
515             message.append(pr.targetHash());
516             message.append("`.\n");
517         }
518 
519         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
520             message.append("\n");
521             var contributor = censusInstance.namespace().get(pr.author().id());
522             if (contributor == null) {
523                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
524             } else {
525                 message.append("As you do not have Committer status in this project, ");
526             }
527 
528             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
529             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
530             var candidates = reviews.stream()
531                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
532                                     .map(review -> "@" + review.reviewer().userName())
533                                     .collect(Collectors.joining(", "));
534             if (candidates.length() > 0) {
535                 message.append("Possible candidates are the reviewers of this PR (");
536                 message.append(candidates);
537                 message.append(") but any other Committer may sponsor as well. ");
538             }
539             message.append("\n\n");
540             message.append("‚û°Ô∏è To flag this PR as ready for integration with the above commit message, type ");
541             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
542             message.append("`/sponsor` in a new comment to perform the integration).\n");
543         } else {
544             message.append("\n");
545             message.append("‚û°Ô∏è To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
546             message.append("`/integrate` in a new comment.\n");
547         }
548         message.append(mergeReadyMarker);
549         return message.toString();
550     }

561     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
562         var existing = findComment(comments, mergeReadyMarker);
563         if (isReady && rebasePossible) {
564             var message = getMergeReadyComment(commitMessage, reviews);
565             if (existing.isEmpty()) {
566                 pr.addComment(message);
567             } else {
568                 pr.updateComment(existing.get().id(), message);
569             }
570         } else {
571             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
572         }
573     }

575     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
576         var existing = findComment(comments, sourceBranchWarningMarker);
577         if (existing.isPresent()) {
578             // Only add the comment once per PR
579             return;
580         }
581         var branch = pr.sourceRef();
582         var message = ":warning: @" + pr.author().userName() + " " +
583             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
584             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
585             "If you eventually integrate this pull request then the branch `" + branch + "` " +
586             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
587             "your personal fork with the upstream repository.\n" +
588             "\n" +
589             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
590             "You can do this by running the following commands in a local repository for your personal fork. " +
591             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
592             "\n" +
593             "```" +
594             "$ git checkout " + branch + "\n" +
595             "$ git checkout -b NEW-BRANCH-NAME\n" +
596             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
597             "$ git push -f origin " + branch + "\n" +
598             "```\n" +
599             "\n" +
600             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
601             "close this one.\n" +
602             sourceBranchWarningMarker;
603         pr.addComment(message);
604     }

606     private void addOutdatedComment(List<Comment> comments) {
607         var existing = findComment(comments, outdatedHelpMarker);
608         if (existing.isPresent()) {
609             // Only add the comment once per PR
610             return;
611         }
612         var message = "@" + pr.author().userName() + " this pull request can not be integrated into " +
613                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
614                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
615                 "```bash\n" +
616                 "git checkout " + pr.sourceRef() + "\n" +
617                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
618                 "git merge FETCH_HEAD\n" +
619                 "# resolve conflicts and follow the instructions given by git merge\n" +
620                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
621                 "git push\n" +
622                 "```\n" +
623                 outdatedHelpMarker;
624         pr.addComment(message);
625     }

627     private void checkStatus() {
628         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
629         var censusDomain = censusInstance.configuration().census().domain();
630         Exception checkException = null;
631 
632         try {
633             // Post check in-progress
634             log.info("Starting to run jcheck on PR head");
635             pr.createCheck(checkBuilder.build());
636 
637             var ignored = new PrintWriter(new StringWriter());
638             var rebasePossible = true;
639             var commitHash = pr.headHash();
640             var mergedHash = checkablePullRequest.mergeTarget(ignored);
641             if (mergedHash.isPresent()) {
642                 commitHash = mergedHash.get();
643             } else {
644                 rebasePossible = false;
645             }
646 
647             List<String> additionalErrors = List.of();
648             Hash localHash;
649             try {
650                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
651             } catch (CommitFailure e) {
652                 additionalErrors = List.of(e.getMessage());
653                 localHash = baseHash;
654             }
655             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
656             if (!localHash.equals(baseHash)) {
657                 // Determine current status
658                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
659                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
660                 additionalErrors = botSpecificChecks(localHash);
661             } else {
662                 if (additionalErrors.isEmpty()) {
663                     additionalErrors = List.of("This PR contains no changes");
664                 }
665             }
666             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
667             updateReadyForReview(visitor, additionalErrors);
668 
669             // Calculate and update the status message if needed
670             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
671             var updatedBody = updateStatusMessage(statusMessage);
672 
673             // Post / update approval messages (only needed if the review itself can't contain a body)
674             if (!pr.repository().forge().supportsReviewBody()) {
675                 updateReviewedMessages(comments, allReviews);
676             }
677 
678             var commit = localRepo.lookup(localHash).orElseThrow();
679             var commitMessage = String.join("\n", commit.message());
680             var readyForIntegration = visitor.messages().isEmpty() && additionalErrors.isEmpty();
681             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
682             if (readyForIntegration && rebasePossible) {
683                 newLabels.add("ready");
684             } else {
685                 newLabels.remove("ready");
686             }
687             if (!rebasePossible) {
688                 if (!labels.contains("failed-auto-merge")) {
689                     addOutdatedComment(comments);
690                 }
691                 newLabels.add("merge-conflict");
692             } else {
693                 newLabels.remove("merge-conflict");
694             }
695 
696             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
697             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
698                 addSourceBranchWarningComment(comments);
699             }
700 
701             // Ensure that the ready for sponsor label is up to date
702             newLabels.remove("sponsor");
703             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
704             if (readyHash.isPresent() && readyForIntegration) {
705                 var acceptedHash = readyHash.get();
706                 if (pr.headHash().equals(acceptedHash)) {
707                     newLabels.add("sponsor");
708                 }
709             }
710 
711             // Calculate current metadata to avoid unnecessary future checks
712             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
713                                                 censusInstance, pr.targetHash(), pr.isDraft());
714             checkBuilder.metadata(metadata);
715         } catch (Exception e) {
716             log.throwing("CommitChecker", "checkStatus", e);
717             newLabels.remove("ready");
718             checkBuilder.metadata("invalid");
719             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
720             checkBuilder.summary(e.getMessage());
721             checkBuilder.complete(false);
722             checkException = e;
723         }
724         var check = checkBuilder.build();
725         pr.updateCheck(check);
726 
727         // Synchronize the wanted set of labels
728         for (var newLabel : newLabels) {
729             if (!labels.contains(newLabel)) {
730                 pr.addLabel(newLabel);
731             }
732         }
733         for (var oldLabel : labels) {
734             if (!newLabels.contains(oldLabel)) {
735                 pr.removeLabel(oldLabel);
736             }
737         }
738 
739         // After updating the PR, rethrow any exception to automatically retry on transient errors
740         if (checkException != null) {
741             throw new RuntimeException("Exception during jcheck", checkException);
742         }
743     }
