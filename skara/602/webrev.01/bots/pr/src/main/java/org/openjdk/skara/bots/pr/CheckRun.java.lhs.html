 78     static void execute(CheckWorkItem workItem, PullRequest pr, Repository localRepo, List<Comment> comments,
 79                         List<Review> allReviews, List<Review> activeReviews, Set<String> labels, CensusInstance censusInstance,
 80                         boolean ignoreStaleReviews) throws IOException {

 85     private boolean isTargetBranchAllowed() {
 86         var matcher = workItem.bot.allowedTargetBranches().matcher(pr.targetRef());
 87         return matcher.matches();
 88     }

 90     private List<String> allowedTargetBranches() {
 91         return pr.repository()
 92                  .branches()
 93                  .stream()
 94                  .map(HostedBranch::name)
 95                  .map(name -> workItem.bot.allowedTargetBranches().matcher(name))
 96                  .filter(Matcher::matches)
 97                  .map(Matcher::group)
 98                  .collect(Collectors.toList());
 99     }

101     // For unknown contributors, check that all commits have the same name and email
102     private boolean checkCommitAuthor(List<CommitMetadata> commits) throws IOException {
103         var author = censusInstance.namespace().get(pr.author().id());
104         if (author != null) {
105             return true;
106         }
107 
108         var names = new HashSet<String>();
109         var emails = new HashSet<String>();
110 
111         for (var commit : commits) {
112             names.add(commit.author().name());
113             emails.add(commit.author().email());
114         }
115 
116         return ((names.size() == 1) && emails.size() == 1);
117     }

119     // Additional bot-specific checks that are not handled by JCheck
120     private List<String> botSpecificChecks(Hash finalHash) throws IOException {
121         var ret = new ArrayList<String>();
122 
123         if (bodyWithoutStatus().isBlank()) {
124             var error = "The pull request body must not be empty.";
125             ret.add(error);
126         }
127 
128         if (!isTargetBranchAllowed()) {
129             var error = "The branch `" + pr.targetRef() + "` is not allowed as target branch. The allowed target branches are:\n" +
130                     allowedTargetBranches().stream()
131                     .map(name -> "   - " + name)
132                     .collect(Collectors.joining("\n"));
133             ret.add(error);
134         }
135 
136         var headHash = pr.headHash();
137         var originalCommits = localRepo.commitMetadata(baseHash, headHash);
138 
139         if (!checkCommitAuthor(originalCommits)) {
140             var error = "For contributors who are not existing OpenJDK Authors, commit attribution will be taken from " +
141                     "the commits in the PR. However, the commits in this PR have inconsistent user names and/or " +
142                     "email addresses. Please amend the commits.";
143             ret.add(error);
144         }
145 
146         for (var blocker : workItem.bot.blockingCheckLabels().entrySet()) {
147             if (labels.contains(blocker.getKey())) {
148                 ret.add(blocker.getValue());
149             }
150         }
151 
152         return ret;
153     }

155     private void updateCheckBuilder(CheckBuilder checkBuilder, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
156         if (visitor.isReadyForReview() && additionalErrors.isEmpty()) {
157             checkBuilder.complete(true);
158         } else {
159             checkBuilder.title("Required");
160             var summary = Stream.concat(visitor.getMessages().stream(), additionalErrors.stream())
161                                 .sorted()
162                                 .map(m -> "- " + m)
163                                 .collect(Collectors.joining("\n"));
164             checkBuilder.summary(summary);
165             for (var annotation : visitor.getAnnotations()) {
166                 checkBuilder.annotation(annotation);
167             }
168             checkBuilder.complete(false);
169         }
170     }

172     private void updateReadyForReview(PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
173         // Additional errors are not allowed
174         if (!additionalErrors.isEmpty()) {
175             newLabels.remove("rfr");
176             return;
177         }
178 
179         // Draft requests are not for review
180         if (pr.isDraft()) {
181             newLabels.remove("rfr");
182             return;
183         }
184 
185         // Check if the visitor found any issues that should be resolved before reviewing
186         if (visitor.isReadyForReview()) {
187             newLabels.add("rfr");
188         } else {
189             newLabels.remove("rfr");
190         }
191     }

193     private String getRole(String username) {
194         var project = censusInstance.project();
195         var version = censusInstance.census().version().format();
196         if (project.isReviewer(username, version)) {
197             return "**Reviewer**";
198         } else if (project.isCommitter(username, version)) {
199             return "Committer";
200         } else if (project.isAuthor(username, version)) {
201             return "Author";
202         } else {
203             return "no project role";
204         }
205     }

207     private String formatReviewer(HostUser reviewer) {
208         var namespace = censusInstance.namespace();
209         var contributor = namespace.get(reviewer.id());
210         if (contributor == null) {
211             return reviewer.userName() + " (no known " + namespace.name() + " user name / role)";
212         } else {
213             var userNameLink = "[" + contributor.username() + "](@" + reviewer.userName() + ")";
214             return contributor.fullName().orElse(contributor.username()) + " (" + userNameLink + " - " +
215                     getRole(contributor.username()) + ")";
216         }
217     }

232     private Optional<String> getReviewersList(List<Review> reviews) {
233         var reviewers = reviews.stream()
234                                .filter(review -> review.verdict() == Review.Verdict.APPROVED)
235                                .map(review -> {
236                                    var entry = " * " + formatReviewer(review.reviewer());
237                                    if (!review.hash().equals(pr.headHash())) {
238                                        if (ignoreStaleReviews) {
239                                            entry += " üîÑ Re-review required (review applies to " + review.hash() + ")";
240                                        } else {
241                                            entry += " ‚ö†Ô∏è Review applies to " + review.hash();
242                                        }
243                                    }
244                                    return entry;
245                                })
246                                .collect(Collectors.joining("\n"));
247         if (reviewers.length() > 0) {
248             return Optional.of(reviewers);
249         } else {
250             return Optional.empty();
251         }
252     }

271     private String getStatusMessage(List<Comment> comments, List<Review> reviews, PullRequestCheckIssueVisitor visitor, List<String> additionalErrors) {
272         var progressBody = new StringBuilder();
273         progressBody.append("---------\n");
274         progressBody.append("### Progress\n");
275         progressBody.append(getChecksList(visitor));
276 
277         if (!additionalErrors.isEmpty()) {
278             progressBody.append("\n\n### Error");
279             if (additionalErrors.size() > 1) {
280                 progressBody.append("s");
281             }
282             progressBody.append("\n");
283             progressBody.append(getAdditionalErrorsList(additionalErrors));
284         }
285 
286         var issue = Issue.fromString(pr.title());
287         var issueProject = workItem.bot.issueProject();
288         if (issueProject != null && issue.isPresent()) {
289             var allIssues = new ArrayList<Issue>();
290             allIssues.add(issue.get());
291             allIssues.addAll(SolvesTracker.currentSolved(pr.repository().forge().currentUser(), comments));
292             progressBody.append("\n\n### Issue");
293             if (allIssues.size() > 1) {
294                 progressBody.append("s");
295             }
296             progressBody.append("\n");
297             for (var currentIssue : allIssues) {
298                 progressBody.append(" * ");
299                 try {
300                     var iss = issueProject.issue(currentIssue.id());
301                     if (iss.isPresent()) {
302                         progressBody.append("[");
303                         progressBody.append(iss.get().id());
304                         progressBody.append("](");
305                         progressBody.append(iss.get().webUrl());
306                         progressBody.append("): ");
307                         progressBody.append(iss.get().title());
308                         progressBody.append("\n");
309                     } else {
310                         progressBody.append("‚ö†Ô∏è Failed to retrieve information on issue `");
311                         progressBody.append(currentIssue.id());
312                         progressBody.append("`.\n");
313                     }
314                 } catch (RuntimeException e) {
315                     progressBody.append("‚ö†Ô∏è Temporary failure when trying to retrieve information on issue `");
316                     progressBody.append(currentIssue.id());
317                     progressBody.append("`.\n");
318                 }
319             }
320         }
321 
322         getReviewersList(reviews).ifPresent(reviewers -> {
323             progressBody.append("\n\n### Reviewers\n");
324             progressBody.append(reviewers);
325         });
326 
327         getContributorsList(comments).ifPresent(contributors -> {
328             progressBody.append("\n\n### Contributors\n");
329             progressBody.append(contributors);
330         });
331 
332         progressBody.append("\n\n### Download\n");
333         progressBody.append(checkoutCommands());
334 
335         return progressBody.toString();
336     }

339         var repoUrl = pr.repository().webUrl();
340         return
341            "`$ git fetch " + repoUrl + " " + pr.fetchRef() + ":pull/" + pr.id() + "`\n" +
342            "`$ git checkout pull/" + pr.id() + "`\n";
343     }

345     private String bodyWithoutStatus() {
346         var description = pr.body();
347         var markerIndex = description.lastIndexOf(progressMarker);
348         return (markerIndex < 0 ?
349                 description :
350                 description.substring(0, markerIndex)).trim();
351     }

353     private String updateStatusMessage(String message) {
354         var description = pr.body();
355         var markerIndex = description.lastIndexOf(progressMarker);
356 
357         if (markerIndex >= 0 && description.substring(markerIndex).equals(message)) {
358             log.info("Progress already up to date");
359             return description;
360         }
361         var newBody = bodyWithoutStatus() + "\n" + progressMarker + "\n" + message;
362 
363         // TODO? Retrieve the body again here to lower the chance of concurrent updates
364         pr.setBody(newBody);
365         return newBody;
366     }

368     private String verdictToString(Review.Verdict verdict) {
369         switch (verdict) {
370             case APPROVED:
371                 return "changes are approved";
372             case DISAPPROVED:
373                 return "more changes needed";
374             case NONE:
375                 return "comment added";
376             default:
377                 throw new RuntimeException("Unknown verdict: " + verdict);
378         }
379     }

394         var self = pr.repository().forge().currentUser();
395         return comments.stream()

397                        .filter(comment -> comment.body().contains(marker))

401     private String getMergeReadyComment(String commitMessage, List<Review> reviews) {
402         var message = new StringBuilder();
403         message.append("@");
404         message.append(pr.author().userName());
405         message.append(" This change now passes all automated pre-integration checks");
406 
407         try {
408             var hasContributingFile =
409                 !localRepo.files(pr.targetHash(), Path.of("CONTRIBUTING.md")).isEmpty();
410             if (hasContributingFile) {
411                 message.append(". When the change also fulfills all ");
412                 message.append("[project specific requirements](https://github.com/");
413                 message.append(pr.repository().name());
414                 message.append("/blob/");
415                 message.append(pr.targetRef());
416                 message.append("/CONTRIBUTING.md)");
417             }
418         } catch (IOException e) {
419             throw new UncheckedIOException(e);
420         }
421 
422         message.append(", type `/integrate` in a new comment to proceed. After integration, ");
423         message.append("the commit message will be:\n");
424         message.append("```\n");
425         message.append(commitMessage);
426         message.append("\n```\n");
427 
428         message.append("- If you would like to add a summary, use the `/summary` command.\n");
429         message.append("- To credit additional contributors, use the `/contributor` command.\n");
430         message.append("- To add additional solved issues, use the `/solves` command.\n");
431 
432         var divergingCommits = checkablePullRequest.divergingCommits();
433         if (divergingCommits.size() > 0) {
434             message.append("\n");
435             message.append("Since the source branch of this PR was last updated there ");
436             if (divergingCommits.size() == 1) {
437                 message.append("has been 1 commit ");
438             } else {
439                 message.append("have been ");
440                 message.append(divergingCommits.size());
441                 message.append(" commits ");
442             }
443             message.append("pushed to the `");
444             message.append(pr.targetRef());
445             message.append("` branch. ");









446             message.append("As there are no conflicts, your changes will automatically be rebased on top of ");
447             message.append("these commits when integrating. If you prefer to avoid automatic rebasing, please merge `");
448             message.append(pr.targetRef());
449             message.append("` into your branch, and then specify the current head hash when integrating, like this: ");
450             message.append("`/integrate ");
451             message.append(pr.targetHash());
452             message.append("`.\n");
453         } else {
454             message.append("\n");
455             message.append("There are currently no new commits on the `");
456             message.append(pr.targetRef());
457             message.append("` branch since the last update of the source branch of this PR. If another commit should be pushed before ");
458             message.append("you perform the `/integrate` command, your PR will be automatically rebased. If you would like to avoid ");
459             message.append("potential automatic rebasing, specify the current head hash when integrating, like this: ");
460             message.append("`/integrate ");
461             message.append(pr.targetHash());
462             message.append("`.\n");
463         }
464 
465         if (!ProjectPermissions.mayCommit(censusInstance, pr.author())) {
466             message.append("\n");
467             var contributor = censusInstance.namespace().get(pr.author().id());
468             if (contributor == null) {
469                 message.append("As you are not a known OpenJDK [Author](https://openjdk.java.net/bylaws#author), ");
470             } else {
471                 message.append("As you do not have Committer status in this project, ");
472             }
473 
474             message.append("an existing [Committer](https://openjdk.java.net/bylaws#committer) must agree to ");
475             message.append("[sponsor](https://openjdk.java.net/sponsor/) your change. ");
476             var candidates = reviews.stream()
477                                     .filter(review -> ProjectPermissions.mayCommit(censusInstance, review.reviewer()))
478                                     .map(review -> "@" + review.reviewer().userName())
479                                     .collect(Collectors.joining(", "));
480             if (candidates.length() > 0) {
481                 message.append("Possible candidates are the reviewers of this PR (");
482                 message.append(candidates);
483                 message.append(") but any other Committer may sponsor as well. ");
484             }
485             message.append("\n\n");
486             message.append("‚û°Ô∏è To flag this PR as ready for integration with the above commit message, type ");
487             message.append("`/integrate` in a new comment. (Afterwards, your sponsor types ");
488             message.append("`/sponsor` in a new comment to perform the integration).\n");
489         } else {
490             message.append("\n");
491             message.append("‚û°Ô∏è To integrate this PR with the above commit message to the `" + pr.targetRef() + "` branch, type ");
492             message.append("`/integrate` in a new comment.\n");
493         }
494         message.append(mergeReadyMarker);
495         return message.toString();
496     }

507     private void updateMergeReadyComment(boolean isReady, String commitMessage, List<Comment> comments, List<Review> reviews, boolean rebasePossible) {
508         var existing = findComment(comments, mergeReadyMarker);
509         if (isReady && rebasePossible) {
510             var message = getMergeReadyComment(commitMessage, reviews);
511             if (existing.isEmpty()) {
512                 pr.addComment(message);
513             } else {
514                 pr.updateComment(existing.get().id(), message);
515             }
516         } else {
517             existing.ifPresent(comment -> pr.updateComment(comment.id(), getMergeNoLongerReadyComment()));
518         }
519     }

521     private void addSourceBranchWarningComment(List<Comment> comments) throws IOException {
522         var existing = findComment(comments, sourceBranchWarningMarker);
523         if (existing.isPresent()) {
524             // Only add the comment once per PR
525             return;
526         }
527         var branch = pr.sourceRef();
528         var message = ":warning: @" + pr.author().userName() + " " +
529             "a branch with the same name as the source branch for this pull request (`" + branch + "`) " +
530             "is present in the [target repository](" + pr.repository().nonTransformedWebUrl() + "). " +
531             "If you eventually integrate this pull request then the branch `" + branch + "` " +
532             "in your [personal fork](" + pr.sourceRepository().nonTransformedWebUrl() + ") will diverge once you sync " +
533             "your personal fork with the upstream repository.\n" +
534             "\n" +
535             "To avoid this situation, create a new branch for your changes and reset the `" + branch + "` branch. " +
536             "You can do this by running the following commands in a local repository for your personal fork. " +
537             "_Note_: you do *not* have to name the new branch `NEW-BRANCH-NAME`." +
538             "\n" +
539             "```" +
540             "$ git checkout " + branch + "\n" +
541             "$ git checkout -b NEW-BRANCH-NAME\n" +
542             "$ git branch -f " + branch + " " + baseHash.hex() + "\n" +
543             "$ git push -f origin " + branch + "\n" +
544             "```\n" +
545             "\n" +
546             "Then proceed to create a new pull request with `NEW-BRANCH-NAME` as the source branch and " +
547             "close this one.\n" +
548             sourceBranchWarningMarker;
549         pr.addComment(message);
550     }

552     private void addOutdatedComment(List<Comment> comments) {
553         var existing = findComment(comments, outdatedHelpMarker);
554         if (existing.isPresent()) {
555             // Only add the comment once per PR
556             return;
557         }
558         var message = "@" + pr.author().userName() + " this pull request can not be integrated into " +
559                 "`" + pr.targetRef() + "` due to one or more merge conflicts. To resolve these merge conflicts " +
560                 "and update this pull request you can run the following commands in the local repository for your personal fork:\n" +
561                 "```bash\n" +
562                 "git checkout " + pr.sourceRef() + "\n" +
563                 "git fetch " + pr.repository().webUrl() + " " + pr.targetRef() + "\n" +
564                 "git merge FETCH_HEAD\n" +
565                 "# resolve conflicts and follow the instructions given by git merge\n" +
566                 "git commit -m \"Merge " + pr.targetRef() + "\"\n" +
567                 "git push\n" +
568                 "```\n" +
569                 outdatedHelpMarker;
570         pr.addComment(message);
571     }

573     private void checkStatus() {
574         var checkBuilder = CheckBuilder.create("jcheck", pr.headHash());
575         var censusDomain = censusInstance.configuration().census().domain();
576         Exception checkException = null;
577 
578         try {
579             // Post check in-progress
580             log.info("Starting to run jcheck on PR head");
581             pr.createCheck(checkBuilder.build());
582 
583             var ignored = new PrintWriter(new StringWriter());
584             var rebasePossible = true;
585             var commitHash = pr.headHash();
586             var mergedHash = checkablePullRequest.mergeTarget(ignored);
587             if (mergedHash.isPresent()) {
588                 commitHash = mergedHash.get();
589             } else {
590                 rebasePossible = false;
591             }
592 
593             List<String> additionalErrors = List.of();
594             Hash localHash;
595             try {
596                 localHash = checkablePullRequest.commit(commitHash, censusInstance.namespace(), censusDomain, null);
597             } catch (CommitFailure e) {
598                 additionalErrors = List.of(e.getMessage());
599                 localHash = baseHash;
600             }
601             PullRequestCheckIssueVisitor visitor = checkablePullRequest.createVisitor(localHash, censusInstance);
602             if (!localHash.equals(baseHash)) {
603                 // Determine current status
604                 var additionalConfiguration = AdditionalConfiguration.get(localRepo, localHash, pr.repository().forge().currentUser(), comments);
605                 checkablePullRequest.executeChecks(localHash, censusInstance, visitor, additionalConfiguration);
606                 additionalErrors = botSpecificChecks(localHash);
607             } else {
608                 if (additionalErrors.isEmpty()) {
609                     additionalErrors = List.of("This PR contains no changes");
610                 }
611             }
612             updateCheckBuilder(checkBuilder, visitor, additionalErrors);
613             updateReadyForReview(visitor, additionalErrors);
614 
615             // Calculate and update the status message if needed
616             var statusMessage = getStatusMessage(comments, activeReviews, visitor, additionalErrors);
617             var updatedBody = updateStatusMessage(statusMessage);
618 
619             // Post / update approval messages (only needed if the review itself can't contain a body)
620             if (!pr.repository().forge().supportsReviewBody()) {
621                 updateReviewedMessages(comments, allReviews);
622             }
623 
624             var commit = localRepo.lookup(localHash).orElseThrow();
625             var commitMessage = String.join("\n", commit.message());
626             var readyForIntegration = visitor.getMessages().isEmpty() && additionalErrors.isEmpty();
627             updateMergeReadyComment(readyForIntegration, commitMessage, comments, activeReviews, rebasePossible);
628             if (readyForIntegration && rebasePossible) {
629                 newLabels.add("ready");
630             } else {
631                 newLabels.remove("ready");
632             }
633             if (!rebasePossible) {
634                 if (!labels.contains("failed-auto-merge")) {
635                     addOutdatedComment(comments);
636                 }
637                 newLabels.add("merge-conflict");
638             } else {
639                 newLabels.remove("merge-conflict");
640             }
641 
642             var branchNames = pr.repository().branches().stream().map(HostedBranch::name).collect(Collectors.toSet());
643             if (!pr.repository().url().equals(pr.sourceRepository().url()) && branchNames.contains(pr.sourceRef())) {
644                 addSourceBranchWarningComment(comments);
645             }
646 
647             // Ensure that the ready for sponsor label is up to date
648             newLabels.remove("sponsor");
649             var readyHash = ReadyForSponsorTracker.latestReadyForSponsor(pr.repository().forge().currentUser(), comments);
650             if (readyHash.isPresent() && readyForIntegration) {
651                 var acceptedHash = readyHash.get();
652                 if (pr.headHash().equals(acceptedHash)) {
653                     newLabels.add("sponsor");
654                 }
655             }
656 
657             // Calculate current metadata to avoid unnecessary future checks
658             var metadata = workItem.getMetadata(pr.title(), updatedBody, pr.comments(), activeReviews, newLabels,
659                                                 censusInstance, pr.targetHash(), pr.isDraft());
660             checkBuilder.metadata(metadata);
661         } catch (Exception e) {
662             log.throwing("CommitChecker", "checkStatus", e);
663             newLabels.remove("ready");
664             checkBuilder.metadata("invalid");
665             checkBuilder.title("Exception occurred during jcheck - the operation will be retried");
666             checkBuilder.summary(e.getMessage());
667             checkBuilder.complete(false);
668             checkException = e;
669         }
670         var check = checkBuilder.build();
671         pr.updateCheck(check);
672 
673         // Synchronize the wanted set of labels
674         for (var newLabel : newLabels) {
675             if (!labels.contains(newLabel)) {
676                 pr.addLabel(newLabel);
677             }
678         }
679         for (var oldLabel : labels) {
680             if (!newLabels.contains(oldLabel)) {
681                 pr.removeLabel(oldLabel);
682             }
683         }
684 
685         // After updating the PR, rethrow any exception to automatically retry on transient errors
686         if (checkException != null) {
687             throw new RuntimeException("Exception during jcheck", checkException);
688         }
689     }
690 }
