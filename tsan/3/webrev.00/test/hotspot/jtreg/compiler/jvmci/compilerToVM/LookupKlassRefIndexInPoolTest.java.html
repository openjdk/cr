 64 /**
 65  * Test for {@code jdk.vm.ci.hotspot.CompilerToVM.lookupKlassRefIndexInPool} method
 66  */
 67 public class LookupKlassRefIndexInPoolTest {
 68 
 69     public static void main(String[] args) throws Exception {
 70         Map<ConstantTypes, Validator> typeTests = new HashMap<>();
 71         typeTests.put(CONSTANT_METHODREF, LookupKlassRefIndexInPoolTest::validate);
 72         typeTests.put(CONSTANT_INTERFACEMETHODREF, LookupKlassRefIndexInPoolTest::validate);
 73         typeTests.put(CONSTANT_FIELDREF, LookupKlassRefIndexInPoolTest::validate);
 74         ConstantPoolTestCase testCase = new ConstantPoolTestCase(typeTests);
 75         testCase.test();
 76         // The next "Class.forName" and repeating "testCase.test()"
 77         // are here for the following reason.
 78         // The first test run is without dummy class initialization,
 79         // which means no constant pool cache exists.
 80         // The second run is with initialized class (with constant pool cache available).
 81         // Some CompilerToVM methods require different input
 82         // depending on whether CP cache exists or not.
 83         for (DummyClasses dummy : DummyClasses.values()) {
 84             Class.forName(dummy.klass.getName());
 85         }
 86         testCase.test();
 87     }
 88 
 89     private static void validate(ConstantPool constantPoolCTVM,
 90                                  ConstantTypes cpType,
 91                                  DummyClasses dummyClass,
 92                                  int cpi) {
 93         TestedCPEntry entry = cpType.getTestedCPEntry(dummyClass, cpi);
 94         if (entry == null) {
 95             return;
 96         }
 97         int index = cpi;
 98         String cached = "";
 99         int cpci = dummyClass.getCPCacheIndex(cpi);
100         if (cpci != ConstantPoolTestsHelper.NO_CP_CACHE_PRESENT) {
101             index = cpci;
102             cached = "cached ";
103         }
104         int indexToVerify = CompilerToVMHelper.lookupKlassRefIndexInPool(constantPoolCTVM, index);
105         int indexToRefer = dummyClass.constantPoolSS.getClassRefIndexAt(cpi);
106         String msg = String.format("Wrong class index returned by lookupKlassRefIndexInPool method "
107                                            + "applied to %sconstant pool index %d",
108                                    cached,
109                                    index);
110         Asserts.assertEQ(indexToRefer, indexToVerify, msg);
111     }
