24 /* @test
25    @bug 4173717
26    @summary Make sure that passing 0 in setReceiveBufferSize will throw
27    IllegalArgumentException
28    */
29 import java.net.*;
30 
31 public class SetReceiveBufferSize {
32 
33   public static void main(String args[]) throws Exception {
34     boolean error = true;
35      DatagramSocket soc = null;
36 
37      try {
38        soc = new DatagramSocket();
39        soc.setReceiveBufferSize(0);
40      } catch(IllegalArgumentException e) {
41        error = false;
42      }
43 
44      if (error) {
45        throw new RuntimeException("Test with 0 buffer size failed!");
46      }
47   }
48 }
