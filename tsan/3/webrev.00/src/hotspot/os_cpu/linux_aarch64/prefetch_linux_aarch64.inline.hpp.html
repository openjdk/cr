26 #ifndef OS_CPU_LINUX_AARCH64_PREFETCH_LINUX_AARCH64_INLINE_HPP
27 #define OS_CPU_LINUX_AARCH64_PREFETCH_LINUX_AARCH64_INLINE_HPP
28 
29 #include "runtime/prefetch.hpp"
30 
31 
32 inline void Prefetch::read (void *loc, intx interval) {
33   if (interval >= 0)
34     asm("prfm PLDL1KEEP, [%0, %1]" : : "r"(loc), "r"(interval));
35 }
36 
37 inline void Prefetch::write(void *loc, intx interval) {
38   if (interval >= 0)
39     asm("prfm PSTL1KEEP, [%0, %1]" : : "r"(loc), "r"(interval));
40 }
41 
42 #endif // OS_CPU_LINUX_AARCH64_PREFETCH_LINUX_AARCH64_INLINE_HPP
