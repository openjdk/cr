28 /**
29  * Signals that an account was locked.
30  *
31  * <p> This exception may be thrown by a LoginModule if it
32  * determines that authentication is being attempted on a
33  * locked account.
34  *
35  * @since 1.5
36  */
37 public class AccountLockedException extends AccountException {
38 

39     private static final long serialVersionUID = 8280345554014066334L;
40 
41     /**
42      * Constructs a AccountLockedException with no detail message.
43      * A detail message is a String that describes this particular exception.
44      */
45     public AccountLockedException() {
46         super();
47     }
48 
49     /**
50      * Constructs a AccountLockedException with the specified
51      * detail message. A detail message is a String that describes
52      * this particular exception.
53      *
54      * @param msg the detail message.
55      */
56     public AccountLockedException(String msg) {
57         super(msg);
58     }
