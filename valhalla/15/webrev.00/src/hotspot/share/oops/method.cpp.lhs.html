  77 Method* Method::allocate(ClassLoaderData* loader_data,
  78                          int byte_code_size,
  79                          AccessFlags access_flags,
  80                          InlineTableSizes* sizes,
  81                          ConstMethod::MethodType method_type,
  82                          TRAPS) {
  83   assert(!access_flags.is_native() || byte_code_size == 0,
  84          "native methods should not contain byte codes");
  85   ConstMethod* cm = ConstMethod::allocate(loader_data,
  86                                           byte_code_size,

  88                                           method_type,
  89                                           CHECK_NULL);
  90   int size = Method::size(access_flags.is_native());
  91   return new (loader_data, size, MetaspaceObj::MethodType, THREAD) Method(cm, access_flags);

 181 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature) {
 182   const char* klass_name = klass->external_name();
 183   int klass_name_len  = (int)strlen(klass_name);
 184   int method_name_len = method_name->utf8_length();
 185   int len             = klass_name_len + 1 + method_name_len + signature->utf8_length();
 186   char* dest          = NEW_RESOURCE_ARRAY(char, len + 1);
 187   strcpy(dest, klass_name);
 188   dest[klass_name_len] = '.';
 189   strcpy(&dest[klass_name_len + 1], method_name->as_C_string());
 190   strcpy(&dest[klass_name_len + 1 + method_name_len], signature->as_C_string());
 191   dest[len] = 0;
 192   return dest;
 193 }

 233 int Method::fast_exception_handler_bci_for(const methodHandle& mh, Klass* ex_klass, int throw_bci, TRAPS) {
 234   // exception table holds quadruple entries of the form (beg_bci, end_bci, handler_bci, klass_index)
 235   // access exception table
 236   ExceptionTable table(mh());
 237   int length = table.length();
 238   // iterate through all entries sequentially
 239   constantPoolHandle pool(THREAD, mh->constants());
 240   for (int i = 0; i < length; i ++) {
 241     //reacquire the table in case a GC happened
 242     ExceptionTable table(mh());
 243     int beg_bci = table.start_pc(i);
 244     int end_bci = table.end_pc(i);
 245     assert(beg_bci <= end_bci, "inconsistent exception table");
 246     if (beg_bci <= throw_bci && throw_bci < end_bci) {
 247       // exception handler bci range covers throw_bci => investigate further
 248       int handler_bci = table.handler_pc(i);
 249       int klass_index = table.catch_type_index(i);
 250       if (klass_index == 0) {
 251         return handler_bci;
 252       } else if (ex_klass == NULL) {
 253         return handler_bci;
 254       } else {
 255         // we know the exception class => get the constraint class
 256         // this may require loading of the constraint class; if verification
 257         // fails or some other exception occurs, return handler_bci
 258         Klass* k = pool->klass_at(klass_index, CHECK_(handler_bci));
 259         assert(k != NULL, "klass not loaded");
 260         if (ex_klass->is_subtype_of(k)) {
 261           return handler_bci;
 262         }
 263       }
 264     }
 265   }
 266 
 267   return -1;
 268 }

 270 void Method::mask_for(int bci, InterpreterOopMap* mask) {
 271   methodHandle h_this(Thread::current(), this);
 272   // Only GC uses the OopMapCache during thread stack root scanning
 273   // any other uses generate an oopmap but do not save it in the cache.
 274   if (Universe::heap()->is_gc_active()) {
 275     method_holder()->mask_for(h_this, bci, mask);
 276   } else {
 277     OopMapCache::compute_one_oop_map(h_this, bci, mask);
 278   }
 279   return;
 280 }

 303 // Return bci if it appears to be a valid bcp
 304 // Return -1 otherwise.
 305 // Used by profiling code, when invalid data is a possibility.
 306 // The caller is responsible for validating the Method* itself.
 307 int Method::validate_bci_from_bcp(address bcp) const {
 308   // keep bci as -1 if not a valid bci
 309   int bci = -1;
 310   if (bcp == 0 || bcp == code_base()) {
 311     // code_size() may return 0 and we allow 0 here
 312     // the method may be native
 313     bci = 0;
 314   } else if (contains(bcp)) {
 315     bci = bcp - code_base();
 316   }
 317   // Assert that if we have dodged any asserts, bci is negative.
 318   assert(bci == -1 || bci == bci_from(bcp_from(bci)), "sane bci if >=0");
 319   return bci;
 320 }

 322 address Method::bcp_from(int bci) const {
 323   assert((is_native() && bci == 0) || (!is_native() && 0 <= bci && bci < code_size()),
 324          "illegal bci: %d for %s method", bci, is_native() ? "native" : "non-native");
 325   address bcp = code_base() + bci;
 326   assert(is_native() && bcp == code_base() || contains(bcp), "bcp doesn't belong to this method");
 327   return bcp;
 328 }

 399 // The RegisterNatives call being attempted tried to register with a method that
 400 // is not native.  Ask JVM TI what prefixes have been specified.  Then check
 401 // to see if the native method is now wrapped with the prefixes.  See the
 402 // SetNativeMethodPrefix(es) functions in the JVM TI Spec for details.
 403 static Method* find_prefixed_native(Klass* k, Symbol* name, Symbol* signature, TRAPS) {
 404 #if INCLUDE_JVMTI
 405   ResourceMark rm(THREAD);
 406   Method* method;
 407   int name_len = name->utf8_length();
 408   char* name_str = name->as_utf8();
 409   int prefix_count;
 410   char** prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
 411   for (int i = 0; i < prefix_count; i++) {
 412     char* prefix = prefixes[i];
 413     int prefix_len = (int)strlen(prefix);
 414 
 415     // try adding this prefix to the method name and see if it matches another method name
 416     int trial_len = name_len + prefix_len;
 417     char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
 418     strcpy(trial_name_str, prefix);
 419     strcat(trial_name_str, name_str);
 420     TempNewSymbol trial_name = SymbolTable::probe(trial_name_str, trial_len);
 421     if (trial_name == NULL) {
 422       continue; // no such symbol, so this prefix wasn't used, try the next prefix
 423     }
 424     method = k->lookup_method(trial_name, signature);
 425     if (method == NULL) {
 426       continue; // signature doesn't match, try the next prefix
 427     }
 428     if (method->is_native()) {
 429       method->set_is_prefixed_native();
 430       return method; // wahoo, we found a prefixed version of the method, return it
 431     }
 432     // found as non-native, so prefix is good, add it, probably just need more prefixes
 433     name_len = trial_len;
 434     name_str = trial_name_str;
 435   }
 436 #endif // INCLUDE_JVMTI
 437   return NULL; // not found
 438 }

 440 bool Method::register_native(Klass* k, Symbol* name, Symbol* signature, address entry, TRAPS) {
 441   Method* method = k->lookup_method(name, signature);
 442   if (method == NULL) {
 443     ResourceMark rm(THREAD);
 444     stringStream st;
 445     st.print("Method '");
 446     print_external_name(&st, k, name, signature);
 447     st.print("' name or signature does not match");
 448     THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 449   }
 450   if (!method->is_native()) {
 451     // trying to register to a non-native method, see if a JVM TI agent has added prefix(es)
 452     method = find_prefixed_native(k, name, signature, THREAD);
 453     if (method == NULL) {
 454       ResourceMark rm(THREAD);
 455       stringStream st;
 456       st.print("Method '");
 457       print_external_name(&st, k, name, signature);
 458       st.print("' is not declared as native");
 459       THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 460     }
 461   }
 462 
 463   if (entry != NULL) {
 464     method->set_native_function(entry, native_bind_event_is_interesting);
 465   } else {
 466     method->clear_native_function();
 467   }
 468   if (log_is_enabled(Debug, jni, resolve)) {
 469     ResourceMark rm(THREAD);
 470     log_debug(jni, resolve)("[Registering JNI native method %s.%s]",
 471                             method->method_holder()->external_name(),
 472                             method->name()->as_C_string());
 473   }
 474   return true;
 475 }

 477 bool Method::was_executed_more_than(int n) {
 478   // Invocation counter is reset when the Method* is compiled.
 479   // If the method has compiled code we therefore assume it has
 480   // be excuted more than n times.
 481   if (is_accessor() || is_empty_method() || (code() != NULL)) {
 482     // interpreter doesn't bump invocation counter of trivial methods
 483     // compiler does not bump invocation counter of compiled methods
 484     return true;
 485   }
 486   else if ((method_counters() != NULL &&
 487             method_counters()->invocation_counter()->carry()) ||
 488            (method_data() != NULL &&
 489             method_data()->invocation_counter()->carry())) {
 490     // The carry bit is set when the counter overflows and causes
 491     // a compilation to occur.  We don't know how many times
 492     // the counter has been reset, so we simply assume it has
 493     // been executed more than n times.
 494     return true;
 495   } else {
 496     return invocation_count() > n;
 497   }
 498 }

 500 void Method::print_invocation_count() {
 501   if (is_static()) tty->print("static ");
 502   if (is_final()) tty->print("final ");
 503   if (is_synchronized()) tty->print("synchronized ");
 504   if (is_native()) tty->print("native ");
 505   tty->print("%s::", method_holder()->external_name());
 506   name()->print_symbol_on(tty);
 507   signature()->print_symbol_on(tty);
 508 
 509   if (WizardMode) {
 510     // dump the size of the byte codes
 511     tty->print(" {%d}", code_size());
 512   }
 513   tty->cr();
 514 
 515   tty->print_cr ("  interpreter_invocation_count: %8d ", interpreter_invocation_count());
 516   tty->print_cr ("  invocation_counter:           %8d ", invocation_count());
 517   tty->print_cr ("  backedge_counter:             %8d ", backedge_count());
 518 #ifndef PRODUCT
 519   if (CountCompiledCalls) {
 520     tty->print_cr ("  compiled_invocation_count: %8d ", compiled_invocation_count());
 521   }
 522 #endif
 523 }

 525 // Build a MethodData* object to hold information about this method
 526 // collected in the interpreter.
 527 void Method::build_interpreter_method_data(const methodHandle& method, TRAPS) {
 528   // Do not profile the method if metaspace has hit an OOM previously
 529   // allocating profiling data. Callers clear pending exception so don't
 530   // add one here.
 531   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 532     return;
 533   }
 534 
 535   // Grab a lock here to prevent multiple
 536   // MethodData*s from being created.
 537   MutexLocker ml(THREAD, MethodData_lock);
 538   if (method->method_data() == NULL) {
 539     ClassLoaderData* loader_data = method->method_holder()->class_loader_data();
 540     MethodData* method_data = MethodData::allocate(loader_data, method, THREAD);
 541     if (HAS_PENDING_EXCEPTION) {
 542       CompileBroker::log_metaspace_failure();
 543       ClassLoaderDataGraph::set_metaspace_oom(true);
 544       return;   // return the exception (which is cleared)
 545     }
 546 
 547     method->set_method_data(method_data);
 548     if (PrintMethodData && (Verbose || WizardMode)) {
 549       ResourceMark rm(THREAD);
 550       tty->print("build_interpreter_method_data for ");
 551       method->print_name(tty);
 552       tty->cr();
 553       // At the end of the run, the MDO, full of data, will be dumped.
 554     }
 555   }
 556 }

 558 MethodCounters* Method::build_method_counters(Method* m, TRAPS) {
 559   // Do not profile the method if metaspace has hit an OOM previously
 560   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 561     return NULL;
 562   }
 563 
 564   methodHandle mh(THREAD, m);
 565   MethodCounters* counters = MethodCounters::allocate(mh, THREAD);
 566   if (HAS_PENDING_EXCEPTION) {
 567     CompileBroker::log_metaspace_failure();
 568     ClassLoaderDataGraph::set_metaspace_oom(true);
 569     return NULL;   // return the exception (which is cleared)
 570   }
 571   if (!mh->init_method_counters(counters)) {
 572     MetadataFactory::free_metadata(mh->method_holder()->class_loader_data(), counters);
 573   }
 574 
 575   if (LogTouchedMethods) {
 576     mh->log_touched(CHECK_NULL);
 577   }
 578 
 579   return mh->method_counters();
 580 }

 627 bool Method::is_vanilla_constructor() const {
 628   // Returns true if this method is a vanilla constructor, i.e. an "<init>" "()V" method
 629   // which only calls the superclass vanilla constructor and possibly does stores of
 630   // zero constants to local fields:
 631   //
 632   //   aload_0
 633   //   invokespecial
 634   //   indexbyte1
 635   //   indexbyte2
 636   //
 637   // followed by an (optional) sequence of:
 638   //
 639   //   aload_0
 640   //   aconst_null / iconst_0 / fconst_0 / dconst_0
 641   //   putfield
 642   //   indexbyte1
 643   //   indexbyte2
 644   //
 645   // followed by:
 646   //
 647   //   return
 648 
 649   assert(name() == vmSymbols::object_initializer_name(),    "Should only be called for default constructors");
 650   assert(signature() == vmSymbols::void_method_signature(), "Should only be called for default constructors");
 651   int size = code_size();
 652   // Check if size match
 653   if (size == 0 || size % 5 != 0) return false;
 654   address cb = code_base();
 655   int last = size - 1;
 656   if (cb[0] != Bytecodes::_aload_0 || cb[1] != Bytecodes::_invokespecial || cb[last] != Bytecodes::_return) {
 657     // Does not call superclass default constructor
 658     return false;
 659   }
 660   // Check optional sequence
 661   for (int i = 4; i < last; i += 5) {
 662     if (cb[i] != Bytecodes::_aload_0) return false;
 663     if (!Bytecodes::is_zero_const(Bytecodes::cast(cb[i+1]))) return false;
 664     if (cb[i+2] != Bytecodes::_putfield) return false;
 665   }
 666   return true;
 667 }

 671   BytecodeStream bcs(methodHandle(Thread::current(), this));
 672   Bytecodes::Code bc;
 673

 677       case Bytecodes::_ifnull:
 678       case Bytecodes::_iflt:
 679       case Bytecodes::_ifle:
 680       case Bytecodes::_ifne:
 681       case Bytecodes::_ifnonnull:
 682       case Bytecodes::_ifgt:
 683       case Bytecodes::_ifge:
 684       case Bytecodes::_if_icmpeq:
 685       case Bytecodes::_if_icmpne:
 686       case Bytecodes::_if_icmplt:
 687       case Bytecodes::_if_icmpgt:
 688       case Bytecodes::_if_icmple:
 689       case Bytecodes::_if_icmpge:
 690       case Bytecodes::_if_acmpeq:
 691       case Bytecodes::_if_acmpne:

 704     }

 860 int Method::line_number_from_bci(int bci) const {
 861   int best_bci  =  0;
 862   int best_line = -1;
 863   if (bci == SynchronizationEntryBCI) bci = 0;
 864   if (0 <= bci && bci < code_size() && has_linenumber_table()) {
 865     // The line numbers are a short array of 2-tuples [start_pc, line_number].
 866     // Not necessarily sorted and not necessarily one-to-one.
 867     CompressedLineNumberReadStream stream(compressed_linenumber_table());
 868     while (stream.read_pair()) {
 869       if (stream.bci() == bci) {
 870         // perfect match
 871         return stream.line();
 872       } else {
 873         // update best_bci/line
 874         if (stream.bci() < bci && stream.bci() >= best_bci) {
 875           best_bci  = stream.bci();
 876           best_line = stream.line();
 877         }
 878       }
 879     }
 880   }
 881   return best_line;
 882 }

 910 void Method::set_native_function(address function, bool post_event_flag) {
 911   assert(function != NULL, "use clear_native_function to unregister natives");
 912   assert(!is_method_handle_intrinsic() || function == SharedRuntime::native_method_throw_unsatisfied_link_error_entry(), "");
 913   address* native_function = native_function_addr();
 914 
 915   // We can see racers trying to place the same native function into place. Once
 916   // is plenty.
 917   address current = *native_function;
 918   if (current == function) return;
 919   if (post_event_flag && JvmtiExport::should_post_native_method_bind() &&
 920       function != NULL) {
 921     // native_method_throw_unsatisfied_link_error_entry() should only
 922     // be passed when post_event_flag is false.
 923     assert(function !=
 924       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
 925       "post_event_flag mis-match");
 926 
 927     // post the bind event, and possible change the bind function
 928     JvmtiExport::post_native_method_bind(this, &function);
 929   }
 930   *native_function = function;
 931   // This function can be called more than once. We must make sure that we always
 932   // use the latest registered method -> check if a stub already has been generated.
 933   // If so, we have to make it not_entrant.
 934   CompiledMethod* nm = code(); // Put it into local variable to guard against concurrent updates
 935   if (nm != NULL) {
 936     nm->make_not_entrant();
 937   }
 938 }

1158 //
1159 // _adapter_trampoline points to a fixed location in the RW section of
1160 // the CDS archive. This location initially contains a NULL pointer. When the
1161 // first of method A or B is linked, an AdapterHandlerEntry is allocated
1162 // dynamically, and its c2i/i2c entries are generated.
1163 //
1164 // _i2i_entry and _from_interpreted_entry initially points to the same
1165 // (fixed) location in the CODE section of the CDS archive. This contains
1166 // an unconditional branch to the actual entry for "zero_locals", which is
1167 // generated at run time and may be on an arbitrary address. Thus, the
1168 // unconditional branch is also generated at run time to jump to the correct
1169 // address.
1170 //
1171 // Similarly, _from_compiled_entry points to a fixed address in the CODE
1172 // section. This address has enough space for an unconditional branch
1173 // instruction, and is initially zero-filled. After the AdapterHandlerEntry is
1174 // initialized, and the address for the actual c2i_entry is known, we emit a
1175 // branch instruction here to branch to the actual c2i_entry.
1176 //
1177 // The effect of the extra branch on the i2i and c2i entries is negligible.
1178 //
1179 // The reason for putting _adapter_trampoline in RO is many shared Methods
1180 // share the same AdapterHandlerEntry, so we can save space in the RW section
1181 // by having the extra indirection.
1182 
1183 
1184 [Method A: RW]
1185   _constMethod ----> [ConstMethod: RO]
1186                        _adapter_trampoline -----------+
1187                                                       |
1188   _i2i_entry              (same value as method B)    |
1189   _from_interpreted_entry (same value as method B)    |
1190   _from_compiled_entry    (same value as method B)    |
1191                                                       |
1192                                                       |
1193 [Method B: RW]                               +--------+
1194   _constMethod ----> [ConstMethod: RO]       |
1195                        _adapter_trampoline --+--->(AdapterHandlerEntry* ptr: RW)-+
1196                                                                                  |
1197                                                  +-------------------------------+
1198                                                  |
1199                                                  +----> [AdapterHandlerEntry] (allocated at run time)
1200                                                               _fingerprint
1201                                                               _c2i_entry ---------------------------------+->[c2i entry..]
1202  _i2i_entry  -------------+                                   _i2c_entry ---------------+-> [i2c entry..] |
1203  _from_interpreted_entry  |                                   _c2i_unverified_entry     |                 |
1204          |                |                                   _c2i_no_clinit_check_entry|                 |
1205          |                |  (_cds_entry_table: CODE)                                   |                 |
1206          |                +->[0]: jmp _entry_table[0] --> (i2i_entry_for "zero_locals") |                 |
1207          |                |                               (allocated at run time)       |                 |
1208          |                |  ...                           [asm code ...]               |                 |
1209          +-[not compiled]-+  [n]: jmp _entry_table[n]                                   |                 |
1210          |                                                                              |                 |
1211          |                                                                              |                 |
1212          +-[compiled]-------------------------------------------------------------------+                 |
1213                                                                                                           |
1214  _from_compiled_entry------------>  (_c2i_entry_trampoline: CODE)                                         |
1215                                     [jmp c2i_entry] ------------------------------------------------------+
1216 
1217 ***/

1219 // Called when the method_holder is getting linked. Setup entrypoints so the method
1220 // is ready to be called from interpreter, compiler, and vtables.
1221 void Method::link_method(const methodHandle& h_method, TRAPS) {
1222   // If the code cache is full, we may reenter this function for the
1223   // leftover methods that weren't linked.
1224   if (is_shared()) {

1225     address entry = Interpreter::entry_for_cds_method(h_method);
1226     assert(entry != NULL && entry == _i2i_entry,
1227            "should be correctly set during dump time");

1228     if (adapter() != NULL) {
1229       return;
1230     }
1231     assert(entry == _from_interpreted_entry,
1232            "should be correctly set during dump time");
1233   } else if (_i2i_entry != NULL) {
1234     return;
1235   }
1236   assert( _code == NULL, "nothing compiled yet" );
1237 
1238   // Setup interpreter entrypoint
1239   assert(this == h_method(), "wrong h_method()" );
1240 
1241   if (!is_shared()) {
1242     assert(adapter() == NULL, "init'd to NULL");
1243     address entry = Interpreter::entry_for_method(h_method);
1244     assert(entry != NULL, "interpreter entry must be non-null");
1245     // Sets both _i2i_entry and _from_interpreted_entry
1246     set_interpreter_entry(entry);
1247   }
1248 
1249   // Don't overwrite already registered native entries.
1250   if (is_native() && !has_native_function()) {
1251     set_native_function(
1252       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
1253       !native_bind_event_is_interesting);
1254   }
1255 
1256   // Setup compiler entrypoint.  This is made eagerly, so we do not need
1257   // special handling of vtables.  An alternative is to make adapters more
1258   // lazily by calling make_adapter() from from_compiled_entry() for the
1259   // normal calls.  For vtable calls life gets more complicated.  When a
1260   // call-site goes mega-morphic we need adapters in all methods which can be
1261   // called from the vtable.  We need adapters on such methods that get loaded
1262   // later.  Ditto for mega-morphic itable calls.  If this proves to be a
1263   // problem we'll make these lazily later.
1264   (void) make_adapters(h_method, CHECK);
1265 
1266   // ONLY USE the h_method now as make_adapter may have blocked
1267 
1268 }

1270 address Method::make_adapters(const methodHandle& mh, TRAPS) {
1271   // Adapters for compiled code are made eagerly here.  They are fairly
1272   // small (generally < 100 bytes) and quick to make (and cached and shared)
1273   // so making them eagerly shouldn't be too expensive.
1274   AdapterHandlerEntry* adapter = AdapterHandlerLibrary::get_adapter(mh);
1275   if (adapter == NULL ) {
1276     if (!is_init_completed()) {
1277       // Don't throw exceptions during VM initialization because java.lang.* classes
1278       // might not have been initialized, causing problems when constructing the
1279       // Java exception object.
1280       vm_exit_during_initialization("Out of space in CodeCache for adapters");
1281     } else {
1282       THROW_MSG_NULL(vmSymbols::java_lang_VirtualMachineError(), "Out of space in CodeCache for adapters");
1283     }
1284   }
1285 
1286   if (mh->is_shared()) {
1287     assert(mh->adapter() == adapter, "must be");
1288     assert(mh->_from_compiled_entry != NULL, "must be");
1289     assert(mh->_from_compiled_value_entry != NULL, "must be");
1290     assert(mh->_from_compiled_value_ro_entry != NULL, "must be");

1292     mh->set_adapter_entry(adapter);
1293     mh->_from_compiled_entry = adapter->get_c2i_entry();

1295     mh->_from_compiled_value_ro_entry = adapter->get_c2i_value_ro_entry();
1296   }
1297   return adapter->get_c2i_entry();
1298 }

1377 // Install compiled code.  Instantly it can execute.
1378 void Method::set_code(const methodHandle& mh, CompiledMethod *code) {
1379   assert_lock_strong(CompiledMethod_lock);
1380   assert( code, "use clear_code to remove code" );
1381   assert( mh->check_code(), "" );
1382 
1383   guarantee(mh->adapter() != NULL, "Adapter blob must already exist!");
1384 
1385   // These writes must happen in this order, because the interpreter will
1386   // directly jump to from_interpreted_entry which jumps to an i2c adapter
1387   // which jumps to _from_compiled_entry.
1388   mh->_code = code;             // Assign before allowing compiled code to exec
1389 
1390   int comp_level = code->comp_level();
1391   // In theory there could be a race here. In practice it is unlikely
1392   // and not worth worrying about.
1393   if (comp_level > mh->highest_comp_level()) {
1394     mh->set_highest_comp_level(comp_level);
1395   }
1396 
1397   OrderAccess::storestore();
1398   mh->_from_compiled_entry = code->verified_entry_point();
1399   mh->_from_compiled_value_entry = code->verified_value_entry_point();
1400   mh->_from_compiled_value_ro_entry = code->verified_value_ro_entry_point();
1401   OrderAccess::storestore();
1402   // Instantly compiled code can execute.
1403   if (!mh->is_method_handle_intrinsic())
1404     mh->_from_interpreted_entry = mh->get_i2c_entry();
1405 }

1496 // Make an instance of a signature-polymorphic internal MH primitive.
1497 methodHandle Method::make_method_handle_intrinsic(vmIntrinsics::ID iid,
1498                                                          Symbol* signature,
1499                                                          TRAPS) {
1500   ResourceMark rm(THREAD);
1501   methodHandle empty;
1502 
1503   InstanceKlass* holder = SystemDictionary::MethodHandle_klass();
1504   Symbol* name = MethodHandles::signature_polymorphic_intrinsic_name(iid);
1505   assert(iid == MethodHandles::signature_polymorphic_name_id(name), "");
1506   if (TraceMethodHandles) {
1507     tty->print_cr("make_method_handle_intrinsic MH.%s%s", name->as_C_string(), signature->as_C_string());
1508   }
1509 
1510   // invariant:   cp->symbol_at_put is preceded by a refcount increment (more usually a lookup)
1511   name->increment_refcount();
1512   signature->increment_refcount();
1513 
1514   int cp_length = _imcp_limit;
1515   ClassLoaderData* loader_data = holder->class_loader_data();
1516   constantPoolHandle cp;
1517   {
1518     ConstantPool* cp_oop = ConstantPool::allocate(loader_data, cp_length, CHECK_(empty));
1519     cp = constantPoolHandle(THREAD, cp_oop);
1520   }
1521   cp->set_pool_holder(holder);
1522   cp->symbol_at_put(_imcp_invoke_name,       name);
1523   cp->symbol_at_put(_imcp_invoke_signature,  signature);
1524   cp->set_has_preresolution();
1525 
1526   // decide on access bits:  public or not?
1527   int flags_bits = (JVM_ACC_NATIVE | JVM_ACC_SYNTHETIC | JVM_ACC_FINAL);
1528   bool must_be_static = MethodHandles::is_signature_polymorphic_static(iid);
1529   if (must_be_static)  flags_bits |= JVM_ACC_STATIC;
1530   assert((flags_bits & JVM_ACC_PUBLIC) == 0, "do not expose these methods");
1531 
1532   methodHandle m;
1533   {
1534     InlineTableSizes sizes;
1535     Method* m_oop = Method::allocate(loader_data, 0,
1536                                      accessFlags_from(flags_bits), &sizes,
1537                                      ConstMethod::NORMAL, CHECK_(empty));
1538     m = methodHandle(THREAD, m_oop);
1539   }
1540   m->set_constants(cp());
1541   m->set_name_index(_imcp_invoke_name);
1542   m->set_signature_index(_imcp_invoke_signature);
1543   assert(MethodHandles::is_signature_polymorphic_name(m->name()), "");
1544   assert(m->signature() == signature, "");
1545   m->compute_from_signature(signature);
1546   m->init_intrinsic_id();
1547   assert(m->is_method_handle_intrinsic(), "");
1548 #ifdef ASSERT
1549   if (!MethodHandles::is_signature_polymorphic(m->intrinsic_id()))  m->print();
1550   assert(MethodHandles::is_signature_polymorphic(m->intrinsic_id()), "must be an invoker");
1551   assert(m->intrinsic_id() == iid, "correctly predicted iid");
1552 #endif //ASSERT
1553 
1554   // Finally, set up its entry points.
1555   assert(m->can_be_statically_bound(), "");
1556   m->set_vtable_index(Method::nonvirtual_vtable_index);
1557   m->link_method(m, CHECK_(empty));
1558 
1559   if (TraceMethodHandles && (Verbose || WizardMode)) {
1560     ttyLocker ttyl;
1561     m->print_on(tty);
1562   }
1563 
1564   return m;
1565 }

1578                                                 u_char* new_compressed_linenumber_table, int new_compressed_linenumber_size, TRAPS) {

1587   int exception_table_len = cm->exception_table_length();

1590   int parameter_annotations_len = cm->parameter_annotations_length();

1595       localvariable_len,
1596       new_compressed_linenumber_size,
1597       exception_table_len,
1598       checked_exceptions_len,
1599       method_parameters_len,
1600       cm->generic_signature_index(),
1601       method_annotations_len,
1602       parameter_annotations_len,
1603       type_annotations_len,
1604       default_annotations_len,
1605       0);
1606 
1607   ClassLoaderData* loader_data = m->method_holder()->class_loader_data();
1608   Method* newm_oop = Method::allocate(loader_data,
1609                                       new_code_length,
1610                                       flags,
1611                                       &sizes,
1612                                       m->method_type(),
1613                                       CHECK_(methodHandle()));
1614   methodHandle newm (THREAD, newm_oop);
1615 
1616   // Create a shallow copy of Method part, but be careful to preserve the new ConstMethod*

1626   memcpy(newcm, m->constMethod(), sizeof(ConstMethod));
1627

1629   newm->set_constMethod(newcm);

1635   assert(newm->exception_table_length() == exception_table_len, "check");

1638   memcpy(newm->code_base(), new_code, new_code_length);
1639   // Copy line number table
1640   if (new_compressed_linenumber_size > 0) {
1641     memcpy(newm->compressed_linenumber_table(),
1642            new_compressed_linenumber_table,
1643            new_compressed_linenumber_size);
1644   }
1645   // Copy method_parameters
1646   if (method_parameters_len > 0) {
1647     memcpy(newm->method_parameters_start(),
1648            m->method_parameters_start(),
1649            method_parameters_len * sizeof(MethodParametersElement));
1650   }
1651   // Copy checked_exceptions
1652   if (checked_exceptions_len > 0) {
1653     memcpy(newm->checked_exceptions_start(),
1654            m->checked_exceptions_start(),
1655            checked_exceptions_len * sizeof(CheckedExceptionElement));
1656   }
1657   // Copy exception table
1658   if (exception_table_len > 0) {
1659     memcpy(newm->exception_table_start(),
1660            m->exception_table_start(),
1661            exception_table_len * sizeof(ExceptionTableElement));
1662   }
1663   // Copy local variable number table
1664   if (localvariable_len > 0) {
1665     memcpy(newm->localvariable_table_start(),
1666            m->localvariable_table_start(),
1667            localvariable_len * sizeof(LocalVariableTableElement));
1668   }
1669   // Copy stackmap table
1670   if (m->has_stackmap_table()) {
1671     int code_attribute_length = m->stackmap_data()->length();
1672     Array<u1>* stackmap_data =
1673       MetadataFactory::new_array<u1>(loader_data, code_attribute_length, 0, CHECK_(methodHandle()));
1674     memcpy((void*)stackmap_data->adr_at(0),
1675            (void*)m->stackmap_data()->adr_at(0), code_attribute_length);
1676     newm->set_stackmap_data(stackmap_data);
1677   }
1678 
1679   // copy annotations over to new method
1680   newcm->copy_annotations_from(loader_data, cm, CHECK_(methodHandle()));
1681   return newm;
1682 }

1701   const uintptr_t max_id_uint = right_n_bits((int)(sizeof(_intrinsic_id) * BitsPerByte));

1703   assert(intrinsic_id_size_in_bytes() == sizeof(_intrinsic_id), "");
1704 
1705   // the klass name is well-known:
1706   vmSymbols::SID klass_id = klass_id_for_intrinsics(method_holder());
1707   assert(klass_id != vmSymbols::NO_SID, "caller responsibility");
1708 
1709   // ditto for method and signature:
1710   vmSymbols::SID  name_id = vmSymbols::find_sid(name());
1711   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1712       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1713       && name_id == vmSymbols::NO_SID) {
1714     return;
1715   }
1716   vmSymbols::SID   sig_id = vmSymbols::find_sid(signature());
1717   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1718       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1719       && sig_id == vmSymbols::NO_SID) {
1720     return;
1721   }
1722   jshort flags = access_flags().as_short();
1723 
1724   vmIntrinsics::ID id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1725   if (id != vmIntrinsics::_none) {
1726     set_intrinsic_id(id);
1727     if (id == vmIntrinsics::_Class_cast) {
1728       // Even if the intrinsic is rejected, we want to inline this simple method.
1729       set_force_inline(true);
1730     }
1731     return;
1732   }
1733 
1734   // A few slightly irregular cases:
1735   switch (klass_id) {
1736   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_StrictMath):
1737     // Second chance: check in regular Math.
1738     switch (name_id) {
1739     case vmSymbols::VM_SYMBOL_ENUM_NAME(min_name):
1740     case vmSymbols::VM_SYMBOL_ENUM_NAME(max_name):
1741     case vmSymbols::VM_SYMBOL_ENUM_NAME(sqrt_name):
1742       // pretend it is the corresponding method in the non-strict class:
1743       klass_id = vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_Math);
1744       id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1745       break;
1746     default:
1747       break;
1748     }
1749     break;
1750 
1751   // Signature-polymorphic methods: MethodHandle.invoke*, InvokeDynamic.*., VarHandle
1752   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle):
1753   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle):
1754     if (!is_native())  break;
1755     id = MethodHandles::signature_polymorphic_name_id(method_holder(), name());
1756     if (is_static() != MethodHandles::is_signature_polymorphic_static(id))
1757       id = vmIntrinsics::_none;
1758     break;
1759 
1760   default:
1761     break;
1762   }
1763 
1764   if (id != vmIntrinsics::_none) {
1765     // Set up its iid.  It is an alias method.
1766     set_intrinsic_id(id);
1767     return;
1768   }
1769 }

1982 static void clear_matches(Method* m, int bci) {
1983   InstanceKlass* ik = m->method_holder();
1984   BreakpointInfo* prev_bp = NULL;
1985   BreakpointInfo* next_bp;
1986   for (BreakpointInfo* bp = ik->breakpoints(); bp != NULL; bp = next_bp) {
1987     next_bp = bp->next();
1988     // bci value of -1 is used to delete all breakpoints in method m (ex: clear_all_breakpoint).
1989     if (bci >= 0 ? bp->match(m, bci) : bp->match(m)) {
1990       // do this first:
1991       bp->clear(m);
1992       // unhook it
1993       if (prev_bp != NULL)
1994         prev_bp->set_next(next_bp);
1995       else
1996         ik->set_breakpoints(next_bp);
1997       delete bp;
1998       // When class is redefined JVMTI sets breakpoint in all versions of EMCP methods
1999       // at same location. So we have multiple matching (method_index and bci)
2000       // BreakpointInfo nodes in BreakpointInfo list. We should just delete one
2001       // breakpoint for clear_breakpoint request and keep all other method versions
2002       // BreakpointInfo for future clear_breakpoint request.
2003       // bcivalue of -1 is used to clear all breakpoints (see clear_all_breakpoints)
2004       // which is being called when class is unloaded. We delete all the Breakpoint
2005       // information for all versions of method. We may not correctly restore the original
2006       // bytecode in all method versions, but that is ok. Because the class is being unloaded
2007       // so these methods won't be used anymore.
2008       if (bci >= 0) {
2009         break;
2010       }
2011     } else {
2012       // This one is a keeper.
2013       prev_bp = bp;
2014     }
2015   }
2016 }

2170 class JNIMethodBlock : public CHeapObj<mtClass> {
2171   JNIMethodBlockNode _head;
2172   JNIMethodBlockNode *_last_free;
2173  public:
2174   static Method* const _free_method;
2175 
2176   JNIMethodBlock(int initial_capacity = min_block_size)
2177       : _head(initial_capacity), _last_free(&_head) {}
2178 
2179   void ensure_methods(int num_addl_methods) {
2180     _last_free->ensure_methods(num_addl_methods);
2181   }
2182 
2183   Method** add_method(Method* m) {
2184     for (JNIMethodBlockNode* b = _last_free; b != NULL; b = b->_next) {
2185       if (b->_top < b->_number_of_methods) {
2186         // top points to the next free entry.
2187         int i = b->_top;
2188         b->_methods[i] = m;
2189         b->_top++;
2190         _last_free = b;
2191         return &(b->_methods[i]);
2192       } else if (b->_top == b->_number_of_methods) {
2193         // if the next free entry ran off the block see if there's a free entry
2194         for (int i = 0; i < b->_number_of_methods; i++) {
2195           if (b->_methods[i] == _free_method) {
2196             b->_methods[i] = m;
2197             _last_free = b;
2198             return &(b->_methods[i]);
2199           }
2200         }
2201         // Only check each block once for frees.  They're very unlikely.
2202         // Increment top past the end of the block.
2203         b->_top++;
2204       }
2205       // need to allocate a next block.
2206       if (b->_next == NULL) {
2207         b->_next = _last_free = new JNIMethodBlockNode();
2208       }
2209     }
2210     guarantee(false, "Should always allocate a free block");
2211     return NULL;
2212   }
2213 
2214   bool contains(Method** m) {
2215     if (m == NULL) return false;
2216     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2217       if (b->_methods <= m && m < b->_methods + b->_number_of_methods) {
2218         // This is a bit of extra checking, for two reasons.  One is
2219         // that contains() deals with pointers that are passed in by
2220         // JNI code, so making sure that the pointer is aligned
2221         // correctly is valuable.  The other is that <= and > are
2222         // technically not defined on pointers, so the if guard can
2223         // pass spuriously; no modern compiler is likely to make that
2224         // a problem, though (and if one did, the guard could also
2225         // fail spuriously, which would be bad).
2226         ptrdiff_t idx = m - b->_methods;
2227         if (b->_methods + idx == m) {
2228           return true;
2229         }
2230       }
2231     }
2232     return false;  // not found
2233   }

2530 // TouchedMethodRecord -- we can't use a HashtableEntry<Method*> because
2531 // the Method may be garbage collected. Let's roll our own hash table.
2532 class TouchedMethodRecord : CHeapObj<mtTracing> {
2533 public:
2534   // It's OK to store Symbols here because they will NOT be GC'ed if
2535   // LogTouchedMethods is enabled.
2536   TouchedMethodRecord* _next;
2537   Symbol* _class_name;
2538   Symbol* _method_name;
2539   Symbol* _method_signature;
2540 };
2541 
2542 static const int TOUCHED_METHOD_TABLE_SIZE = 20011;
2543 static TouchedMethodRecord** _touched_method_table = NULL;
2544 
2545 void Method::log_touched(TRAPS) {
2546 
2547   const int table_size = TOUCHED_METHOD_TABLE_SIZE;
2548   Symbol* my_class = klass_name();
2549   Symbol* my_name  = name();
2550   Symbol* my_sig   = signature();
2551 
2552   unsigned int hash = my_class->identity_hash() +
2553                       my_name->identity_hash() +
2554                       my_sig->identity_hash();
2555   juint index = juint(hash) % table_size;
2556 
2557   MutexLocker ml(THREAD, TouchedMethodLog_lock);
2558   if (_touched_method_table == NULL) {
2559     _touched_method_table = NEW_C_HEAP_ARRAY2(TouchedMethodRecord*, table_size,
2560                                               mtTracing, CURRENT_PC);
2561     memset(_touched_method_table, 0, sizeof(TouchedMethodRecord*)*table_size);
2562   }
2563 
2564   TouchedMethodRecord* ptr = _touched_method_table[index];
2565   while (ptr) {
2566     if (ptr->_class_name       == my_class &&
2567         ptr->_method_name      == my_name &&
2568         ptr->_method_signature == my_sig) {
2569       return;
2570     }
2571     if (ptr->_next == NULL) break;
2572     ptr = ptr->_next;
2573   }
2574   TouchedMethodRecord* nptr = NEW_C_HEAP_OBJ(TouchedMethodRecord, mtTracing);
2575   my_class->increment_refcount();
2576   my_name->increment_refcount();
2577   my_sig->increment_refcount();
2578   nptr->_class_name         = my_class;
2579   nptr->_method_name        = my_name;
2580   nptr->_method_signature   = my_sig;
2581   nptr->_next               = NULL;
2582 
2583   if (ptr == NULL) {
2584     // first
2585     _touched_method_table[index] = nptr;
2586   } else {
2587     ptr->_next = nptr;
2588   }
2589 }
