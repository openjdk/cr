273     // Ninety days in milliseconds
274     private static final long NINETY_DAYS = 7776000000L;
275 
276     private static boolean atLeastOneFailed = false;
277 
278     private static MessageDigest md;
279 
280     public static void main(String[] args) throws Exception {
281         System.out.println("cacerts file: " + CACERTS);
282         md = MessageDigest.getInstance("SHA-256");
283 
284         byte[] data = Files.readAllBytes(Path.of(CACERTS));
285         String checksum = toHexString(md.digest(data));
286         if (!checksum.equals(CHECKSUM)) {
287             atLeastOneFailed = true;
288             System.err.println("ERROR: wrong checksum\n" + checksum);
289             System.err.println("Expected checksum\n" + CHECKSUM);
290         }
291 
292         KeyStore ks = KeyStore.getInstance("JKS");
293         ks.load(new ByteArrayInputStream(data), "changeit".toCharArray());
294 
295         // check the count of certs inside
296         if (ks.size() != COUNT) {
297             atLeastOneFailed = true;
298             System.err.println("ERROR: " + ks.size() + " entries, should be "
299                     + COUNT);
300         }
301 
302         // check that all entries in the map are in the keystore
303         for (String alias : FINGERPRINT_MAP.keySet()) {
304             if (!ks.isCertificateEntry(alias)) {
305                 atLeastOneFailed = true;
306                 System.err.println("ERROR: " + alias + " is not in cacerts");
307             }
308         }
309 
310         // pull all the trusted self-signed CA certs out of the cacerts file
311         // and verify their signatures
312         Enumeration<String> aliases = ks.aliases();
313         while (aliases.hasMoreElements()) {
314             String alias = aliases.nextElement();
315             System.out.println("\nVerifying " + alias);
316             if (!ks.isCertificateEntry(alias)) {
317                 atLeastOneFailed = true;
318                 System.err.println("ERROR: " + alias
319                         + " is not a trusted cert entry");
320             }
321             X509Certificate cert = (X509Certificate) ks.getCertificate(alias);
322             if (!checkFingerprint(alias, cert)) {
323                 atLeastOneFailed = true;
324                 System.err.println("ERROR: " + alias + " SHA-256 fingerprint is incorrect");
325             }
326             // Make sure cert can be self-verified
327             try {
328                 cert.verify(cert.getPublicKey());
329             } catch (Exception e) {
330                 atLeastOneFailed = true;
331                 System.err.println("ERROR: cert cannot be verified:"
332                         + e.getMessage());
333             }
334 
335             // Make sure cert is not expired or not yet valid
336             try {
337                 cert.checkValidity();
338             } catch (CertificateExpiredException cee) {
339                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
340                     atLeastOneFailed = true;
341                     System.err.println("ERROR: cert is expired");
342                 }
343             } catch (CertificateNotYetValidException cne) {
344                 atLeastOneFailed = true;
345                 System.err.println("ERROR: cert is not yet valid");
346             }
347 
348             // If cert is within 90 days of expiring, mark as failure so
349             // that cert can be scheduled to be removed/renewed.
350             Date notAfter = cert.getNotAfter();
351             if (notAfter.getTime() - System.currentTimeMillis() < NINETY_DAYS) {
352                 if (!EXPIRY_EXC_ENTRIES.contains(alias)) {
353                     atLeastOneFailed = true;
354                     System.err.println("ERROR: cert \"" + alias + "\" expiry \""
355                             + notAfter.toString() + "\" will expire within 90 days");
356                 }
357             }
358         }
359 
360         if (atLeastOneFailed) {
361             throw new Exception("At least one cacert test failed");
362         }
363     }

365     private static boolean checkFingerprint(String alias, Certificate cert)
366             throws Exception {
367         String fingerprint = FINGERPRINT_MAP.get(alias);
368         if (fingerprint == null) {
369             // no entry for alias
370             return true;
371         }
372         System.out.println("Checking fingerprint of " + alias);
373         byte[] digest = md.digest(cert.getEncoded());
374         return fingerprint.equals(toHexString(digest));
375     }

377     private static String toHexString(byte[] block) {
378         StringBuilder buf = new StringBuilder();
379         int len = block.length;
380         for (int i = 0; i < len; i++) {
381             buf.append(String.format("%02X", block[i]));
382             if (i < len - 1) {
383                 buf.append(":");
384             }
385         }
386         return buf.toString();
387     }
