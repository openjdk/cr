  72 static void post_safepoint_begin_event(EventSafepointBegin& event,
  73                                        uint64_t safepoint_id,
  74                                        int thread_count,
  75                                        int critical_thread_count) {
  76   if (event.should_commit()) {
  77     event.set_safepointId(safepoint_id);
  78     event.set_totalThreadCount(thread_count);
  79     event.set_jniCriticalThreadCount(critical_thread_count);
  80     event.commit();
  81   }
  82 }
  83 
  84 static void post_safepoint_cleanup_event(EventSafepointCleanup& event, uint64_t safepoint_id) {
  85   if (event.should_commit()) {
  86     event.set_safepointId(safepoint_id);
  87     event.commit();
  88   }
  89 }
  90 
  91 static void post_safepoint_synchronize_event(EventSafepointStateSynchronization& event,
  92                                              uint64_t safepoint_id,
  93                                              int initial_number_of_threads,
  94                                              int threads_waiting_to_block,
  95                                              uint64_t iterations) {
  96   if (event.should_commit()) {
  97     event.set_safepointId(safepoint_id);
  98     event.set_initialThreadCount(initial_number_of_threads);
  99     event.set_runningThreadCount(threads_waiting_to_block);
 100     event.set_iterations(iterations);
 101     event.commit();
 102   }
 103 }

 105 static void post_safepoint_cleanup_task_event(EventSafepointCleanupTask& event,
 106                                               uint64_t safepoint_id,
 107                                               const char* name) {
 108   if (event.should_commit()) {
 109     event.set_safepointId(safepoint_id);
 110     event.set_name(name);
 111     event.commit();
 112   }
 113 }

 115 static void post_safepoint_end_event(EventSafepointEnd& event, uint64_t safepoint_id) {
 116   if (event.should_commit()) {
 117     event.set_safepointId(safepoint_id);
 118     event.commit();
 119   }

 167 bool SafepointSynchronize::thread_not_running(ThreadSafepointState *cur_state) {
 168   if (!cur_state->is_running()) {
 169     return true;
 170   }
 171   cur_state->examine_state_of_thread(SafepointSynchronize::safepoint_counter());
 172   if (!cur_state->is_running()) {
 173     return true;
 174   }
 175   LogTarget(Trace, safepoint) lt;
 176   if (lt.is_enabled()) {
 177     ResourceMark rm;
 178     LogStream ls(lt);
 179     cur_state->print_on(&ls);
 180   }
 181   return false;
 182 }

 184 #ifdef ASSERT
 185 static void assert_list_is_valid(const ThreadSafepointState* tss_head, int still_running) {
 186   int a = 0;
 187   const ThreadSafepointState *tmp_tss = tss_head;
 188   while (tmp_tss != NULL) {
 189     ++a;
 190     assert(tmp_tss->is_running(), "Illegal initial state");
 191     tmp_tss = tmp_tss->get_next();
 192   }
 193   assert(a == still_running, "Must be the same");
 194 }
 195 #endif // ASSERT

 197 static void back_off(int64_t start_time) {
 198   // We start with fine-grained nanosleeping until a millisecond has
 199   // passed, at which point we resort to plain naked_short_sleep.
 200   if (os::javaTimeNanos() - start_time < NANOSECS_PER_MILLISEC) {
 201     os::naked_short_nanosleep(10 * (NANOUNITS / MICROUNITS));
 202   } else {
 203     os::naked_short_sleep(1);
 204   }
 205 }

 207 int SafepointSynchronize::synchronize_threads(jlong safepoint_limit_time, int nof_threads, int* initial_running)
 208 {
 209   JavaThreadIteratorWithHandle jtiwh;
 210 
 211 #ifdef ASSERT
 212   for (; JavaThread *cur = jtiwh.next(); ) {
 213     assert(cur->safepoint_state()->is_running(), "Illegal initial state");
 214   }
 215   jtiwh.rewind();
 216 #endif // ASSERT
 217 
 218   // Iterate through all threads until it has been determined how to stop them all at a safepoint.
 219   int still_running = nof_threads;
 220   ThreadSafepointState *tss_head = NULL;
 221   ThreadSafepointState **p_prev = &tss_head;
 222   for (; JavaThread *cur = jtiwh.next(); ) {
 223     ThreadSafepointState *cur_tss = cur->safepoint_state();
 224     assert(cur_tss->get_next() == NULL, "Must be NULL");
 225     if (thread_not_running(cur_tss)) {
 226       --still_running;
 227     } else {
 228       *p_prev = cur_tss;
 229       p_prev = cur_tss->next_ptr();
 230     }
 231   }
 232   *p_prev = NULL;
 233 
 234   DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 235 
 236   *initial_running = still_running;
 237 
 238   // If there is no thread still running, we are already done.
 239   if (still_running <= 0) {
 240     assert(tss_head == NULL, "Must be empty");
 241     return 1;
 242   }
 243 
 244   int iterations = 1; // The first iteration is above.
 245   int64_t start_time = os::javaTimeNanos();
 246 
 247   do {
 248     // Check if this has taken too long:
 249     if (SafepointTimeout && safepoint_limit_time < os::javaTimeNanos()) {
 250       print_safepoint_timeout();
 251     }
 252 
 253     p_prev = &tss_head;
 254     ThreadSafepointState *cur_tss = tss_head;
 255     while (cur_tss != NULL) {
 256       assert(cur_tss->is_running(), "Illegal initial state");
 257       if (thread_not_running(cur_tss)) {
 258         --still_running;
 259         *p_prev = NULL;
 260         ThreadSafepointState *tmp = cur_tss;
 261         cur_tss = cur_tss->get_next();
 262         tmp->set_next(NULL);
 263       } else {
 264         *p_prev = cur_tss;
 265         p_prev = cur_tss->next_ptr();
 266         cur_tss = cur_tss->get_next();
 267       }
 268     }
 269 
 270     DEBUG_ONLY(assert_list_is_valid(tss_head, still_running);)
 271 
 272     if (still_running > 0) {
 273       back_off(start_time);
 274     }
 275 
 276     iterations++;
 277   } while (still_running > 0);
 278 
 279   assert(tss_head == NULL, "Must be empty");
 280 
 281   return iterations;
 282 }

 284 void SafepointSynchronize::arm_safepoint() {
 285   // Begin the process of bringing the system to a safepoint.
 286   // Java threads can be in several different states and are
 287   // stopped by different mechanisms:
 288   //
 289   //  1. Running interpreted
 290   //     When executing branching/returning byte codes interpreter
 291   //     checks if the poll is armed, if so blocks in SS::block().
 292   //  2. Running in native code
 293   //     When returning from the native code, a Java thread must check
 294   //     the safepoint _state to see if we must block.  If the
 295   //     VM thread sees a Java thread in native, it does
 296   //     not wait for this thread to block.  The order of the memory
 297   //     writes and reads of both the safepoint state and the Java
 298   //     threads state is critical.  In order to guarantee that the
 299   //     memory writes are serialized with respect to each other,
 300   //     the VM thread issues a memory barrier instruction.
 301   //  3. Running compiled Code
 302   //     Compiled code reads the local polling page that
 303   //     is set to fault if we are trying to get to a safepoint.
 304   //  4. Blocked
 305   //     A thread which is blocked will not be allowed to return from the
 306   //     block condition until the safepoint operation is complete.
 307   //  5. In VM or Transitioning between states
 308   //     If a Java thread is currently running in the VM or transitioning
 309   //     between states, the safepointing code will poll the thread state
 310   //     until the thread blocks itself when it attempts transitions to a
 311   //     new state or locking a safepoint checked monitor.
 312 
 313   // We must never miss a thread with correct safepoint id, so we must make sure we arm
 314   // the wait barrier for the next safepoint id/counter.
 315   // Arming must be done after resetting _current_jni_active_count, _waiting_to_block.
 316   _wait_barrier->arm(static_cast<int>(_safepoint_counter + 1));
 317 
 318   assert((_safepoint_counter & 0x1) == 0, "must be even");
 319   // The store to _safepoint_counter must happen after any stores in arming.
 320   Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 321 
 322   // We are synchronizing
 323   OrderAccess::storestore(); // Ordered with _safepoint_counter
 324   _state = _synchronizing;
 325 
 326   // Arming the per thread poll while having _state != _not_synchronized means safepointing
 327   log_trace(safepoint)("Setting thread local yield flag for threads");
 328   OrderAccess::storestore(); // storestore, global state -> local state
 329   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 330     // Make sure the threads start polling, it is time to yield.
 331     SafepointMechanism::arm_local_poll(cur);
 332   }
 333 
 334   OrderAccess::fence(); // storestore|storeload, global state -> local state
 335 }

 337 // Roll all threads forward to a safepoint and suspend them all
 338 void SafepointSynchronize::begin() {
 339   assert(Thread::current()->is_VM_thread(), "Only VM thread may execute a safepoint");
 340 
 341   EventSafepointBegin begin_event;
 342   SafepointTracing::begin(VMThread::vm_op_type());
 343 
 344   Universe::heap()->safepoint_synchronize_begin();
 345 
 346   // By getting the Threads_lock, we assure that no threads are about to start or
 347   // exit. It is released again in SafepointSynchronize::end().
 348   Threads_lock->lock();
 349 
 350   assert( _state == _not_synchronized, "trying to safepoint synchronize with wrong state");
 351 
 352   int nof_threads = Threads::number_of_threads();
 353 
 354   _nof_threads_hit_polling_page = 0;
 355 
 356   log_debug(safepoint)("Safepoint synchronization initiated using %s wait barrier. (%d threads)", _wait_barrier->description(), nof_threads);
 357 
 358   // Reset the count of active JNI critical threads
 359   _current_jni_active_count = 0;
 360 
 361   // Set number of threads to wait for
 362   _waiting_to_block = nof_threads;
 363 
 364   jlong safepoint_limit_time = 0;
 365   if (SafepointTimeout) {
 366     // Set the limit time, so that it can be compared to see if this has taken
 367     // too long to complete.
 368     safepoint_limit_time = SafepointTracing::start_of_safepoint() + (jlong)SafepointTimeoutDelay * (NANOUNITS / MILLIUNITS);
 369     timeout_error_printed = false;
 370   }
 371 
 372   EventSafepointStateSynchronization sync_event;
 373   int initial_running = 0;
 374 
 375   // Arms the safepoint, _current_jni_active_count and _waiting_to_block must be set before.
 376   arm_safepoint();
 377 
 378   // Will spin until all threads are safe.
 379   int iterations = synchronize_threads(safepoint_limit_time, nof_threads, &initial_running);
 380   assert(_waiting_to_block == 0, "No thread should be running");
 381 
 382 #ifndef PRODUCT
 383   if (safepoint_limit_time != 0) {
 384     jlong current_time = os::javaTimeNanos();
 385     if (safepoint_limit_time < current_time) {
 386       log_warning(safepoint)("# SafepointSynchronize: Finished after "
 387                     INT64_FORMAT_W(6) " ms",
 388                     (int64_t)(current_time - SafepointTracing::start_of_safepoint()) / (NANOUNITS / MILLIUNITS));
 389     }
 390   }
 391 #endif
 392 
 393   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 394 
 395   // Record state
 396   _state = _synchronized;
 397 
 398   OrderAccess::fence();
 399 
 400   // Set the new id
 401   ++_safepoint_id;
 402 
 403 #ifdef ASSERT
 404   // Make sure all the threads were visited.
 405   for (JavaThreadIteratorWithHandle jtiwh; JavaThread *cur = jtiwh.next(); ) {
 406     assert(cur->was_visited_for_critical_count(_safepoint_counter), "missed a thread");
 407   }
 408 #endif // ASSERT
 409 
 410   // Update the count of active JNI critical regions
 411   GCLocker::set_jni_lock_count(_current_jni_active_count);
 412 
 413   post_safepoint_synchronize_event(sync_event,
 414                                    _safepoint_id,
 415                                    initial_running,
 416                                    _waiting_to_block, iterations);
 417 
 418   SafepointTracing::synchronized(nof_threads, initial_running, _nof_threads_hit_polling_page);
 419 
 420   // We do the safepoint cleanup first since a GC related safepoint
 421   // needs cleanup to be completed before running the GC op.
 422   EventSafepointCleanup cleanup_event;
 423   do_cleanup_tasks();
 424   post_safepoint_cleanup_event(cleanup_event, _safepoint_id);
 425 
 426   post_safepoint_begin_event(begin_event, _safepoint_id, nof_threads, _current_jni_active_count);
 427   SafepointTracing::cleanup();
 428 }

 430 void SafepointSynchronize::disarm_safepoint() {
 431   uint64_t active_safepoint_counter = _safepoint_counter;
 432   {
 433     JavaThreadIteratorWithHandle jtiwh;
 434 #ifdef ASSERT
 435     // A pending_exception cannot be installed during a safepoint.  The threads
 436     // may install an async exception after they come back from a safepoint into
 437     // pending_exception after they unblock.  But that should happen later.
 438     for (; JavaThread *cur = jtiwh.next(); ) {
 439       assert (!(cur->has_pending_exception() &&
 440                 cur->safepoint_state()->is_at_poll_safepoint()),
 441               "safepoint installed a pending exception");
 442     }
 443 #endif // ASSERT
 444 
 445     OrderAccess::fence(); // keep read and write of _state from floating up
 446     assert(_state == _synchronized, "must be synchronized before ending safepoint synchronization");
 447 
 448     // Change state first to _not_synchronized.
 449     // No threads should see _synchronized when running.
 450     _state = _not_synchronized;
 451 
 452     // Set the next dormant (even) safepoint id.
 453     assert((_safepoint_counter & 0x1) == 1, "must be odd");
 454     Atomic::release_store(&_safepoint_counter, _safepoint_counter + 1);
 455 
 456     OrderAccess::fence(); // Keep the local state from floating up.
 457 
 458     jtiwh.rewind();
 459     for (; JavaThread *current = jtiwh.next(); ) {
 460       // Clear the visited flag to ensure that the critical counts are collected properly.
 461       DEBUG_ONLY(current->reset_visited_for_critical_count(active_safepoint_counter);)
 462       ThreadSafepointState* cur_state = current->safepoint_state();
 463       assert(!cur_state->is_running(), "Thread not suspended at safepoint");
 464       cur_state->restart(); // TSS _running
 465       assert(cur_state->is_running(), "safepoint state has not been reset");
 466     }
 467   } // ~JavaThreadIteratorWithHandle
 468 
 469   // Release threads lock, so threads can be created/destroyed again.
 470   Threads_lock->unlock();
 471 
 472   // Wake threads after local state is correctly set.
 473   _wait_barrier->disarm();
 474 }

 476 // Wake up all threads, so they are ready to resume execution after the safepoint
 477 // operation has been carried out
 478 void SafepointSynchronize::end() {
 479   assert(Threads_lock->owned_by_self(), "must hold Threads_lock");
 480   EventSafepointEnd event;
 481   assert(Thread::current()->is_VM_thread(), "Only VM thread can execute a safepoint");
 482 
 483   disarm_safepoint();
 484 
 485   Universe::heap()->safepoint_synchronize_end();
 486 
 487   SafepointTracing::end();
 488 
 489   post_safepoint_end_event(event, safepoint_id());
 490 }

 492 bool SafepointSynchronize::is_cleanup_needed() {
 493   // Need a cleanup safepoint if there are too many monitors in use
 494   // and the monitor deflation needs to be done at a safepoint.
 495   if (ObjectSynchronizer::is_safepoint_deflation_needed()) return true;
 496   // Need a safepoint if some inline cache buffers is non-empty
 497   if (!InlineCacheBuffer::is_empty()) return true;
 498   if (StringTable::needs_rehashing()) return true;
 499   if (SymbolTable::needs_rehashing()) return true;
 500   return false;
 501 }

 503 class ParallelSPCleanupThreadClosure : public ThreadClosure {
 504 private:
 505   CodeBlobClosure* _nmethod_cl;
 506   DeflateMonitorCounters* _counters;
 507 
 508 public:
 509   ParallelSPCleanupThreadClosure(DeflateMonitorCounters* counters) :
 510     _nmethod_cl(UseCodeAging ? NMethodSweeper::prepare_reset_hotness_counters() : NULL),
 511     _counters(counters) {}
 512 
 513   void do_thread(Thread* thread) {
 514     // deflate_thread_local_monitors() handles or requests deflation of
 515     // this thread's idle monitors. If !AsyncDeflateIdleMonitors or if
 516     // there is a special cleanup request, deflation is handled now.
 517     // Otherwise, async deflation is requested via a flag.
 518     ObjectSynchronizer::deflate_thread_local_monitors(thread, _counters);
 519     if (_nmethod_cl != NULL && thread->is_Java_thread() &&
 520         ! thread->is_Code_cache_sweeper_thread()) {
 521       JavaThread* jt = (JavaThread*) thread;
 522       jt->nmethods_do(_nmethod_cl);
 523     }
 524   }
 525 };

 527 class ParallelSPCleanupTask : public AbstractGangTask {
 528 private:
 529   SubTasksDone _subtasks;
 530   ParallelSPCleanupThreadClosure _cleanup_threads_cl;
 531   uint _num_workers;
 532   DeflateMonitorCounters* _counters;
 533 public:
 534   ParallelSPCleanupTask(uint num_workers, DeflateMonitorCounters* counters) :
 535     AbstractGangTask("Parallel Safepoint Cleanup"),
 536     _subtasks(SafepointSynchronize::SAFEPOINT_CLEANUP_NUM_TASKS),
 537     _cleanup_threads_cl(ParallelSPCleanupThreadClosure(counters)),
 538     _num_workers(num_workers),
 539     _counters(counters) {}
 540 
 541   void work(uint worker_id) {
 542     uint64_t safepoint_id = SafepointSynchronize::safepoint_id();
 543     // All threads deflate monitors and mark nmethods (if necessary).
 544     Threads::possibly_parallel_threads_do(true, &_cleanup_threads_cl);
 545 
 546     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_DEFLATE_MONITORS)) {
 547       const char* name = "deflating global idle monitors";
 548       EventSafepointCleanupTask event;
 549       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 550       // AsyncDeflateIdleMonitors only uses DeflateMonitorCounters
 551       // when a special cleanup has been requested.
 552       // Note: This logging output will include global idle monitor
 553       // elapsed times, but not global idle monitor deflation count.
 554       ObjectSynchronizer::do_safepoint_work(_counters);
 555 
 556       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 557     }
 558 
 559     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_UPDATE_INLINE_CACHES)) {
 560       const char* name = "updating inline caches";
 561       EventSafepointCleanupTask event;
 562       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 563       InlineCacheBuffer::update_inline_caches();
 564 
 565       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 566     }
 567 
 568     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_COMPILATION_POLICY)) {
 569       const char* name = "compilation policy safepoint handler";
 570       EventSafepointCleanupTask event;
 571       TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 572       CompilationPolicy::policy()->do_safepoint_work();
 573 
 574       post_safepoint_cleanup_task_event(event, safepoint_id, name);
 575     }
 576 
 577     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYMBOL_TABLE_REHASH)) {
 578       if (SymbolTable::needs_rehashing()) {
 579         const char* name = "rehashing symbol table";
 580         EventSafepointCleanupTask event;
 581         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 582         SymbolTable::rehash_table();
 583 
 584         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 585       }
 586     }
 587 
 588     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_STRING_TABLE_REHASH)) {
 589       if (StringTable::needs_rehashing()) {
 590         const char* name = "rehashing string table";
 591         EventSafepointCleanupTask event;
 592         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 593         StringTable::rehash_table();
 594 
 595         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 596       }
 597     }
 598 
 599     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_SYSTEM_DICTIONARY_RESIZE)) {
 600       if (Dictionary::does_any_dictionary_needs_resizing()) {
 601         const char* name = "resizing system dictionaries";
 602         EventSafepointCleanupTask event;
 603         TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 604         ClassLoaderDataGraph::resize_dictionaries();
 605 
 606         post_safepoint_cleanup_task_event(event, safepoint_id, name);
 607       }
 608     }
 609 
 610     if (_subtasks.try_claim_task(SafepointSynchronize::SAFEPOINT_CLEANUP_REQUEST_OOPSTORAGE_CLEANUP)) {
 611       // Don't bother reporting event or time for this very short operation.
 612       // To have any utility we'd also want to report whether needed.
 613       OopStorage::trigger_cleanup_if_needed();
 614     }
 615 
 616     _subtasks.all_tasks_completed(_num_workers);
 617   }
 618 };

 620 // Various cleaning tasks that should be done periodically at safepoints.
 621 void SafepointSynchronize::do_cleanup_tasks() {
 622 
 623   TraceTime timer("safepoint cleanup tasks", TRACETIME_LOG(Info, safepoint, cleanup));
 624 
 625   // Prepare for monitor deflation.
 626   DeflateMonitorCounters deflate_counters;
 627   ObjectSynchronizer::prepare_deflate_idle_monitors(&deflate_counters);
 628 
 629   CollectedHeap* heap = Universe::heap();
 630   assert(heap != NULL, "heap not initialized yet?");
 631   WorkGang* cleanup_workers = heap->get_safepoint_workers();
 632   if (cleanup_workers != NULL) {
 633     // Parallel cleanup using GC provided thread pool.
 634     uint num_cleanup_workers = cleanup_workers->active_workers();
 635     ParallelSPCleanupTask cleanup(num_cleanup_workers, &deflate_counters);
 636     StrongRootsScope srs(num_cleanup_workers);
 637     cleanup_workers->run_task(&cleanup);
 638   } else {
 639     // Serial cleanup using VMThread.
 640     ParallelSPCleanupTask cleanup(1, &deflate_counters);
 641     StrongRootsScope srs(1);
 642     cleanup.work(0);
 643   }
 644 
 645   // Needs to be done single threaded by the VMThread.  This walks
 646   // the thread stacks looking for references to metadata before
 647   // deciding to remove it from the metaspaces.
 648   if (ClassLoaderDataGraph::should_clean_metaspaces_and_reset()) {
 649     const char* name = "cleanup live ClassLoaderData metaspaces";
 650     TraceTime timer(name, TRACETIME_LOG(Info, safepoint, cleanup));
 651     ClassLoaderDataGraph::walk_metadata_and_clean_metaspaces();
 652   }
 653 
 654   // Finish monitor deflation.
 655   ObjectSynchronizer::finish_deflate_idle_monitors(&deflate_counters);
 656 
 657   assert(InlineCacheBuffer::is_empty(), "should have cleaned up ICBuffer");
 658 }

 660 // Methods for determining if a JavaThread is safepoint safe.
 661 
 662 // False means unsafe with undetermined state.
 663 // True means a determined state, but it may be an unsafe state.
 664 // If called from a non-safepoint context safepoint_count MUST be InactiveSafepointCounter.
 665 bool SafepointSynchronize::try_stable_load_state(JavaThreadState *state, JavaThread *thread, uint64_t safepoint_count) {
 666   assert((safepoint_count != InactiveSafepointCounter &&
 667           Thread::current() == (Thread*)VMThread::vm_thread() &&
 668           SafepointSynchronize::_state != _not_synchronized)
 669          || safepoint_count == InactiveSafepointCounter, "Invalid check");
 670 
 671   // To handle the thread_blocked state on the backedge of the WaitBarrier from
 672   // previous safepoint and reading the reset value (0/InactiveSafepointCounter) we
 673   // re-read state after we read thread safepoint id. The JavaThread changes its
 674   // thread state from thread_blocked before resetting safepoint id to 0.
 675   // This guarantees the second read will be from an updated thread state. It can
 676   // either be different state making this an unsafe state or it can see blocked
 677   // again. When we see blocked twice with a 0 safepoint id, either:
 678   // - It is normally blocked, e.g. on Mutex, TBIVM.
 679   // - It was in SS:block(), looped around to SS:block() and is blocked on the WaitBarrier.
 680   // - It was in SS:block() but now on a Mutex.
 681   // All of these cases are safe.
 682 
 683   *state = thread->thread_state();
 684   OrderAccess::loadload();
 685   uint64_t sid = thread->safepoint_state()->get_safepoint_id();  // Load acquire
 686   if (sid != InactiveSafepointCounter && sid != safepoint_count) {
 687     // In an old safepoint, state not relevant.
 688     return false;
 689   }
 690   return *state == thread->thread_state();
 691 }
 692 
 693 static bool safepoint_safe_with(JavaThread *thread, JavaThreadState state) {
 694   switch(state) {
 695   case _thread_in_native:
 696     // native threads are safe if they have no java stack or have walkable stack
 697     return !thread->has_last_Java_frame() || thread->frame_anchor()->walkable();
 698 
 699   case _thread_blocked:
 700     // On wait_barrier or blocked.
 701     // Blocked threads should already have walkable stack.
 702     assert(!thread->has_last_Java_frame() || thread->frame_anchor()->walkable(), "blocked and not walkable");
 703     return true;
 704 
 705   default:
 706     return false;
 707   }
 708 }
 709 
 710 bool SafepointSynchronize::handshake_safe(JavaThread *thread) {
 711   if (thread->is_ext_suspended() || thread->is_terminated()) {
 712     return true;
 713   }
 714   JavaThreadState stable_state;
 715   if (try_stable_load_state(&stable_state, thread, InactiveSafepointCounter)) {
 716     return safepoint_safe_with(thread, stable_state);
 717   }
 718   return false;
 719 }
 720 
 721 // See if the thread is running inside a lazy critical native and
 722 // update the thread critical count if so.  Also set a suspend flag to
 723 // cause the native wrapper to return into the JVM to do the unlock
 724 // once the native finishes.
 725 static void check_for_lazy_critical_native(JavaThread *thread, JavaThreadState state) {
 726   if (state == _thread_in_native &&
 727       thread->has_last_Java_frame() &&
 728       thread->frame_anchor()->walkable()) {
 729     // This thread might be in a critical native nmethod so look at
 730     // the top of the stack and increment the critical count if it
 731     // is.
 732     frame wrapper_frame = thread->last_frame();
 733     CodeBlob* stub_cb = wrapper_frame.cb();
 734     if (stub_cb != NULL &&
 735         stub_cb->is_nmethod() &&
 736         stub_cb->as_nmethod_or_null()->is_lazy_critical_native()) {
 737       // A thread could potentially be in a critical native across
 738       // more than one safepoint, so only update the critical state on
 739       // the first one.  When it returns it will perform the unlock.
 740       if (!thread->do_critical_native_unlock()) {
 741 #ifdef ASSERT
 742         if (!thread->in_critical()) {
 743           GCLocker::increment_debug_jni_lock_count();
 744         }
 745 #endif
 746         thread->enter_critical();
 747         // Make sure the native wrapper calls back on return to
 748         // perform the needed critical unlock.
 749         thread->set_critical_native_unlock();
 750       }
 751     }
 752   }
 753 }

 755 // -------------------------------------------------------------------------------------------------------
 756 // Implementation of Safepoint blocking point
 757 
 758 void SafepointSynchronize::block(JavaThread *thread) {
 759   assert(thread != NULL, "thread must be set");
 760   assert(thread->is_Java_thread(), "not a Java thread");
 761 
 762   // Threads shouldn't block if they are in the middle of printing, but...
 763   ttyLocker::break_tty_lock_for_safepoint(os::current_thread_id());
 764 
 765   // Only bail from the block() call if the thread is gone from the
 766   // thread list; starting to exit should still block.
 767   if (thread->is_terminated()) {
 768      // block current thread if we come here from native code when VM is gone
 769      thread->block_if_vm_exited();
 770 
 771      // otherwise do nothing
 772      return;
 773   }
 774 
 775   JavaThreadState state = thread->thread_state();
 776   thread->frame_anchor()->make_walkable(thread);
 777 
 778   uint64_t safepoint_id = SafepointSynchronize::safepoint_counter();
 779   // Check that we have a valid thread_state at this point
 780   switch(state) {
 781     case _thread_in_vm_trans:
 782     case _thread_in_Java:        // From compiled code
 783     case _thread_in_native_trans:
 784     case _thread_blocked_trans:
 785     case _thread_new_trans:
 786 
 787       // We have no idea where the VMThread is, it might even be at next safepoint.
 788       // So we can miss this poll, but stop at next.
 789 
 790       // Load dependent store, it must not pass loading of safepoint_id.
 791       thread->safepoint_state()->set_safepoint_id(safepoint_id); // Release store
 792 
 793       // This part we can skip if we notice we miss or are in a future safepoint.
 794       OrderAccess::storestore();
 795       // Load in wait barrier should not float up
 796       thread->set_thread_state_fence(_thread_blocked);
 797 
 798       _wait_barrier->wait(static_cast<int>(safepoint_id));
 799       assert(_state != _synchronized, "Can't be");
 800 
 801       // If barrier is disarmed stop store from floating above loads in barrier.
 802       OrderAccess::loadstore();
 803       thread->set_thread_state(state);
 804 
 805       // Then we reset the safepoint id to inactive.
 806       thread->safepoint_state()->reset_safepoint_id(); // Release store
 807 
 808       OrderAccess::fence();
 809 
 810       break;
 811 
 812     default:
 813      fatal("Illegal threadstate encountered: %d", state);
 814   }
 815   guarantee(thread->safepoint_state()->get_safepoint_id() == InactiveSafepointCounter,
 816             "The safepoint id should be set only in block path");
 817 
 818   // Check for pending. async. exceptions or suspends - except if the
 819   // thread was blocked inside the VM. has_special_runtime_exit_condition()
 820   // is called last since it grabs a lock and we only want to do that when
 821   // we must.
 822   //
 823   // Note: we never deliver an async exception at a polling point as the
 824   // compiler may not have an exception handler for it. The polling
 825   // code will notice the async and deoptimize and the exception will
 826   // be delivered. (Polling at a return point is ok though). Sure is
 827   // a lot of bother for a deprecated feature...
 828   //
 829   // We don't deliver an async exception if the thread state is
 830   // _thread_in_native_trans so JNI functions won't be called with
 831   // a surprising pending exception. If the thread state is going back to java,
 832   // async exception is checked in check_special_condition_for_native_trans().
 833 
 834   if (state != _thread_blocked_trans &&
 835       state != _thread_in_vm_trans &&
 836       thread->has_special_runtime_exit_condition()) {
 837     thread->handle_special_runtime_exit_condition(
 838       !thread->is_at_poll_safepoint() && (state != _thread_in_native_trans));
 839   }
 840 
 841   // cross_modify_fence is done by SafepointMechanism::block_if_requested_slow
 842   // which is the only caller here.
 843 }

 938 void ThreadSafepointState::examine_state_of_thread(uint64_t safepoint_count) {
 939   assert(is_running(), "better be running or just have hit safepoint poll");
 940 
 941   JavaThreadState stable_state;
 942   if (!SafepointSynchronize::try_stable_load_state(&stable_state, _thread, safepoint_count)) {
 943     // We could not get stable state of the JavaThread.
 944     // Consider it running and just return.
 945     return;
 946   }
 947 
 948   // Check for a thread that is suspended. Note that thread resume tries
 949   // to grab the Threads_lock which we own here, so a thread cannot be
 950   // resumed during safepoint synchronization.
 951 
 952   // We check to see if this thread is suspended without locking to
 953   // avoid deadlocking with a third thread that is waiting for this
 954   // thread to be suspended. The third thread can notice the safepoint
 955   // that we're trying to start at the beginning of its SR_lock->wait()
 956   // call. If that happens, then the third thread will block on the
 957   // safepoint while still holding the underlying SR_lock. We won't be
 958   // able to get the SR_lock and we'll deadlock.
 959   //
 960   // We don't need to grab the SR_lock here for two reasons:
 961   // 1) The suspend flags are both volatile and are set with an
 962   //    Atomic::cmpxchg() call so we should see the suspended
 963   //    state right away.
 964   // 2) We're being called from the safepoint polling loop; if
 965   //    we don't see the suspended state on this iteration, then
 966   //    we'll come around again.
 967   //
 968   bool is_suspended = _thread->is_ext_suspended();
 969   if (is_suspended) {
 970     account_safe_thread();
 971     return;
 972   }
 973 
 974   if (safepoint_safe_with(_thread, stable_state)) {
 975     check_for_lazy_critical_native(_thread, stable_state);
 976     account_safe_thread();
 977     return;
 978   }
 979 
 980   // All other thread states will continue to run until they
 981   // transition and self-block in state _blocked
 982   // Safepoint polling in compiled code causes the Java threads to do the same.
 983   // Note: new threads may require a malloc so they must be allowed to finish
 984 
 985   assert(is_running(), "examine_state_of_thread on non-running thread");
 986   return;
 987 }

 989 void ThreadSafepointState::account_safe_thread() {
 990   SafepointSynchronize::decrement_waiting_to_block();
 991   if (_thread->in_critical()) {
 992     // Notice that this thread is in a critical section
 993     SafepointSynchronize::increment_jni_active_count();
 994   }
 995   DEBUG_ONLY(_thread->set_visited_for_critical_count(SafepointSynchronize::safepoint_counter());)
 996   assert(!_safepoint_safe, "Must be unsafe before safe");
 997   _safepoint_safe = true;
 998 }

1017 // Block the thread at poll or poll return for safepoint/handshake.
1018 void ThreadSafepointState::handle_polling_page_exception() {
1019 
1020   // Step 1: Find the nmethod from the return address
1021   address real_return_addr = thread()->saved_exception_pc();
1022 
1023   CodeBlob *cb = CodeCache::find_blob(real_return_addr);
1024   assert(cb != NULL && cb->is_compiled(), "return address should be in nmethod");
1025   CompiledMethod* nm = (CompiledMethod*)cb;
1026 
1027   // Find frame of caller
1028   frame stub_fr = thread()->last_frame();
1029   CodeBlob* stub_cb = stub_fr.cb();
1030   assert(stub_cb->is_safepoint_stub(), "must be a safepoint stub");
1031   RegisterMap map(thread(), true);
1032   frame caller_fr = stub_fr.sender(&map);
1033 
1034   // Should only be poll_return or poll
1035   assert( nm->is_at_poll_or_poll_return(real_return_addr), "should not be at call" );
1036 
1037   // This is a poll immediately before a return. The exception handling code
1038   // has already had the effect of causing the return to occur, so the execution
1039   // will continue immediately after the call. In addition, the oopmap at the
1040   // return point does not mark the return value as an oop (if it is), so
1041   // it needs a handle here to be updated.
1042   if( nm->is_at_poll_return(real_return_addr) ) {

1043     // See if return type is an oop.
1044     bool return_oop = nm->method()->is_returning_oop();
1045     Handle return_value;























1046     if (return_oop) {
1047       // The oop result has been saved on the stack together with all
1048       // the other registers. In order to preserve it over GCs we need
1049       // to keep it in a handle.
1050       oop result = caller_fr.saved_oop_result(&map);
1051       assert(oopDesc::is_oop_or_null(result), "must be oop");
1052       return_value = Handle(thread(), result);
1053       assert(Universe::heap()->is_in_or_null(result), "must be heap pointer");
1054     }
1055 
1056     // Block the thread
1057     SafepointMechanism::block_if_requested(thread());
1058 
1059     // restore oop result, if any
1060     if (return_oop) {
1061       caller_fr.set_saved_oop_result(&map, return_value());



1062     }
1063   }
1064 
1065   // This is a safepoint poll. Verify the return address and block.
1066   else {
1067     set_at_poll_safepoint(true);
1068 
1069     // verify the blob built the "return address" correctly
1070     assert(real_return_addr == caller_fr.pc(), "must match");
1071 
1072     // Block the thread
1073     SafepointMechanism::block_if_requested(thread());
1074     set_at_poll_safepoint(false);
1075 
1076     // If we have a pending async exception deoptimize the frame
1077     // as otherwise we may never deliver it.
1078     if (thread()->has_async_condition()) {
1079       ThreadInVMfromJavaNoAsyncException __tiv(thread());
1080       Deoptimization::deoptimize_frame(thread(), caller_fr.id());
1081     }
1082 
1083     // If an exception has been installed we must check for a pending deoptimization
1084     // Deoptimize frame if exception has been thrown.
1085 
1086     if (thread()->has_pending_exception() ) {
1087       RegisterMap map(thread(), true);
1088       frame caller_fr = stub_fr.sender(&map);
1089       if (caller_fr.is_deoptimized_frame()) {
1090         // The exception patch will destroy registers that are still
1091         // live and will be needed during deoptimization. Defer the
1092         // Async exception should have deferred the exception until the
1093         // next safepoint which will be detected when we get into
1094         // the interpreter so if we have an exception now things
1095         // are messed up.
1096 
1097         fatal("Exception installed and deoptimization is pending");
1098       }
1099     }
1100   }
1101 }

1165            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns),
1166            (int64_t)(_last_safepoint_end_time_ns - _last_safepoint_begin_time_ns));
1167 
1168   ls.print_cr(INT32_FORMAT_W(16), _page_trap);
1169 }

1171 // This method will be called when VM exits. This tries to summarize the sampling.
1172 // Current thread may already be deleted, so don't use ResourceMark.
1173 void SafepointTracing::statistics_exit_log() {
1174   if (!log_is_enabled(Info, safepoint, stats)) {
1175     return;
1176   }
1177   for (int index = 0; index < VM_Operation::VMOp_Terminating; index++) {
1178     if (_op_count[index] != 0) {
1179       log_info(safepoint, stats)("%-28s" UINT64_FORMAT_W(10), VM_Operation::name(index),
1180                _op_count[index]);
1181     }
1182   }
1183 
1184   log_info(safepoint, stats)("VM operations coalesced during safepoint " INT64_FORMAT,
1185                               VMThread::get_coalesced_count());
1186   log_info(safepoint, stats)("Maximum sync time  " INT64_FORMAT" ns",
1187                               (int64_t)(_max_sync_time));
1188   log_info(safepoint, stats)("Maximum vm operation time (except for Exit VM operation)  "
1189                               INT64_FORMAT " ns",
1190                               (int64_t)(_max_vmop_time));
1191 }

1193 void SafepointTracing::begin(VM_Operation::VMOp_Type type) {
1194   _op_count[type]++;
1195   _current_type = type;
1196 
1197   // update the time stamp to begin recording safepoint time
1198   _last_safepoint_begin_time_ns = os::javaTimeNanos();
1199   _last_safepoint_sync_time_ns = 0;
1200   _last_safepoint_cleanup_time_ns = 0;
1201 
1202   _last_app_time_ns = _last_safepoint_begin_time_ns - _last_safepoint_end_time_ns;
1203   _last_safepoint_end_time_ns = 0;
1204 
1205   RuntimeService::record_safepoint_begin(_last_app_time_ns);
1206 }

1208 void SafepointTracing::synchronized(int nof_threads, int nof_running, int traps) {
1209   _last_safepoint_sync_time_ns = os::javaTimeNanos();
1210   _nof_threads = nof_threads;
1211   _nof_running = nof_running;
1212   _page_trap   = traps;
1213   RuntimeService::record_safepoint_synchronized(_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns);
1214 }

1216 void SafepointTracing::cleanup() {
1217   _last_safepoint_cleanup_time_ns = os::javaTimeNanos();
1218 }
1219 
1220 void SafepointTracing::end() {
1221   _last_safepoint_end_time_ns = os::javaTimeNanos();
1222 
1223   if (_max_sync_time < (_last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns)) {
1224     _max_sync_time = _last_safepoint_sync_time_ns - _last_safepoint_begin_time_ns;
1225   }
1226   if (_max_vmop_time < (_last_safepoint_end_time_ns - _last_safepoint_sync_time_ns)) {
1227     _max_vmop_time = _last_safepoint_end_time_ns - _last_safepoint_sync_time_ns;
1228   }
1229   if (log_is_enabled(Info, safepoint, stats)) {
1230     statistics_log();
1231   }
1232 
1233   log_info(safepoint)(
1234      "Safepoint \"%s\", "
1235      "Time since last: " JLONG_FORMAT " ns, "
1236      "Reaching safepoint: " JLONG_FORMAT " ns, "
1237      "At safepoint: " JLONG_FORMAT " ns, "
1238      "Total: " JLONG_FORMAT " ns",
1239       VM_Operation::name(_current_type),
1240       _last_app_time_ns,
1241       _last_safepoint_cleanup_time_ns - _last_safepoint_begin_time_ns,
1242       _last_safepoint_end_time_ns     - _last_safepoint_cleanup_time_ns,
1243       _last_safepoint_end_time_ns     - _last_safepoint_begin_time_ns
1244      );
1245 
1246   RuntimeService::record_safepoint_end(_last_safepoint_end_time_ns - _last_safepoint_cleanup_time_ns);
1247 }
