 108 void MemNode::dump_adr_type(const Node* mem, const TypePtr* adr_type, outputStream *st) {
 109   st->print(" @");
 110   if (adr_type == NULL) {
 111     st->print("NULL");
 112   } else {
 113     adr_type->dump_on(st);
 114     Compile* C = Compile::current();
 115     Compile::AliasType* atp = NULL;
 116     if (C->have_alias_type(adr_type))  atp = C->alias_type(adr_type);
 117     if (atp == NULL)
 118       st->print(", idx=?\?;");
 119     else if (atp->index() == Compile::AliasIdxBot)
 120       st->print(", idx=Bot;");
 121     else if (atp->index() == Compile::AliasIdxTop)
 122       st->print(", idx=Top;");
 123     else if (atp->index() == Compile::AliasIdxRaw)
 124       st->print(", idx=Raw;");
 125     else {
 126       ciField* field = atp->field();
 127       if (field) {
 128         st->print(", name=");
 129         field->print_name_on(st);
 130       }
 131       st->print(", idx=%d;", atp->index());
 132     }
 133   }
 134 }

 140 Node *MemNode::optimize_simple_memory_chain(Node *mchain, const TypeOopPtr *t_oop, Node *load, PhaseGVN *phase) {
 141   assert((t_oop != NULL), "sanity");
 142   bool is_instance = t_oop->is_known_instance_field();
 143   bool is_boxed_value_load = t_oop->is_ptr_to_boxed_value() &&
 144                              (load != NULL) && load->is_Load() &&
 145                              (phase->is_IterGVN() != NULL);
 146   if (!(is_instance || is_boxed_value_load))
 147     return mchain;  // don't try to optimize non-instance types
 148   uint instance_id = t_oop->instance_id();
 149   Node *start_mem = phase->C->start()->proj_out_or_null(TypeFunc::Memory);
 150   Node *prev = NULL;
 151   Node *result = mchain;
 152   while (prev != result) {
 153     prev = result;
 154     if (result == start_mem)
 155       break;  // hit one of our sentinels
 156     // skip over a call which does not affect this memory slice
 157     if (result->is_Proj() && result->as_Proj()->_con == TypeFunc::Memory) {
 158       Node *proj_in = result->in(0);
 159       if (proj_in->is_Allocate() && proj_in->_idx == instance_id) {
 160         break;  // hit one of our sentinels
 161       } else if (proj_in->is_Call()) {
 162         // ArrayCopyNodes processed here as well
 163         CallNode *call = proj_in->as_Call();
 164         if (!call->may_modify(t_oop, phase)) { // returns false for instances
 165           result = call->in(TypeFunc::Memory);
 166         }
 167       } else if (proj_in->is_Initialize()) {
 168         AllocateNode* alloc = proj_in->as_Initialize()->allocation();
 169         // Stop if this is the initialization for the object instance which
 170         // contains this memory slice, otherwise skip over it.
 171         if ((alloc == NULL) || (alloc->_idx == instance_id)) {
 172           break;
 173         }
 174         if (is_instance) {
 175           result = proj_in->in(TypeFunc::Memory);
 176         } else if (is_boxed_value_load) {
 177           Node* klass = alloc->in(AllocateNode::KlassNode);
 178           const TypeKlassPtr* tklass = phase->type(klass)->is_klassptr();
 179           if (tklass->klass_is_exact() && !tklass->klass()->equals(t_oop->klass())) {
 180             result = proj_in->in(TypeFunc::Memory); // not related allocation
 181           }
 182         }
 183       } else if (proj_in->is_MemBar()) {
 184         ArrayCopyNode* ac = NULL;
 185         if (ArrayCopyNode::may_modify(t_oop, proj_in->as_MemBar(), phase, ac)) {
 186           break;
 187         }
 188         result = proj_in->in(TypeFunc::Memory);
 189       } else {
 190         assert(false, "unexpected projection");
 191       }
 192     } else if (result->is_ClearArray()) {
 193       if (!is_instance || !ClearArrayNode::step_through(&result, instance_id, phase)) {
 194         // Can not bypass initialization of the instance
 195         // we are looking for.
 196         break;
 197       }
 198       // Otherwise skip it (the call updated 'result' value).
 199     } else if (result->is_MergeMem()) {
 200       result = step_through_mergemem(phase, result->as_MergeMem(), t_oop, NULL, tty);
 201     }
 202   }
 203   return result;
 204 }

 206 Node *MemNode::optimize_memory_chain(Node *mchain, const TypePtr *t_adr, Node *load, PhaseGVN *phase) {
 207   const TypeOopPtr* t_oop = t_adr->isa_oopptr();
 208   if (t_oop == NULL)
 209     return mchain;  // don't try to optimize non-oop types
 210   Node* result = optimize_simple_memory_chain(mchain, t_oop, load, phase);
 211   bool is_instance = t_oop->is_known_instance_field();
 212   PhaseIterGVN *igvn = phase->is_IterGVN();
 213   if (is_instance && igvn != NULL && result->is_Phi()) {
 214     PhiNode *mphi = result->as_Phi();
 215     assert(mphi->bottom_type() == Type::MEMORY, "memory phi required");
 216     const TypePtr *t = mphi->adr_type();
 217     if (t == TypePtr::BOTTOM || t == TypeRawPtr::BOTTOM ||
 218         (t->isa_oopptr() && !t->is_oopptr()->is_known_instance() &&
 219          t->is_oopptr()->cast_to_exactness(true)
 220            ->is_oopptr()->cast_to_ptr_type(t_oop->ptr())
 221             ->is_oopptr()->cast_to_instance_id(t_oop->instance_id()) == t_oop)) {
 222       // clone the Phi with our address type
 223       result = mphi->split_out_instance(t_adr, igvn);
 224     } else {
 225       assert(phase->C->get_alias_index(t) == phase->C->get_alias_index(t_adr), "correct memory chain");
 226     }
 227   }
 228   return result;
 229 }

 231 static Node *step_through_mergemem(PhaseGVN *phase, MergeMemNode *mmem,  const TypePtr *tp, const TypePtr *adr_check, outputStream *st) {
 232   uint alias_idx = phase->C->get_alias_index(tp);
 233   Node *mem = mmem;
 234 #ifdef ASSERT
 235   {
 236     // Check that current type is consistent with the alias index used during graph construction
 237     assert(alias_idx >= Compile::AliasIdxRaw, "must not be a bad alias_idx");
 238     bool consistent =  adr_check == NULL || adr_check->empty() ||
 239                        phase->C->must_alias(adr_check, alias_idx );
 240     // Sometimes dead array references collapse to a[-1], a[-2], or a[-3]
 241     if( !consistent && adr_check != NULL && !adr_check->empty() &&
 242         tp->isa_aryptr() &&        tp->offset() == Type::OffsetBot &&
 243         adr_check->isa_aryptr() && adr_check->offset() != Type::OffsetBot &&
 244         ( adr_check->offset() == arrayOopDesc::length_offset_in_bytes() ||
 245           adr_check->offset() == oopDesc::klass_offset_in_bytes() ||
 246           adr_check->offset() == oopDesc::mark_offset_in_bytes() ) ) {
 247       // don't assert if it is dead code.
 248       consistent = true;
 249     }
 250     if( !consistent ) {
 251       st->print("alias_idx==%d, adr_check==", alias_idx);
 252       if( adr_check == NULL ) {
 253         st->print("NULL");
 254       } else {
 255         adr_check->dump();
 256       }
 257       st->cr();
 258       print_alias_types();
 259       assert(consistent, "adr_check must match alias idx");
 260     }
 261   }
 262 #endif
 263   // TypeOopPtr::NOTNULL+any is an OOP with unknown offset - generally
 264   // means an array I have not precisely typed yet.  Do not do any
 265   // alias stuff with it any time soon.
 266   const TypeOopPtr *toop = tp->isa_oopptr();
 267   if( tp->base() != Type::AnyPtr &&
 268       !(toop &&
 269         toop->klass() != NULL &&
 270         toop->klass()->is_java_lang_Object() &&
 271         toop->offset() == Type::OffsetBot) ) {
 272     // compress paths and change unreachable cycles to TOP
 273     // If not, we can update the input infinitely along a MergeMem cycle
 274     // Equivalent code in PhiNode::Ideal
 275     Node* m  = phase->transform(mmem);
 276     // If transformed to a MergeMem, get the desired slice
 277     // Otherwise the returned node represents memory for every slice
 278     mem = (m->is_MergeMem())? m->as_MergeMem()->memory_at(alias_idx) : m;
 279     // Update input if it is progress over what we have now
 280   }
 281   return mem;
 282 }

 284 //--------------------------Ideal_common---------------------------------------
 285 // Look for degenerate control and memory inputs.  Bypass MergeMem inputs.
 286 // Unhook non-raw memories from complete (macro-expanded) initializations.
 287 Node *MemNode::Ideal_common(PhaseGVN *phase, bool can_reshape) {
 288   // If our control input is a dead region, kill all below the region
 289   Node *ctl = in(MemNode::Control);
 290   if (ctl && remove_dead_region(phase, can_reshape))
 291     return this;
 292   ctl = in(MemNode::Control);
 293   // Don't bother trying to transform a dead node
 294   if (ctl && ctl->is_top())  return NodeSentinel;
 295 
 296   PhaseIterGVN *igvn = phase->is_IterGVN();
 297   // Wait if control on the worklist.
 298   if (ctl && can_reshape && igvn != NULL) {
 299     Node* bol = NULL;
 300     Node* cmp = NULL;
 301     if (ctl->in(0)->is_If()) {
 302       assert(ctl->is_IfTrue() || ctl->is_IfFalse(), "sanity");
 303       bol = ctl->in(0)->in(1);
 304       if (bol->is_Bool())
 305         cmp = ctl->in(0)->in(1)->in(1);
 306     }
 307     if (igvn->_worklist.member(ctl) ||
 308         (bol != NULL && igvn->_worklist.member(bol)) ||
 309         (cmp != NULL && igvn->_worklist.member(cmp)) ) {
 310       // This control path may be dead.
 311       // Delay this memory node transformation until the control is processed.
 312       phase->is_IterGVN()->_worklist.push(this);
 313       return NodeSentinel; // caller will return NULL
 314     }
 315   }
 316   // Ignore if memory is dead, or self-loop
 317   Node *mem = in(MemNode::Memory);
 318   if (phase->type( mem ) == Type::TOP) return NodeSentinel; // caller will return NULL
 319   assert(mem != this, "dead loop in MemNode::Ideal");
 320 
 321   if (can_reshape && igvn != NULL && igvn->_worklist.member(mem)) {
 322     // This memory slice may be dead.
 323     // Delay this mem node transformation until the memory is processed.
 324     phase->is_IterGVN()->_worklist.push(this);
 325     return NodeSentinel; // caller will return NULL
 326   }
 327 
 328   Node *address = in(MemNode::Address);
 329   const Type *t_adr = phase->type(address);
 330   if (t_adr == Type::TOP)              return NodeSentinel; // caller will return NULL
 331 
 332   if (can_reshape && is_unsafe_access() && (t_adr == TypePtr::NULL_PTR)) {
 333     // Unsafe off-heap access with zero address. Remove access and other control users
 334     // to not confuse optimizations and add a HaltNode to fail if this is ever executed.
 335     assert(ctl != NULL, "unsafe accesses should be control dependent");
 336     for (DUIterator_Fast imax, i = ctl->fast_outs(imax); i < imax; i++) {
 337       Node* u = ctl->fast_out(i);
 338       if (u != ctl) {
 339         igvn->rehash_node_delayed(u);
 340         int nb = u->replace_edge(ctl, phase->C->top());
 341         --i, imax -= nb;
 342       }
 343     }
 344     Node* frame = igvn->transform(new ParmNode(phase->C->start(), TypeFunc::FramePtr));
 345     Node* halt = igvn->transform(new HaltNode(ctl, frame, "unsafe off-heap access with zero address"));
 346     phase->C->root()->add_req(halt);
 347     return this;
 348   }
 349 
 350   if (can_reshape && igvn != NULL &&
 351       (igvn->_worklist.member(address) ||
 352        (igvn->_worklist.size() > 0 && t_adr != adr_type())) ) {
 353     // The address's base and type may change when the address is processed.
 354     // Delay this mem node transformation until the address is processed.
 355     phase->is_IterGVN()->_worklist.push(this);
 356     return NodeSentinel; // caller will return NULL
 357   }
 358 
 359   // Do NOT remove or optimize the next lines: ensure a new alias index
 360   // is allocated for an oop pointer type before Escape Analysis.
 361   // Note: C++ will not remove it since the call has side effect.
 362   if (t_adr->isa_oopptr()) {
 363     int alias_idx = phase->C->get_alias_index(t_adr->is_ptr());
 364   }
 365 
 366   Node* base = NULL;
 367   if (address->is_AddP()) {
 368     base = address->in(AddPNode::Base);
 369   }
 370   if (base != NULL && phase->type(base)->higher_equal(TypePtr::NULL_PTR) &&
 371       !t_adr->isa_rawptr()) {
 372     // Note: raw address has TOP base and top->higher_equal(TypePtr::NULL_PTR) is true.
 373     // Skip this node optimization if its address has TOP base.
 374     return NodeSentinel; // caller will return NULL
 375   }
 376 
 377   // Avoid independent memory operations
 378   Node* old_mem = mem;
 379 
 380   // The code which unhooks non-raw memories from complete (macro-expanded)
 381   // initializations was removed. After macro-expansion all stores catched
 382   // by Initialize node became raw stores and there is no information
 383   // which memory slices they modify. So it is unsafe to move any memory
 384   // operation above these stores. Also in most cases hooked non-raw memories
 385   // were already unhooked by using information from detect_ptr_independence()
 386   // and find_previous_store().
 387 
 388   if (mem->is_MergeMem()) {
 389     MergeMemNode* mmem = mem->as_MergeMem();
 390     const TypePtr *tp = t_adr->is_ptr();
 391 
 392     mem = step_through_mergemem(phase, mmem, tp, adr_type(), tty);
 393   }
 394 
 395   if (mem != old_mem) {
 396     set_req(MemNode::Memory, mem);
 397     if (can_reshape && old_mem->outcnt() == 0 && igvn != NULL) {
 398       igvn->_worklist.push(old_mem);
 399     }
 400     if (phase->type(mem) == Type::TOP) return NodeSentinel;
 401     return this;
 402   }
 403 
 404   // let the subclass continue analyzing...
 405   return NULL;
 406 }

 408 // Helper function for proving some simple control dominations.
 409 // Attempt to prove that all control inputs of 'dom' dominate 'sub'.
 410 // Already assumes that 'dom' is available at 'sub', and that 'sub'
 411 // is not a constant (dominated by the method's StartNode).
 412 // Used by MemNode::find_previous_store to prove that the
 413 // control input of a memory operation predates (dominates)
 414 // an allocation it wants to look past.
 415 bool MemNode::all_controls_dominate(Node* dom, Node* sub) {
 416   if (dom == NULL || dom->is_top() || sub == NULL || sub->is_top())
 417     return false; // Conservative answer for dead code
 418 
 419   // Check 'dom'. Skip Proj and CatchProj nodes.
 420   dom = dom->find_exact_control(dom);
 421   if (dom == NULL || dom->is_top())
 422     return false; // Conservative answer for dead code
 423 
 424   if (dom == sub) {
 425     // For the case when, for example, 'sub' is Initialize and the original
 426     // 'dom' is Proj node of the 'sub'.
 427     return false;
 428   }
 429 
 430   if (dom->is_Con() || dom->is_Start() || dom->is_Root() || dom == sub)
 431     return true;
 432 
 433   // 'dom' dominates 'sub' if its control edge and control edges
 434   // of all its inputs dominate or equal to sub's control edge.
 435 
 436   // Currently 'sub' is either Allocate, Initialize or Start nodes.
 437   // Or Region for the check in LoadNode::Ideal();
 438   // 'sub' should have sub->in(0) != NULL.
 439   assert(sub->is_Allocate() || sub->is_Initialize() || sub->is_Start() ||
 440          sub->is_Region() || sub->is_Call(), "expecting only these nodes");
 441 
 442   // Get control edge of 'sub'.
 443   Node* orig_sub = sub;
 444   sub = sub->find_exact_control(sub->in(0));
 445   if (sub == NULL || sub->is_top())
 446     return false; // Conservative answer for dead code
 447 
 448   assert(sub->is_CFG(), "expecting control");
 449 
 450   if (sub == dom)
 451     return true;
 452 
 453   if (sub->is_Start() || sub->is_Root())
 454     return false;
 455 
 456   {
 457     // Check all control edges of 'dom'.
 458 
 459     ResourceMark rm;
 460     Arena* arena = Thread::current()->resource_area();
 461     Node_List nlist(arena);
 462     Unique_Node_List dom_list(arena);
 463 
 464     dom_list.push(dom);
 465     bool only_dominating_controls = false;
 466 
 467     for (uint next = 0; next < dom_list.size(); next++) {
 468       Node* n = dom_list.at(next);
 469       if (n == orig_sub)
 470         return false; // One of dom's inputs dominated by sub.
 471       if (!n->is_CFG() && n->pinned()) {
 472         // Check only own control edge for pinned non-control nodes.
 473         n = n->find_exact_control(n->in(0));
 474         if (n == NULL || n->is_top())
 475           return false; // Conservative answer for dead code
 476         assert(n->is_CFG(), "expecting control");
 477         dom_list.push(n);
 478       } else if (n->is_Con() || n->is_Start() || n->is_Root()) {
 479         only_dominating_controls = true;
 480       } else if (n->is_CFG()) {
 481         if (n->dominates(sub, nlist))
 482           only_dominating_controls = true;
 483         else
 484           return false;
 485       } else {
 486         // First, own control edge.
 487         Node* m = n->find_exact_control(n->in(0));
 488         if (m != NULL) {
 489           if (m->is_top())
 490             return false; // Conservative answer for dead code
 491           dom_list.push(m);
 492         }
 493         // Now, the rest of edges.
 494         uint cnt = n->req();
 495         for (uint i = 1; i < cnt; i++) {
 496           m = n->find_exact_control(n->in(i));
 497           if (m == NULL || m->is_top())
 498             continue;
 499           dom_list.push(m);
 500         }
 501       }
 502     }
 503     return only_dominating_controls;
 504   }
 505 }

 507 //---------------------detect_ptr_independence---------------------------------
 508 // Used by MemNode::find_previous_store to prove that two base
 509 // pointers are never equal.
 510 // The pointers are accompanied by their associated allocations,
 511 // if any, which have been previously discovered by the caller.
 512 bool MemNode::detect_ptr_independence(Node* p1, AllocateNode* a1,
 513                                       Node* p2, AllocateNode* a2,
 514                                       PhaseTransform* phase) {
 515   // Attempt to prove that these two pointers cannot be aliased.
 516   // They may both manifestly be allocations, and they should differ.
 517   // Or, if they are not both allocations, they can be distinct constants.
 518   // Otherwise, one is an allocation and the other a pre-existing value.
 519   if (a1 == NULL && a2 == NULL) {           // neither an allocation
 520     return (p1 != p2) && p1->is_Con() && p2->is_Con();
 521   } else if (a1 != NULL && a2 != NULL) {    // both allocations
 522     return (a1 != a2);
 523   } else if (a1 != NULL) {                  // one allocation a1
 524     // (Note:  p2->is_Con implies p2->in(0)->is_Root, which dominates.)
 525     return all_controls_dominate(p2, a1);
 526   } else { //(a2 != NULL)                   // one allocation a2
 527     return all_controls_dominate(p1, a2);
 528   }
 529   return false;
 530 }

 533 // Find an arraycopy that must have set (can_see_stored_value=true) or
 534 // could have set (can_see_stored_value=false) the value for this load
 535 Node* LoadNode::find_previous_arraycopy(PhaseTransform* phase, Node* ld_alloc, Node*& mem, bool can_see_stored_value) const {
 536   if (mem->is_Proj() && mem->in(0) != NULL && (mem->in(0)->Opcode() == Op_MemBarStoreStore ||
 537                                                mem->in(0)->Opcode() == Op_MemBarCPUOrder)) {
 538     Node* mb = mem->in(0);
 539     if (mb->in(0) != NULL && mb->in(0)->is_Proj() &&
 540         mb->in(0)->in(0) != NULL && mb->in(0)->in(0)->is_ArrayCopy()) {
 541       ArrayCopyNode* ac = mb->in(0)->in(0)->as_ArrayCopy();
 542       if (ac->is_clonebasic()) {
 543         AllocateNode* alloc = AllocateNode::Ideal_allocation(ac->in(ArrayCopyNode::Dest), phase);
 544         if (alloc != NULL && alloc == ld_alloc) {
 545           return ac;
 546         }
 547       }
 548     }
 549   } else if (mem->is_Proj() && mem->in(0) != NULL && mem->in(0)->is_ArrayCopy()) {
 550     ArrayCopyNode* ac = mem->in(0)->as_ArrayCopy();
 551 
 552     if (ac->is_arraycopy_validated() ||
 553         ac->is_copyof_validated() ||
 554         ac->is_copyofrange_validated()) {
 555       Node* ld_addp = in(MemNode::Address);
 556       if (ld_addp->is_AddP()) {
 557         Node* ld_base = ld_addp->in(AddPNode::Address);
 558         Node* ld_offs = ld_addp->in(AddPNode::Offset);
 559 
 560         Node* dest = ac->in(ArrayCopyNode::Dest);
 561 
 562         if (dest == ld_base) {
 563           const TypeX *ld_offs_t = phase->type(ld_offs)->isa_intptr_t();
 564           if (ac->modifies(ld_offs_t->_lo, ld_offs_t->_hi, phase, can_see_stored_value)) {
 565             return ac;
 566           }
 567           if (!can_see_stored_value) {
 568             mem = ac->in(TypeFunc::Memory);
 569           }
 570         }
 571       }
 572     }
 573   }
 574   return NULL;

 577 // The logic for reordering loads and stores uses four steps:
 578 // (a) Walk carefully past stores and initializations which we
 579 //     can prove are independent of this load.
 580 // (b) Observe that the next memory state makes an exact match
 581 //     with self (load or store), and locate the relevant store.
 582 // (c) Ensure that, if we were to wire self directly to the store,
 583 //     the optimizer would fold it up somehow.
 584 // (d) Do the rewiring, and return, depending on some other part of
 585 //     the optimizer to fold up the load.
 586 // This routine handles steps (a) and (b).  Steps (c) and (d) are
 587 // specific to loads and stores, so they are handled by the callers.
 588 // (Currently, only LoadNode::Ideal has steps (c), (d).  More later.)
 589 //
 590 Node* MemNode::find_previous_store(PhaseTransform* phase) {
 591   Node*         ctrl   = in(MemNode::Control);
 592   Node*         adr    = in(MemNode::Address);
 593   intptr_t      offset = 0;
 594   Node*         base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
 595   AllocateNode* alloc  = AllocateNode::Ideal_allocation(base, phase);
 596 
 597   if (offset == Type::OffsetBot)
 598     return NULL;            // cannot unalias unless there are precise offsets
 599 
 600   const bool adr_maybe_raw = check_if_adr_maybe_raw(adr);
 601   const TypeOopPtr *addr_t = adr->bottom_type()->isa_oopptr();
 602 
 603   intptr_t size_in_bytes = memory_size();
 604 
 605   Node* mem = in(MemNode::Memory);   // start searching here...
 606 
 607   int cnt = 50;             // Cycle limiter
 608   for (;;) {                // While we can dance past unrelated stores...
 609     if (--cnt < 0)  break;  // Caught in cycle or a complicated dance?
 610 
 611     Node* prev = mem;
 612     if (mem->is_Store()) {
 613       Node* st_adr = mem->in(MemNode::Address);
 614       intptr_t st_offset = 0;
 615       Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_offset);
 616       if (st_base == NULL)
 617         break;              // inscrutable pointer
 618 
 619       // For raw accesses it's not enough to prove that constant offsets don't intersect.
 620       // We need the bases to be the equal in order for the offset check to make sense.
 621       if ((adr_maybe_raw || check_if_adr_maybe_raw(st_adr)) && st_base != base) {
 622         break;
 623       }
 624 
 625       if (st_offset != offset && st_offset != Type::OffsetBot) {
 626         const int MAX_STORE = BytesPerLong;
 627         if (st_offset >= offset + size_in_bytes ||
 628             st_offset <= offset - MAX_STORE ||
 629             st_offset <= offset - mem->as_Store()->memory_size()) {
 630           // Success:  The offsets are provably independent.
 631           // (You may ask, why not just test st_offset != offset and be done?
 632           // The answer is that stores of different sizes can co-exist
 633           // in the same sequence of RawMem effects.  We sometimes initialize
 634           // a whole 'tile' of array elements with a single jint or jlong.)
 635           mem = mem->in(MemNode::Memory);
 636           continue;           // (a) advance through independent store memory
 637         }
 638       }
 639       if (st_base != base &&
 640           detect_ptr_independence(base, alloc,
 641                                   st_base,
 642                                   AllocateNode::Ideal_allocation(st_base, phase),
 643                                   phase)) {
 644         // Success:  The bases are provably independent.
 645         mem = mem->in(MemNode::Memory);
 646         continue;           // (a) advance through independent store memory
 647       }
 648 
 649       // (b) At this point, if the bases or offsets do not agree, we lose,
 650       // since we have not managed to prove 'this' and 'mem' independent.
 651       if (st_base == base && st_offset == offset) {
 652         return mem;         // let caller handle steps (c), (d)
 653       }
 654 
 655     } else if (mem->is_Proj() && mem->in(0)->is_Initialize()) {
 656       InitializeNode* st_init = mem->in(0)->as_Initialize();
 657       AllocateNode*  st_alloc = st_init->allocation();
 658       if (st_alloc == NULL)
 659         break;              // something degenerated
 660       bool known_identical = false;
 661       bool known_independent = false;
 662       if (alloc == st_alloc)
 663         known_identical = true;
 664       else if (alloc != NULL)
 665         known_independent = true;
 666       else if (all_controls_dominate(this, st_alloc))
 667         known_independent = true;
 668 
 669       if (known_independent) {
 670         // The bases are provably independent: Either they are
 671         // manifestly distinct allocations, or else the control
 672         // of this load dominates the store's allocation.
 673         int alias_idx = phase->C->get_alias_index(adr_type());
 674         if (alias_idx == Compile::AliasIdxRaw) {
 675           mem = st_alloc->in(TypeFunc::Memory);
 676         } else {
 677           mem = st_init->memory(alias_idx);
 678         }
 679         continue;           // (a) advance through independent store memory
 680       }
 681 
 682       // (b) at this point, if we are not looking at a store initializing
 683       // the same allocation we are loading from, we lose.
 684       if (known_identical) {
 685         // From caller, can_see_stored_value will consult find_captured_store.
 686         return mem;         // let caller handle steps (c), (d)
 687       }
 688 
 689     } else if (find_previous_arraycopy(phase, alloc, mem, false) != NULL) {
 690       if (prev != mem) {
 691         // Found an arraycopy but it doesn't affect that load
 692         continue;
 693       }
 694       // Found an arraycopy that may affect that load
 695       return mem;
 696     } else if (addr_t != NULL && addr_t->is_known_instance_field()) {
 697       // Can't use optimize_simple_memory_chain() since it needs PhaseGVN.
 698       if (mem->is_Proj() && mem->in(0)->is_Call()) {
 699         // ArrayCopyNodes processed here as well.
 700         CallNode *call = mem->in(0)->as_Call();
 701         if (!call->may_modify(addr_t, phase)) {
 702           mem = call->in(TypeFunc::Memory);
 703           continue;         // (a) advance through independent call memory
 704         }
 705       } else if (mem->is_Proj() && mem->in(0)->is_MemBar()) {
 706         ArrayCopyNode* ac = NULL;
 707         if (ArrayCopyNode::may_modify(addr_t, mem->in(0)->as_MemBar(), phase, ac)) {
 708           break;
 709         }
 710         mem = mem->in(0)->in(TypeFunc::Memory);
 711         continue;           // (a) advance through independent MemBar memory
 712       } else if (mem->is_ClearArray()) {
 713         if (ClearArrayNode::step_through(&mem, (uint)addr_t->instance_id(), phase)) {
 714           // (the call updated 'mem' value)
 715           continue;         // (a) advance through independent allocation memory
 716         } else {
 717           // Can not bypass initialization of the instance
 718           // we are looking for.
 719           return mem;
 720         }
 721       } else if (mem->is_MergeMem()) {
 722         int alias_idx = phase->C->get_alias_index(adr_type());
 723         mem = mem->as_MergeMem()->memory_at(alias_idx);
 724         continue;           // (a) advance through independent MergeMem memory
 725       }
 726     }
 727 
 728     // Unless there is an explicit 'continue', we must bail out here,
 729     // because 'mem' is an inscrutable memory state (e.g., a call).
 730     break;
 731   }
 732 
 733   return NULL;              // bail out
 734 }

 736 //----------------------calculate_adr_type-------------------------------------
 737 // Helper function.  Notices when the given type of address hits top or bottom.
 738 // Also, asserts a cross-check of the type against the expected address type.
 739 const TypePtr* MemNode::calculate_adr_type(const Type* t, const TypePtr* cross_check) {
 740   if (t == Type::TOP)  return NULL; // does not touch memory any more?
 741   #ifdef ASSERT
 742   if (!VerifyAliases || VMError::is_error_reported() || Node::in_dump())  cross_check = NULL;
 743   #endif
 744   const TypePtr* tp = t->isa_ptr();
 745   if (tp == NULL) {
 746     assert(cross_check == NULL || cross_check == TypePtr::BOTTOM, "expected memory type must be wide");
 747     return TypePtr::BOTTOM;           // touches lots of memory
 748   } else {
 749     #ifdef ASSERT
 750     // %%%% [phh] We don't check the alias index if cross_check is
 751     //            TypeRawPtr::BOTTOM.  Needs to be investigated.
 752     if (cross_check != NULL &&
 753         cross_check != TypePtr::BOTTOM &&
 754         cross_check != TypeRawPtr::BOTTOM) {
 755       // Recheck the alias index, to see if it has changed (due to a bug).
 756       Compile* C = Compile::current();

 760       // disregarding "null"-ness.

 764              "real address must not escape from expected memory type");
 765     }
 766     #endif
 767     return tp;
 768   }

 927 // Is the value loaded previously stored by an arraycopy? If so return
 928 // a load node that reads from the source array so we may be able to
 929 // optimize out the ArrayCopy node later.
 930 Node* LoadNode::can_see_arraycopy_value(Node* st, PhaseGVN* phase) const {
 931   Node* ld_adr = in(MemNode::Address);
 932   intptr_t ld_off = 0;
 933   AllocateNode* ld_alloc = AllocateNode::Ideal_allocation(ld_adr, phase, ld_off);
 934   Node* ac = find_previous_arraycopy(phase, ld_alloc, st, true);
 935   if (ac != NULL) {
 936     assert(ac->is_ArrayCopy(), "what kind of node can this be?");
 937 
 938     Node* mem = ac->in(TypeFunc::Memory);
 939     Node* ctl = ac->in(0);
 940     Node* src = ac->in(ArrayCopyNode::Src);
 941 
 942     if (!ac->as_ArrayCopy()->is_clonebasic() && !phase->type(src)->isa_aryptr()) {
 943       return NULL;
 944     }
 945 
 946     LoadNode* ld = clone()->as_Load();
 947     Node* addp = in(MemNode::Address)->clone();
 948     if (ac->as_ArrayCopy()->is_clonebasic()) {
 949       assert(ld_alloc != NULL, "need an alloc");
 950       assert(addp->is_AddP(), "address must be addp");
 951       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
 952       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Base)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 953       assert(bs->step_over_gc_barrier(addp->in(AddPNode::Address)) == bs->step_over_gc_barrier(ac->in(ArrayCopyNode::Dest)), "strange pattern");
 954       addp->set_req(AddPNode::Base, src);
 955       addp->set_req(AddPNode::Address, src);
 956     } else {
 957       assert(ac->as_ArrayCopy()->is_arraycopy_validated() ||
 958              ac->as_ArrayCopy()->is_copyof_validated() ||
 959              ac->as_ArrayCopy()->is_copyofrange_validated(), "only supported cases");
 960       assert(addp->in(AddPNode::Base) == addp->in(AddPNode::Address), "should be");
 961       addp->set_req(AddPNode::Base, src);
 962       addp->set_req(AddPNode::Address, src);
 963 
 964       const TypeAryPtr* ary_t = phase->type(in(MemNode::Address))->isa_aryptr();
 965       BasicType ary_elem  = ary_t->klass()->as_array_klass()->element_type()->basic_type();
 966       uint header = arrayOopDesc::base_offset_in_bytes(ary_elem);
 967       uint shift  = exact_log2(type2aelembytes(ary_elem));
 968       if (ary_t->klass()->is_flat_array_klass()) {
 969         ciFlatArrayKlass* vak = ary_t->klass()->as_flat_array_klass();
 970         shift = vak->log2_element_size();
 971       }
 972 
 973       Node* diff = phase->transform(new SubINode(ac->in(ArrayCopyNode::SrcPos), ac->in(ArrayCopyNode::DestPos)));
 974 #ifdef _LP64
 975       diff = phase->transform(new ConvI2LNode(diff));
 976 #endif
 977       diff = phase->transform(new LShiftXNode(diff, phase->intcon(shift)));
 978 
 979       Node* offset = phase->transform(new AddXNode(addp->in(AddPNode::Offset), diff));
 980       addp->set_req(AddPNode::Offset, offset);
 981     }
 982     addp = phase->transform(addp);
 983 #ifdef ASSERT
 984     const TypePtr* adr_type = phase->type(addp)->is_ptr();
 985     ld->_adr_type = adr_type;
 986 #endif
 987     ld->set_req(MemNode::Address, addp);
 988     ld->set_req(0, ctl);
 989     ld->set_req(MemNode::Memory, mem);
 990     // load depends on the tests that validate the arraycopy
 991     ld->_control_dependency = UnknownControl;
 992     return ld;
 993   }
 994   return NULL;
 995 }

 998 //---------------------------can_see_stored_value------------------------------
 999 // This routine exists to make sure this set of tests is done the same
1000 // everywhere.  We need to make a coordinated change: first LoadNode::Ideal
1001 // will change the graph shape in a way which makes memory alive twice at the
1002 // same time (uses the Oracle model of aliasing), then some
1003 // LoadXNode::Identity will fold things back to the equivalence-class model
1004 // of aliasing.
1005 Node* MemNode::can_see_stored_value(Node* st, PhaseTransform* phase) const {
1006   Node* ld_adr = in(MemNode::Address);
1007   intptr_t ld_off = 0;
1008   Node* ld_base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ld_off);
1009   Node* ld_alloc = AllocateNode::Ideal_allocation(ld_base, phase);
1010   const TypeInstPtr* tp = phase->type(ld_adr)->isa_instptr();
1011   Compile::AliasType* atp = (tp != NULL) ? phase->C->alias_type(tp) : NULL;
1012   // This is more general than load from boxing objects.
1013   if (skip_through_membars(atp, tp, phase->C->eliminate_boxing())) {
1014     uint alias_idx = atp->index();
1015     bool final = !atp->is_rewritable();
1016     Node* result = NULL;
1017     Node* current = st;
1018     // Skip through chains of MemBarNodes checking the MergeMems for
1019     // new states for the slice of this load.  Stop once any other
1020     // kind of node is encountered.  Loads from final memory can skip
1021     // through any kind of MemBar but normal loads shouldn't skip
1022     // through MemBarAcquire since the could allow them to move out of
1023     // a synchronized region.
1024     while (current->is_Proj()) {
1025       int opc = current->in(0)->Opcode();
1026       if ((final && (opc == Op_MemBarAcquire ||
1027                      opc == Op_MemBarAcquireLock ||
1028                      opc == Op_LoadFence)) ||
1029           opc == Op_MemBarRelease ||
1030           opc == Op_StoreFence ||
1031           opc == Op_MemBarReleaseLock ||
1032           opc == Op_MemBarStoreStore ||
1033           opc == Op_MemBarCPUOrder) {
1034         Node* mem = current->in(0)->in(TypeFunc::Memory);
1035         if (mem->is_MergeMem()) {
1036           MergeMemNode* merge = mem->as_MergeMem();
1037           Node* new_st = merge->memory_at(alias_idx);
1038           if (new_st == merge->base_memory()) {
1039             // Keep searching
1040             current = new_st;
1041             continue;
1042           }
1043           // Save the new memory state for the slice and fall through
1044           // to exit.
1045           result = new_st;
1046         }
1047       }
1048       break;
1049     }
1050     if (result != NULL) {
1051       st = result;
1052     }
1053   }
1054 
1055   // Loop around twice in the case Load -> Initialize -> Store.
1056   // (See PhaseIterGVN::add_users_to_worklist, which knows about this case.)
1057   for (int trip = 0; trip <= 1; trip++) {
1058 
1059     if (st->is_Store()) {
1060       Node* st_adr = st->in(MemNode::Address);
1061       if (!phase->eqv(st_adr, ld_adr)) {
1062         // Try harder before giving up. Unify base pointers with casts (e.g., raw/non-raw pointers).
1063         intptr_t st_off = 0;
1064         Node* st_base = AddPNode::Ideal_base_and_offset(st_adr, phase, st_off);
1065         if (ld_base == NULL)                                   return NULL;
1066         if (st_base == NULL)                                   return NULL;
1067         if (!ld_base->eqv_uncast(st_base, /*keep_deps=*/true)) return NULL;
1068         if (ld_off != st_off)                                  return NULL;
1069         if (ld_off == Type::OffsetBot)                         return NULL;
1070         // Same base, same offset.
1071         // Possible improvement for arrays: check index value instead of absolute offset.
1072 
1073         // At this point we have proven something like this setup:
1074         //   B = << base >>
1075         //   L =  LoadQ(AddP(Check/CastPP(B), #Off))
1076         //   S = StoreQ(AddP(             B , #Off), V)
1077         // (Actually, we haven't yet proven the Q's are the same.)
1078         // In other words, we are loading from a casted version of
1079         // the same pointer-and-offset that we stored to.
1080         // Casted version may carry a dependency and it is respected.
1081         // Thus, we are able to replace L by V.
1082       }
1083       // Now prove that we have a LoadQ matched to a StoreQ, for some Q.
1084       if (store_Opcode() != st->Opcode())
1085         return NULL;
1086       return st->in(MemNode::ValueIn);
1087     }
1088 
1089     // A load from a freshly-created object always returns zero.
1090     // (This can happen after LoadNode::Ideal resets the load's memory input
1091     // to find_captured_store, which returned InitializeNode::zero_memory.)
1092     if (st->is_Proj() && st->in(0)->is_Allocate() &&
1093         (st->in(0) == ld_alloc) &&
1094         (ld_off >= st->in(0)->as_Allocate()->minimum_header_size())) {
1095       // return a zero value for the load's basic type
1096       // (This is one of the few places where a generic PhaseTransform
1097       // can create new nodes.  Think of it as lazily manifesting
1098       // virtually pre-existing constants.)
1099       assert(memory_type() != T_INLINE_TYPE, "should not be used for inline types");
1100       Node* default_value = ld_alloc->in(AllocateNode::DefaultValue);
1101       if (default_value != NULL) {
1102         return default_value;
1103       }
1104       assert(ld_alloc->in(AllocateNode::RawDefaultValue) == NULL, "default value may not be null");
1105       return phase->zerocon(memory_type());
1106     }
1107 
1108     // A load from an initialization barrier can match a captured store.
1109     if (st->is_Proj() && st->in(0)->is_Initialize()) {
1110       InitializeNode* init = st->in(0)->as_Initialize();
1111       AllocateNode* alloc = init->allocation();
1112       if ((alloc != NULL) && (alloc == ld_alloc)) {
1113         // examine a captured store value
1114         st = init->find_captured_store(ld_off, memory_size(), phase);
1115         if (st != NULL) {
1116           continue;             // take one more trip around
1117         }
1118       }
1119     }
1120 
1121     // Load boxed value from result of valueOf() call is input parameter.
1122     if (this->is_Load() && ld_adr->is_AddP() &&
1123         (tp != NULL) && tp->is_ptr_to_boxed_value()) {
1124       intptr_t ignore = 0;
1125       Node* base = AddPNode::Ideal_base_and_offset(ld_adr, phase, ignore);
1126       BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
1127       base = bs->step_over_gc_barrier(base);
1128       if (base != NULL && base->is_Proj() &&
1129           base->as_Proj()->_con == TypeFunc::Parms &&
1130           base->in(0)->is_CallStaticJava() &&
1131           base->in(0)->as_CallStaticJava()->is_boxing_method()) {
1132         return base->in(0)->in(TypeFunc::Parms);
1133       }
1134     }
1135 
1136     break;
1137   }
1138 
1139   return NULL;
1140 }

1159 //------------------------------Identity---------------------------------------
1160 // Loads are identity if previous store is to same address
1161 Node* LoadNode::Identity(PhaseGVN* phase) {
1162   // Loading from a InlineTypePtr? The InlineTypePtr has the values of
1163   // all fields as input. Look for the field with matching offset.
1164   Node* addr = in(Address);
1165   intptr_t offset;
1166   Node* base = AddPNode::Ideal_base_and_offset(addr, phase, offset);
1167   if (base != NULL && base->is_InlineTypePtr() && offset > oopDesc::klass_offset_in_bytes()) {
1168     Node* value = base->as_InlineTypePtr()->field_value_by_offset((int)offset, true);
1169     if (value->is_InlineType()) {
1170       // Non-flattened inline type field
1171       InlineTypeNode* vt = value->as_InlineType();
1172       if (vt->is_allocated(phase)) {
1173         value = vt->get_oop();
1174       } else {
1175         // Not yet allocated, bail out
1176         value = NULL;
1177       }
1178     }
1179     if (value != NULL) {
1180       if (Opcode() == Op_LoadN) {
1181         // Encode oop value if we are loading a narrow oop
1182         assert(!phase->type(value)->isa_narrowoop(), "should already be decoded");
1183         value = phase->transform(new EncodePNode(value, bottom_type()));
1184       }
1185       return value;
1186     }
1187   }
1188 
1189   // If the previous store-maker is the right kind of Store, and the store is
1190   // to the same address, then we are equal to the value stored.
1191   Node* mem = in(Memory);
1192   Node* value = can_see_stored_value(mem, phase);
1193   if( value ) {
1194     // byte, short & char stores truncate naturally.
1195     // A load has to load the truncated value which requires
1196     // some sort of masking operation and that requires an
1197     // Ideal call instead of an Identity call.
1198     if (memory_size() < BytesPerInt) {
1199       // If the input to the store does not fit with the load's result type,
1200       // it must be truncated via an Ideal call.
1201       if (!phase->type(value)->higher_equal(phase->type(this)))
1202         return this;
1203     }
1204     // (This works even when value is a Con, but LoadNode::Value
1205     // usually runs first, producing the singleton type of the Con.)
1206     return value;
1207   }
1208 
1209   // Search for an existing data phi which was generated before for the same
1210   // instance's field to avoid infinite generation of phis in a loop.
1211   Node *region = mem->in(0);
1212   if (is_instance_field_load_with_local_phi(region)) {
1213     const TypeOopPtr *addr_t = in(Address)->bottom_type()->isa_oopptr();
1214     int this_index  = phase->C->get_alias_index(addr_t);
1215     int this_offset = addr_t->offset();
1216     int this_iid    = addr_t->instance_id();
1217     if (!addr_t->is_known_instance() &&
1218          addr_t->is_ptr_to_boxed_value()) {
1219       // Use _idx of address base (could be Phi node) for boxed values.
1220       intptr_t   ignore = 0;
1221       Node*      base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1222       if (base == NULL) {
1223         return this;
1224       }
1225       this_iid = base->_idx;
1226     }
1227     const Type* this_type = bottom_type();
1228     for (DUIterator_Fast imax, i = region->fast_outs(imax); i < imax; i++) {
1229       Node* phi = region->fast_out(i);
1230       if (phi->is_Phi() && phi != mem &&
1231           phi->as_Phi()->is_same_inst_field(this_type, (int)mem->_idx, this_iid, this_index, this_offset)) {
1232         return phi;
1233       }
1234     }
1235   }
1236 
1237   return this;
1238 }

1278 // We're loading from an object which has autobox behaviour.
1279 // If this object is result of a valueOf call we'll have a phi
1280 // merging a newly allocated object and a load from the cache.
1281 // We want to replace this load with the original incoming
1282 // argument to the valueOf call.
1283 Node* LoadNode::eliminate_autobox(PhaseGVN* phase) {
1284   assert(phase->C->eliminate_boxing(), "sanity");
1285   intptr_t ignore = 0;
1286   Node* base = AddPNode::Ideal_base_and_offset(in(Address), phase, ignore);
1287   if ((base == NULL) || base->is_Phi()) {
1288     // Push the loads from the phi that comes from valueOf up
1289     // through it to allow elimination of the loads and the recovery
1290     // of the original value. It is done in split_through_phi().
1291     return NULL;
1292   } else if (base->is_Load() ||
1293              (base->is_DecodeN() && base->in(1)->is_Load())) {
1294     // Eliminate the load of boxed value for integer types from the cache
1295     // array by deriving the value from the index into the array.
1296     // Capture the offset of the load and then reverse the computation.
1297 
1298     // Get LoadN node which loads a boxing object from 'cache' array.
1299     if (base->is_DecodeN()) {
1300       base = base->in(1);
1301     }
1302     if (!base->in(Address)->is_AddP()) {
1303       return NULL; // Complex address
1304     }
1305     AddPNode* address = base->in(Address)->as_AddP();
1306     Node* cache_base = address->in(AddPNode::Base);
1307     if ((cache_base != NULL) && cache_base->is_DecodeN()) {
1308       // Get ConP node which is static 'cache' field.
1309       cache_base = cache_base->in(1);
1310     }
1311     if ((cache_base != NULL) && cache_base->is_Con()) {
1312       const TypeAryPtr* base_type = cache_base->bottom_type()->isa_aryptr();
1313       if ((base_type != NULL) && base_type->is_autobox_cache()) {
1314         Node* elements[4];
1315         int shift = exact_log2(type2aelembytes(T_OBJECT));
1316         int count = address->unpack_offsets(elements, ARRAY_SIZE(elements));
1317         if (count > 0 && elements[0]->is_Con() &&
1318             (count == 1 ||
1319              (count == 2 && elements[1]->Opcode() == Op_LShiftX &&
1320                             elements[1]->in(2) == phase->intcon(shift)))) {
1321           ciObjArray* array = base_type->const_oop()->as_obj_array();
1322           // Fetch the box object cache[0] at the base of the array and get its value
1323           ciInstance* box = array->obj_at(0)->as_instance();
1324           ciInstanceKlass* ik = box->klass()->as_instance_klass();
1325           assert(ik->is_box_klass(), "sanity");
1326           assert(ik->nof_nonstatic_fields() == 1, "change following code");
1327           if (ik->nof_nonstatic_fields() == 1) {
1328             // This should be true nonstatic_field_at requires calling
1329             // nof_nonstatic_fields so check it anyway
1330             ciConstant c = box->field_value(ik->nonstatic_field_at(0));
1331             BasicType bt = c.basic_type();
1332             // Only integer types have boxing cache.
1333             assert(bt == T_BOOLEAN || bt == T_CHAR  ||
1334                    bt == T_BYTE    || bt == T_SHORT ||
1335                    bt == T_INT     || bt == T_LONG, "wrong type = %s", type2name(bt));
1336             jlong cache_low = (bt == T_LONG) ? c.as_long() : c.as_int();
1337             if (cache_low != (int)cache_low) {
1338               return NULL; // should not happen since cache is array indexed by value
1339             }
1340             jlong offset = arrayOopDesc::base_offset_in_bytes(T_OBJECT) - (cache_low << shift);
1341             if (offset != (int)offset) {
1342               return NULL; // should not happen since cache is array indexed by value
1343             }
1344            // Add up all the offsets making of the address of the load
1345             Node* result = elements[0];
1346             for (int i = 1; i < count; i++) {
1347               result = phase->transform(new AddXNode(result, elements[i]));
1348             }
1349             // Remove the constant offset from the address and then
1350             result = phase->transform(new AddXNode(result, phase->MakeConX(-(int)offset)));
1351             // remove the scaling of the offset to recover the original index.
1352             if (result->Opcode() == Op_LShiftX && result->in(2) == phase->intcon(shift)) {
1353               // Peel the shift off directly but wrap it in a dummy node
1354               // since Ideal can't return existing nodes
1355               result = new RShiftXNode(result->in(1), phase->intcon(0));
1356             } else if (result->is_Add() && result->in(2)->is_Con() &&
1357                        result->in(1)->Opcode() == Op_LShiftX &&
1358                        result->in(1)->in(2) == phase->intcon(shift)) {
1359               // We can't do general optimization: ((X<<Z) + Y) >> Z ==> X + (Y>>Z)
1360               // but for boxing cache access we know that X<<Z will not overflow
1361               // (there is range check) so we do this optimizatrion by hand here.
1362               Node* add_con = new RShiftXNode(result->in(2), phase->intcon(shift));
1363               result = new AddXNode(result->in(1)->in(1), phase->transform(add_con));
1364             } else {
1365               result = new RShiftXNode(result, phase->intcon(shift));
1366             }
1367 #ifdef _LP64
1368             if (bt != T_LONG) {
1369               result = new ConvL2INode(phase->transform(result));
1370             }
1371 #else
1372             if (bt == T_LONG) {
1373               result = new ConvI2LNode(phase->transform(result));
1374             }
1375 #endif
1376             // Boxing/unboxing can be done from signed & unsigned loads (e.g. LoadUB -> ... -> LoadB pair).
1377             // Need to preserve unboxing load type if it is unsigned.
1378             switch(this->Opcode()) {
1379               case Op_LoadUB:
1380                 result = new AndINode(phase->transform(result), phase->intcon(0xFF));
1381                 break;
1382               case Op_LoadUS:
1383                 result = new AndINode(phase->transform(result), phase->intcon(0xFFFF));
1384                 break;
1385             }
1386             return result;
1387           }
1388         }
1389       }
1390     }
1391   }
1392   return NULL;
1393 }

1395 static bool stable_phi(PhiNode* phi, PhaseGVN *phase) {
1396   Node* region = phi->in(0);
1397   if (region == NULL) {
1398     return false; // Wait stable graph
1399   }
1400   uint cnt = phi->req();
1401   for (uint i = 1; i < cnt; i++) {
1402     Node* rc = region->in(i);
1403     if (rc == NULL || phase->type(rc) == Type::TOP)
1404       return false; // Wait stable graph
1405     Node* in = phi->in(i);
1406     if (in == NULL || phase->type(in) == Type::TOP)
1407       return false; // Wait stable graph
1408   }
1409   return true;

1411 //------------------------------split_through_phi------------------------------
1412 // Split instance or boxed field load through Phi.
1413 Node *LoadNode::split_through_phi(PhaseGVN *phase) {
1414   Node* mem     = in(Memory);
1415   Node* address = in(Address);
1416   const TypeOopPtr *t_oop = phase->type(address)->isa_oopptr();
1417 
1418   assert((t_oop != NULL) &&
1419          (t_oop->is_known_instance_field() ||
1420           t_oop->is_ptr_to_boxed_value()), "invalide conditions");
1421 
1422   Compile* C = phase->C;
1423   intptr_t ignore = 0;
1424   Node*    base = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1425   bool base_is_phi = (base != NULL) && base->is_Phi();
1426   bool load_boxed_values = t_oop->is_ptr_to_boxed_value() && C->aggressive_unboxing() &&
1427                            (base != NULL) && (base == address->in(AddPNode::Base)) &&
1428                            phase->type(base)->higher_equal(TypePtr::NOTNULL);
1429 
1430   if (!((mem->is_Phi() || base_is_phi) &&
1431         (load_boxed_values || t_oop->is_known_instance_field()))) {
1432     return NULL; // memory is not Phi
1433   }
1434 
1435   if (mem->is_Phi()) {
1436     if (!stable_phi(mem->as_Phi(), phase)) {
1437       return NULL; // Wait stable graph
1438     }
1439     uint cnt = mem->req();
1440     // Check for loop invariant memory.
1441     if (cnt == 3) {
1442       for (uint i = 1; i < cnt; i++) {
1443         Node* in = mem->in(i);
1444         Node*  m = optimize_memory_chain(in, t_oop, this, phase);
1445         if (m == mem) {
1446           if (i == 1) {
1447             // if the first edge was a loop, check second edge too.
1448             // If both are replaceable - we are in an infinite loop
1449             Node *n = optimize_memory_chain(mem->in(2), t_oop, this, phase);
1450             if (n == mem) {
1451               break;
1452             }
1453           }
1454           set_req(Memory, mem->in(cnt - i));
1455           return this; // made change
1456         }
1457       }
1458     }
1459   }
1460   if (base_is_phi) {
1461     if (!stable_phi(base->as_Phi(), phase)) {
1462       return NULL; // Wait stable graph
1463     }
1464     uint cnt = base->req();
1465     // Check for loop invariant memory.
1466     if (cnt == 3) {
1467       for (uint i = 1; i < cnt; i++) {
1468         if (base->in(i) == base) {
1469           return NULL; // Wait stable graph
1470         }
1471       }
1472     }
1473   }
1474 
1475   // Split through Phi (see original code in loopopts.cpp).
1476   assert(C->have_alias_type(t_oop), "instance should have alias type");
1477 
1478   // Do nothing here if Identity will find a value
1479   // (to avoid infinite chain of value phis generation).
1480   if (!phase->eqv(this, this->Identity(phase))) {
1481     return NULL;
1482   }
1483 
1484   // Select Region to split through.
1485   Node* region;
1486   if (!base_is_phi) {
1487     assert(mem->is_Phi(), "sanity");
1488     region = mem->in(0);
1489     // Skip if the region dominates some control edge of the address.
1490     if (!MemNode::all_controls_dominate(address, region))
1491       return NULL;
1492   } else if (!mem->is_Phi()) {
1493     assert(base_is_phi, "sanity");
1494     region = base->in(0);
1495     // Skip if the region dominates some control edge of the memory.
1496     if (!MemNode::all_controls_dominate(mem, region))
1497       return NULL;
1498   } else if (base->in(0) != mem->in(0)) {
1499     assert(base_is_phi && mem->is_Phi(), "sanity");
1500     if (MemNode::all_controls_dominate(mem, base->in(0))) {
1501       region = base->in(0);
1502     } else if (MemNode::all_controls_dominate(address, mem->in(0))) {
1503       region = mem->in(0);
1504     } else {
1505       return NULL; // complex graph
1506     }
1507   } else {
1508     assert(base->in(0) == mem->in(0), "sanity");
1509     region = mem->in(0);
1510   }
1511 
1512   const Type* this_type = this->bottom_type();
1513   int this_index  = C->get_alias_index(t_oop);
1514   int this_offset = t_oop->offset();
1515   int this_iid    = t_oop->instance_id();
1516   if (!t_oop->is_known_instance() && load_boxed_values) {
1517     // Use _idx of address base for boxed values.
1518     this_iid = base->_idx;
1519   }
1520   PhaseIterGVN* igvn = phase->is_IterGVN();
1521   Node* phi = new PhiNode(region, this_type, NULL, mem->_idx, this_iid, this_index, this_offset);
1522   for (uint i = 1; i < region->req(); i++) {
1523     Node* x;
1524     Node* the_clone = NULL;
1525     Node* in = region->in(i);
1526     if (region->is_CountedLoop() && region->as_Loop()->is_strip_mined() && i == LoopNode::EntryControl &&
1527         in != NULL && in->is_OuterStripMinedLoop()) {
1528       // No node should go in the outer strip mined loop
1529       in = in->in(LoopNode::EntryControl);
1530     }
1531     if (in == NULL || in == C->top()) {
1532       x = C->top();      // Dead path?  Use a dead data op
1533     } else {
1534       x = this->clone();        // Else clone up the data op
1535       the_clone = x;            // Remember for possible deletion.
1536       // Alter data node to use pre-phi inputs
1537       if (this->in(0) == region) {
1538         x->set_req(0, in);
1539       } else {
1540         x->set_req(0, NULL);
1541       }
1542       if (mem->is_Phi() && (mem->in(0) == region)) {
1543         x->set_req(Memory, mem->in(i)); // Use pre-Phi input for the clone.
1544       }
1545       if (address->is_Phi() && address->in(0) == region) {
1546         x->set_req(Address, address->in(i)); // Use pre-Phi input for the clone
1547       }
1548       if (base_is_phi && (base->in(0) == region)) {
1549         Node* base_x = base->in(i); // Clone address for loads from boxed objects.
1550         Node* adr_x = phase->transform(new AddPNode(base_x,base_x,address->in(AddPNode::Offset)));
1551         x->set_req(Address, adr_x);
1552       }
1553     }
1554     // Check for a 'win' on some paths
1555     const Type *t = x->Value(igvn);
1556 
1557     bool singleton = t->singleton();
1558 
1559     // See comments in PhaseIdealLoop::split_thru_phi().
1560     if (singleton && t == Type::TOP) {
1561       singleton &= region->is_Loop() && (i != LoopNode::EntryControl);
1562     }
1563 
1564     if (singleton) {
1565       x = igvn->makecon(t);
1566     } else {
1567       // We now call Identity to try to simplify the cloned node.
1568       // Note that some Identity methods call phase->type(this).
1569       // Make sure that the type array is big enough for
1570       // our new node, even though we may throw the node away.
1571       // (This tweaking with igvn only works because x is a new node.)
1572       igvn->set_type(x, t);
1573       // If x is a TypeNode, capture any more-precise type permanently into Node
1574       // otherwise it will be not updated during igvn->transform since
1575       // igvn->type(x) is set to x->Value() already.
1576       x->raise_bottom_type(t);
1577       Node* y = x->Identity(igvn);
1578       if (y != x) {
1579         x = y;
1580       } else {
1581         y = igvn->hash_find_insert(x);
1582         if (y) {
1583           x = y;
1584         } else {
1585           // Else x is a new node we are keeping
1586           // We do not need register_new_node_with_optimizer
1587           // because set_type has already been called.
1588           igvn->_worklist.push(x);
1589         }
1590       }
1591     }
1592     if (x != the_clone && the_clone != NULL) {
1593       igvn->remove_dead_node(the_clone);
1594     }
1595     phi->set_req(i, x);
1596   }
1597   // Record Phi
1598   igvn->register_new_node_with_optimizer(phi);
1599   return phi;

1618 //------------------------------Ideal------------------------------------------
1619 // If the load is from Field memory and the pointer is non-null, it might be possible to
1620 // zero out the control input.
1621 // If the offset is constant and the base is an object allocation,
1622 // try to hook me up to the exact initializing store.
1623 Node *LoadNode::Ideal(PhaseGVN *phase, bool can_reshape) {
1624   Node* p = MemNode::Ideal_common(phase, can_reshape);
1625   if (p)  return (p == NodeSentinel) ? NULL : p;
1626 
1627   Node* ctrl    = in(MemNode::Control);
1628   Node* address = in(MemNode::Address);
1629   bool progress = false;
1630 
1631   bool addr_mark = ((phase->type(address)->isa_oopptr() || phase->type(address)->isa_narrowoop()) &&
1632          phase->type(address)->is_ptr()->offset() == oopDesc::mark_offset_in_bytes());
1633 
1634   // Skip up past a SafePoint control.  Cannot do this for Stores because
1635   // pointer stores & cardmarks must stay on the same side of a SafePoint.
1636   if( ctrl != NULL && ctrl->Opcode() == Op_SafePoint &&
1637       phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw  &&
1638       !addr_mark &&
1639       (depends_only_on_test() || has_unknown_control_dependency())) {
1640     ctrl = ctrl->in(0);
1641     set_req(MemNode::Control,ctrl);
1642     progress = true;
1643   }
1644 
1645   intptr_t ignore = 0;
1646   Node*    base   = AddPNode::Ideal_base_and_offset(address, phase, ignore);
1647   if (base != NULL
1648       && phase->C->get_alias_index(phase->type(address)->is_ptr()) != Compile::AliasIdxRaw) {
1649     // Check for useless control edge in some common special cases
1650     if (in(MemNode::Control) != NULL
1651         && can_remove_control()
1652         && phase->type(base)->higher_equal(TypePtr::NOTNULL)
1653         && all_controls_dominate(base, phase->C->start())) {
1654       // A method-invariant, non-null address (constant or 'this' argument).
1655       set_req(MemNode::Control, NULL);
1656       progress = true;
1657     }
1658   }
1659 
1660   Node* mem = in(MemNode::Memory);
1661   const TypePtr *addr_t = phase->type(address)->isa_ptr();
1662 
1663   if (can_reshape && (addr_t != NULL)) {
1664     // try to optimize our memory input
1665     Node* opt_mem = MemNode::optimize_memory_chain(mem, addr_t, this, phase);
1666     if (opt_mem != mem) {
1667       set_req(MemNode::Memory, opt_mem);
1668       if (phase->type( opt_mem ) == Type::TOP) return NULL;
1669       return this;
1670     }
1671     const TypeOopPtr *t_oop = addr_t->isa_oopptr();
1672     if ((t_oop != NULL) &&
1673         (t_oop->is_known_instance_field() ||
1674          t_oop->is_ptr_to_boxed_value())) {
1675       PhaseIterGVN *igvn = phase->is_IterGVN();
1676       if (igvn != NULL && igvn->_worklist.member(opt_mem)) {
1677         // Delay this transformation until memory Phi is processed.
1678         phase->is_IterGVN()->_worklist.push(this);
1679         return NULL;
1680       }
1681       // Split instance field load through Phi.
1682       Node* result = split_through_phi(phase);
1683       if (result != NULL) return result;
1684 
1685       if (t_oop->is_ptr_to_boxed_value()) {
1686         Node* result = eliminate_autobox(phase);
1687         if (result != NULL) return result;
1688       }
1689     }
1690   }
1691 
1692   // Is there a dominating load that loads the same value?  Leave
1693   // anything that is not a load of a field/array element (like
1694   // barriers etc.) alone
1695   if (in(0) != NULL && !adr_type()->isa_rawptr() && can_reshape) {
1696     for (DUIterator_Fast imax, i = mem->fast_outs(imax); i < imax; i++) {
1697       Node *use = mem->fast_out(i);
1698       if (use != this &&
1699           use->Opcode() == Opcode() &&
1700           use->in(0) != NULL &&
1701           use->in(0) != in(0) &&
1702           use->in(Address) == in(Address)) {
1703         Node* ctl = in(0);
1704         for (int i = 0; i < 10 && ctl != NULL; i++) {
1705           ctl = IfNode::up_one_dom(ctl);
1706           if (ctl == use->in(0)) {
1707             set_req(0, use->in(0));
1708             return this;
1709           }
1710         }
1711       }
1712     }
1713   }
1714 
1715   // Check for prior store with a different base or offset; make Load
1716   // independent.  Skip through any number of them.  Bail out if the stores
1717   // are in an endless dead cycle and report no progress.  This is a key
1718   // transform for Reflection.  However, if after skipping through the Stores
1719   // we can't then fold up against a prior store do NOT do the transform as
1720   // this amounts to using the 'Oracle' model of aliasing.  It leaves the same
1721   // array memory alive twice: once for the hoisted Load and again after the
1722   // bypassed Store.  This situation only works if EVERYBODY who does
1723   // anti-dependence work knows how to bypass.  I.e. we need all
1724   // anti-dependence checks to ask the same Oracle.  Right now, that Oracle is
1725   // the alias index stuff.  So instead, peek through Stores and IFF we can
1726   // fold up, do so.
1727   Node* prev_mem = find_previous_store(phase);
1728   if (prev_mem != NULL) {
1729     Node* value = can_see_arraycopy_value(prev_mem, phase);
1730     if (value != NULL) {
1731       return value;
1732     }
1733   }
1734   // Steps (a), (b):  Walk past independent stores to find an exact match.
1735   if (prev_mem != NULL && prev_mem != in(MemNode::Memory)) {
1736     // (c) See if we can fold up on the spot, but don't fold up here.
1737     // Fold-up might require truncation (for LoadB/LoadS/LoadUS) or
1738     // just return a prior value, which is done by Identity calls.
1739     if (can_see_stored_value(prev_mem, phase)) {
1740       // Make ready for step (d):
1741       set_req(MemNode::Memory, prev_mem);
1742       return this;
1743     }
1744   }
1745 
1746   AllocateNode* alloc = AllocateNode::Ideal_allocation(address, phase);
1747   if (alloc != NULL && mem->is_Proj() &&
1748       mem->in(0) != NULL &&
1749       mem->in(0) == alloc->initialization() &&
1750       Opcode() == Op_LoadX &&
1751       alloc->initialization()->proj_out_or_null(0) != NULL) {
1752     InitializeNode* init = alloc->initialization();
1753     Node* control = init->proj_out(0);
1754     return alloc->make_ideal_mark(phase, control, mem);
1755   }
1756 
1757   return progress ? this : NULL;
1758 }

1760 // Helper to recognize certain Klass fields which are invariant across
1761 // some group of array types (e.g., int[] or all T[] where T < Object).
1762 const Type*
1763 LoadNode::load_array_final_field(const TypeKlassPtr *tkls,
1764                                  ciKlass* klass) const {
1765   if (tkls->offset() == in_bytes(Klass::modifier_flags_offset())) {
1766     // The field is Klass::_modifier_flags.  Return its (constant) value.
1767     // (Folds up the 2nd indirection in aClassConstant.getModifiers().)
1768     assert(this->Opcode() == Op_LoadI, "must load an int from _modifier_flags");
1769     return TypeInt::make(klass->modifier_flags());
1770   }
1771   if (tkls->offset() == in_bytes(Klass::access_flags_offset())) {
1772     // The field is Klass::_access_flags.  Return its (constant) value.
1773     // (Folds up the 2nd indirection in Reflection.getClassAccessFlags(aClassConstant).)
1774     assert(this->Opcode() == Op_LoadI, "must load an int from _access_flags");
1775     return TypeInt::make(klass->access_flags());
1776   }
1777   if (tkls->offset() == in_bytes(Klass::layout_helper_offset())) {
1778     // The field is Klass::_layout_helper.  Return its constant value if known.
1779     assert(this->Opcode() == Op_LoadI, "must load an int from _layout_helper");
1780     return TypeInt::make(klass->layout_helper());
1781   }
1782 
1783   // No match.
1784   return NULL;
1785 }

1787 //------------------------------Value-----------------------------------------
1788 const Type* LoadNode::Value(PhaseGVN* phase) const {
1789   // Either input is TOP ==> the result is TOP
1790   Node* mem = in(MemNode::Memory);
1791   const Type *t1 = phase->type(mem);
1792   if (t1 == Type::TOP)  return Type::TOP;
1793   Node* adr = in(MemNode::Address);
1794   const TypePtr* tp = phase->type(adr)->isa_ptr();
1795   if (tp == NULL || tp->empty())  return Type::TOP;
1796   int off = tp->offset();
1797   assert(off != Type::OffsetTop, "case covered by TypePtr::empty");
1798   Compile* C = phase->C;
1799 
1800   // Try to guess loaded type from pointer type
1801   if (tp->isa_aryptr()) {
1802     const TypeAryPtr* ary = tp->is_aryptr();
1803     const Type* t = ary->elem();
1804 
1805     // Determine whether the reference is beyond the header or not, by comparing
1806     // the offset against the offset of the start of the array's data.
1807     // Different array types begin at slightly different offsets (12 vs. 16).
1808     // We choose T_BYTE as an example base type that is least restrictive
1809     // as to alignment, which will therefore produce the smallest
1810     // possible base offset.
1811     const int min_base_off = arrayOopDesc::base_offset_in_bytes(T_BYTE);
1812     const bool off_beyond_header = (off >= min_base_off);
1813 
1814     // Try to constant-fold a stable array element.
1815     if (FoldStableValues && !is_mismatched_access() && ary->is_stable()) {
1816       // Make sure the reference is not into the header and the offset is constant
1817       ciObject* aobj = ary->const_oop();
1818       if (aobj != NULL && off_beyond_header && adr->is_AddP() && off != Type::OffsetBot) {
1819         int stable_dimension = (ary->stable_dimension() > 0 ? ary->stable_dimension() - 1 : 0);
1820         const Type* con_type = Type::make_constant_from_array_element(aobj->as_array(), off,
1821                                                                       stable_dimension,
1822                                                                       memory_type(), is_unsigned());
1823         if (con_type != NULL) {
1824           return con_type;
1825         }
1826       }
1827     }
1828 
1829     // Don't do this for integer types. There is only potential profit if
1830     // the element type t is lower than _type; that is, for int types, if _type is
1831     // more restrictive than t.  This only happens here if one is short and the other
1832     // char (both 16 bits), and in those cases we've made an intentional decision
1833     // to use one kind of load over the other. See AndINode::Ideal and 4965907.
1834     // Also, do not try to narrow the type for a LoadKlass, regardless of offset.
1835     //
1836     // Yes, it is possible to encounter an expression like (LoadKlass p1:(AddP x x 8))
1837     // where the _gvn.type of the AddP is wider than 8.  This occurs when an earlier
1838     // copy p0 of (AddP x x 8) has been proven equal to p1, and the p0 has been
1839     // subsumed by p1.  If p1 is on the worklist but has not yet been re-transformed,
1840     // it is possible that p1 will have a type like Foo*[int+]:NotNull*+any.
1841     // In fact, that could have been the original type of p1, and p1 could have
1842     // had an original form like p1:(AddP x x (LShiftL quux 3)), where the
1843     // expression (LShiftL quux 3) independently optimized to the constant 8.
1844     if ((t->isa_int() == NULL) && (t->isa_long() == NULL)
1845         && (_type->isa_vect() == NULL)
1846         && t->isa_inlinetype() == NULL
1847         && Opcode() != Op_LoadKlass && Opcode() != Op_LoadNKlass) {
1848       // t might actually be lower than _type, if _type is a unique
1849       // concrete subclass of abstract class t.
1850       if (off_beyond_header || off == Type::OffsetBot) {  // is the offset beyond the header?
1851         const Type* jt = t->join_speculative(_type);
1852         // In any case, do not allow the join, per se, to empty out the type.
1853         if (jt->empty() && !t->empty()) {
1854           // This can happen if a interface-typed array narrows to a class type.
1855           jt = _type;
1856         }
1857 #ifdef ASSERT
1858         if (phase->C->eliminate_boxing() && adr->is_AddP()) {
1859           // The pointers in the autobox arrays are always non-null
1860           Node* base = adr->in(AddPNode::Base);
1861           if ((base != NULL) && base->is_DecodeN()) {
1862             // Get LoadN node which loads IntegerCache.cache field
1863             base = base->in(1);
1864           }
1865           if ((base != NULL) && base->is_Con()) {
1866             const TypeAryPtr* base_type = base->bottom_type()->isa_aryptr();
1867             if ((base_type != NULL) && base_type->is_autobox_cache()) {
1868               // It could be narrow oop
1869               assert(jt->make_ptr()->ptr() == TypePtr::NotNull,"sanity");
1870             }
1871           }
1872         }
1873 #endif
1874         return jt;
1875       }
1876     }
1877   } else if (tp->base() == Type::InstPtr) {
1878     assert( off != Type::OffsetBot ||
1879             // arrays can be cast to Objects
1880             tp->is_oopptr()->klass()->is_java_lang_Object() ||
1881             tp->is_oopptr()->klass() == ciEnv::current()->Class_klass() ||
1882             // unsafe field access may not have a constant offset
1883             C->has_unsafe_access(),
1884             "Field accesses must be precise" );
1885     // For oop loads, we expect the _type to be precise.
1886 
1887     const TypeInstPtr* tinst = tp->is_instptr();
1888     BasicType bt = memory_type();
1889 
1890     // Optimize loads from constant fields.
1891     ciObject* const_oop = tinst->const_oop();
1892     if (!is_mismatched_access() && off != Type::OffsetBot && const_oop != NULL && const_oop->is_instance()) {
1893       ciType* mirror_type = const_oop->as_instance()->java_mirror_type();
1894       if (mirror_type != NULL && mirror_type->is_inlinetype()) {
1895         ciInlineKlass* vk = mirror_type->as_inline_klass();
1896         if (off == vk->default_value_offset()) {
1897           // Loading a special hidden field that contains the oop of the default inline type
1898           const Type* const_oop = TypeInstPtr::make(vk->default_instance());
1899           return (bt == T_NARROWOOP) ? const_oop->make_narrowoop() : const_oop;
1900         }
1901       }
1902       const Type* con_type = Type::make_constant_from_field(const_oop->as_instance(), off, is_unsigned(), bt);
1903       if (con_type != NULL) {
1904         return con_type;
1905       }
1906     }
1907   } else if (tp->base() == Type::KlassPtr) {
1908     assert( off != Type::OffsetBot ||
1909             // arrays can be cast to Objects
1910             tp->is_klassptr()->klass() == NULL ||
1911             tp->is_klassptr()->klass()->is_java_lang_Object() ||
1912             // also allow array-loading from the primary supertype
1913             // array during subtype checks
1914             Opcode() == Op_LoadKlass,
1915             "Field accesses must be precise" );
1916     // For klass/static loads, we expect the _type to be precise
1917   } else if (tp->base() == Type::RawPtr && !StressReflectiveCode) {
1918     if (adr->is_Load() && off == 0) {
1919       /* With mirrors being an indirect in the Klass*
1920        * the VM is now using two loads. LoadKlass(LoadP(LoadP(Klass, mirror_offset), zero_offset))
1921        * The LoadP from the Klass has a RawPtr type (see LibraryCallKit::load_mirror_from_klass).
1922        *
1923        * So check the type and klass of the node before the LoadP.
1924        */
1925       Node* adr2 = adr->in(MemNode::Address);
1926       const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
1927       if (tkls != NULL) {
1928         ciKlass* klass = tkls->klass();
1929         if (klass != NULL && klass->is_loaded() && tkls->klass_is_exact() && tkls->offset() == in_bytes(Klass::java_mirror_offset())) {
1930           assert(adr->Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1931           assert(Opcode() == Op_LoadP, "must load an oop from _java_mirror");
1932           return TypeInstPtr::make(klass->java_mirror());
1933         }
1934       }
1935     } else {
1936       // Check for a load of the default value offset from the InlineKlassFixedBlock:
1937       // LoadI(LoadP(inline_klass, adr_inlineklass_fixed_block_offset), default_value_offset_offset)
1938       intptr_t offset = 0;
1939       Node* base = AddPNode::Ideal_base_and_offset(adr, phase, offset);
1940       if (base != NULL && base->is_Load() && offset == in_bytes(InlineKlass::default_value_offset_offset())) {
1941         const TypeKlassPtr* tkls = phase->type(base->in(MemNode::Address))->isa_klassptr();
1942         if (tkls != NULL && tkls->is_loaded() && tkls->klass_is_exact() && tkls->isa_inlinetype() &&
1943             tkls->offset() == in_bytes(InstanceKlass::adr_inlineklass_fixed_block_offset())) {
1944           assert(base->Opcode() == Op_LoadP, "must load an oop from klass");
1945           assert(Opcode() == Op_LoadI, "must load an int from fixed block");
1946           return TypeInt::make(tkls->klass()->as_inline_klass()->default_value_offset());
1947         }
1948       }
1949     }
1950   }
1951 
1952   const TypeKlassPtr *tkls = tp->isa_klassptr();
1953   if (tkls != NULL && !StressReflectiveCode) {
1954     ciKlass* klass = tkls->klass();
1955     if (tkls->is_loaded() && tkls->klass_is_exact()) {
1956       // We are loading a field from a Klass metaobject whose identity
1957       // is known at compile time (the type is "exact" or "precise").
1958       // Check for fields we know are maintained as constants by the VM.
1959       if (tkls->offset() == in_bytes(Klass::super_check_offset_offset())) {
1960         // The field is Klass::_super_check_offset.  Return its (constant) value.
1961         // (Folds up type checking code.)
1962         assert(Opcode() == Op_LoadI, "must load an int from _super_check_offset");
1963         return TypeInt::make(klass->super_check_offset());
1964       }
1965       // Compute index into primary_supers array
1966       juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1967       // Check for overflowing; use unsigned compare to handle the negative case.
1968       if( depth < ciKlass::primary_super_limit() ) {
1969         // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1970         // (Folds up type checking code.)
1971         assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1972         ciKlass *ss = klass->super_of_depth(depth);
1973         return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1974       }
1975       const Type* aift = load_array_final_field(tkls, klass);
1976       if (aift != NULL)  return aift;
1977     }
1978 
1979     // We can still check if we are loading from the primary_supers array at a
1980     // shallow enough depth.  Even though the klass is not exact, entries less
1981     // than or equal to its super depth are correct.
1982     if (tkls->is_loaded()) {
1983       ciType *inner = klass;
1984       while( inner->is_obj_array_klass() )
1985         inner = inner->as_obj_array_klass()->base_element_type();
1986       if( inner->is_instance_klass() &&
1987           !inner->as_instance_klass()->flags().is_interface() ) {
1988         // Compute index into primary_supers array
1989         juint depth = (tkls->offset() - in_bytes(Klass::primary_supers_offset())) / sizeof(Klass*);
1990         // Check for overflowing; use unsigned compare to handle the negative case.
1991         if( depth < ciKlass::primary_super_limit() &&
1992             depth <= klass->super_depth() ) { // allow self-depth checks to handle self-check case
1993           // The field is an element of Klass::_primary_supers.  Return its (constant) value.
1994           // (Folds up type checking code.)
1995           assert(Opcode() == Op_LoadKlass, "must load a klass from _primary_supers");
1996           ciKlass *ss = klass->super_of_depth(depth);
1997           return ss ? TypeKlassPtr::make(ss) : TypePtr::NULL_PTR;
1998         }
1999       }
2000     }
2001 
2002     // If the type is enough to determine that the thing is not an array,
2003     // we can give the layout_helper a positive interval type.
2004     // This will help short-circuit some reflective code.
2005     if (tkls->offset() == in_bytes(Klass::layout_helper_offset())
2006         && !klass->is_array_klass() // not directly typed as an array
2007         && !klass->is_interface()  // specifically not Serializable & Cloneable
2008         && !klass->is_java_lang_Object()   // not the supertype of all T[]
2009         ) {
2010       // Note:  When interfaces are reliable, we can narrow the interface
2011       // test to (klass != Serializable && klass != Cloneable).
2012       assert(Opcode() == Op_LoadI, "must load an int from _layout_helper");
2013       jint min_size = Klass::instance_layout_helper(oopDesc::header_size(), false);
2014       // The key property of this type is that it folds up tests
2015       // for array-ness, since it proves that the layout_helper is positive.
2016       // Thus, a generic value like the basic object layout helper works fine.
2017       return TypeInt::make(min_size, max_jint, Type::WidenMin);
2018     }
2019   }
2020 
2021   // If we are loading from a freshly-allocated object, produce a zero,
2022   // if the load is provably beyond the header of the object.
2023   // (Also allow a variable load from a fresh array to produce zero.)
2024   const TypeOopPtr *tinst = tp->isa_oopptr();
2025   bool is_instance = (tinst != NULL) && tinst->is_known_instance_field();
2026   bool is_boxed_value = (tinst != NULL) && tinst->is_ptr_to_boxed_value();
2027   if (ReduceFieldZeroing || is_instance || is_boxed_value) {
2028     Node* value = can_see_stored_value(mem,phase);
2029     if (value != NULL && value->is_Con()) {
2030       assert(value->bottom_type()->higher_equal(_type),"sanity");
2031       return value->bottom_type();
2032     }
2033   }
2034 
2035   if (is_instance) {
2036     // If we have an instance type and our memory input is the
2037     // programs's initial memory state, there is no matching store,
2038     // so just return a zero of the appropriate type
2039     Node *mem = in(MemNode::Memory);
2040     if (mem->is_Parm() && mem->in(0)->is_Start()) {
2041       assert(mem->as_Parm()->_con == TypeFunc::Memory, "must be memory Parm");
2042       return Type::get_zero_type(_type->basic_type());
2043     }
2044   }
2045 
2046   Node* alloc = is_new_object_mark_load(phase);
2047   if (alloc != NULL && !(alloc->Opcode() == Op_Allocate && UseBiasedLocking)) {
2048     return TypeX::make(markWord::prototype().value());

2216   const Type *t1 = phase->type( in(MemNode::Memory) );

2219   const Type *t2 = phase->type( adr );
2220   if (t2 == Type::TOP)  return Type::TOP;
2221   const TypePtr *tp = t2->is_ptr();
2222   if (TypePtr::above_centerline(tp->ptr()) ||
2223       tp->ptr() == TypePtr::Null)  return Type::TOP;
2224 
2225   // Return a more precise klass, if possible
2226   const TypeInstPtr *tinst = tp->isa_instptr();
2227   if (tinst != NULL) {
2228     ciInstanceKlass* ik = tinst->klass()->as_instance_klass();
2229     int offset = tinst->offset();
2230     if (ik == phase->C->env()->Class_klass()
2231         && (offset == java_lang_Class::klass_offset() ||
2232             offset == java_lang_Class::array_klass_offset())) {
2233       // We are loading a special hidden field from a Class mirror object,
2234       // the field which points to the VM's Klass metaobject.
2235       ciType* t = tinst->java_mirror_type();
2236       // java_mirror_type returns non-null for compile-time Class constants.
2237       if (t != NULL) {
2238         // constant oop => constant klass
2239         if (offset == java_lang_Class::array_klass_offset()) {
2240           if (t->is_void()) {
2241             // We cannot create a void array.  Since void is a primitive type return null
2242             // klass.  Users of this result need to do a null check on the returned klass.
2243             return TypePtr::NULL_PTR;
2244           }
2245           return TypeKlassPtr::make(ciArrayKlass::make(t));
2246         }
2247         if (!t->is_klass()) {
2248           // a primitive Class (e.g., int.class) has NULL for a klass field
2249           return TypePtr::NULL_PTR;
2250         }

2252         return TypeKlassPtr::make(t->as_klass());
2253       }
2254       // non-constant mirror, so we can't tell what's going on
2255     }
2256     if( !ik->is_loaded() )
2257       return _type;             // Bail out if not loaded
2258     if (offset == oopDesc::klass_offset_in_bytes()) {
2259       if (tinst->klass_is_exact()) {
2260         return TypeKlassPtr::make(ik);
2261       }
2262       // See if we can become precise: no subklasses and no interface
2263       // (Note:  We need to support verified interfaces.)
2264       if (!ik->is_interface() && !ik->has_subklass()) {
2265         //assert(!UseExactTypes, "this code should be useless with exact types");
2266         // Add a dependence; if any subclass added we need to recompile
2267         if (!ik->is_final()) {
2268           // %%% should use stronger assert_unique_concrete_subtype instead
2269           phase->C->dependencies()->assert_leaf_type(ik);
2270         }
2271         // Return precise klass
2272         return TypeKlassPtr::make(ik);
2273       }
2274 
2275       // Return root of possible klass
2276       return TypeKlassPtr::make(TypePtr::NotNull, ik, Type::Offset(0), tinst->flat_array());
2277     }
2278   }
2279

2281   const TypeAryPtr *tary = tp->isa_aryptr();

2287         return TypeKlassPtr::make(tary_klass);
2288       }
2289       ciArrayKlass* ak = tary_klass->as_array_klass();
2290       // If the klass is an object array, we defer the question to the
2291       // array component klass.

2293         assert(ak->is_loaded(), "");

2296           ciInstanceKlass *ik = base_k->as_instance_klass();
2297           // See if we can become precise: no subklasses and no interface
2298           if (!ik->is_interface() && !ik->has_subklass()) {
2299             //assert(!UseExactTypes, "this code should be useless with exact types");
2300             // Add a dependence; if any subclass added we need to recompile
2301             if (!ik->is_final()) {
2302               phase->C->dependencies()->assert_leaf_type(ik);
2303             }
2304             // Return precise array klass
2305             return TypeKlassPtr::make(ak);
2306           }
2307         }
2308         return TypeKlassPtr::make(TypePtr::NotNull, ak, Type::Offset(0), false);
2309       } else if (ak->is_type_array_klass()) {
2310         //assert(!UseExactTypes, "this code should be useless with exact types");
2311         return TypeKlassPtr::make(ak); // These are always precise
2312       }
2313     }
2314   }

2317   const TypeKlassPtr *tkls = tp->isa_klassptr();
2318   if (tkls != NULL && !StressReflectiveCode) {
2319     if (!tkls->is_loaded()) {
2320       return _type;             // Bail out if not loaded
2321     }
2322     ciKlass* klass = tkls->klass();
2323     if( klass->is_obj_array_klass() &&
2324         tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2325       ciKlass* elem = klass->as_obj_array_klass()->element_klass();
2326       // // Always returning precise element type is incorrect,
2327       // // e.g., element type could be object and array may contain strings
2328       // return TypeKlassPtr::make(TypePtr::Constant, elem, 0);
2329

2331       // according to the element type's subclassing.
2332       return TypeKlassPtr::make(tkls->ptr(), elem, Type::Offset(0), elem->flatten_array());
2333     } else if (klass->is_flat_array_klass() &&
2334                tkls->offset() == in_bytes(ObjArrayKlass::element_klass_offset())) {
2335       ciKlass* elem = klass->as_flat_array_klass()->element_klass();

2340       ciKlass* sup = klass->as_instance_klass()->super();
2341       // The field is Klass::_super.  Return its (constant) value.
2342       // (Folds up the 2nd indirection in aClassConstant.getSuperClass().)

2358 Node* LoadNode::klass_identity_common(PhaseGVN* phase) {
2359   Node* x = LoadNode::Identity(phase);
2360   if (x != this)  return x;
2361 
2362   // Take apart the address into an oop and and offset.
2363   // Return 'this' if we cannot.
2364   Node*    adr    = in(MemNode::Address);
2365   intptr_t offset = 0;
2366   Node*    base   = AddPNode::Ideal_base_and_offset(adr, phase, offset);
2367   if (base == NULL)     return this;
2368   const TypeOopPtr* toop = phase->type(adr)->isa_oopptr();
2369   if (toop == NULL)     return this;
2370 
2371   // Step over potential GC barrier for OopHandle resolve
2372   BarrierSetC2* bs = BarrierSet::barrier_set()->barrier_set_c2();
2373   if (bs->is_gc_barrier_node(base)) {
2374     base = bs->step_over_gc_barrier(base);
2375   }
2376 
2377   // We can fetch the klass directly through an AllocateNode.
2378   // This works even if the klass is not constant (clone or newArray).
2379   if (offset == oopDesc::klass_offset_in_bytes()) {
2380     Node* allocated_klass = AllocateNode::Ideal_klass(base, phase);
2381     if (allocated_klass != NULL) {
2382       return allocated_klass;
2383     }
2384   }
2385 
2386   // Simplify k.java_mirror.as_klass to plain k, where k is a Klass*.
2387   // See inline_native_Class_query for occurrences of these patterns.
2388   // Java Example:  x.getClass().isAssignableFrom(y)
2389   //
2390   // This improves reflective code, often making the Class
2391   // mirror go completely dead.  (Current exception:  Class
2392   // mirrors may appear in debug info, but we could clean them out by
2393   // introducing a new debug info operator for Klass.java_mirror).
2394 
2395   if (toop->isa_instptr() && toop->klass() == phase->C->env()->Class_klass()
2396       && offset == java_lang_Class::klass_offset()) {
2397     if (base->is_Load()) {
2398       Node* base2 = base->in(MemNode::Address);
2399       if (base2->is_Load()) { /* direct load of a load which is the OopHandle */
2400         Node* adr2 = base2->in(MemNode::Address);
2401         const TypeKlassPtr* tkls = phase->type(adr2)->isa_klassptr();
2402         if (tkls != NULL && !tkls->empty()
2403             && (tkls->klass()->is_instance_klass() ||
2404               tkls->klass()->is_array_klass())
2405             && adr2->is_AddP()
2406            ) {
2407           int mirror_field = in_bytes(Klass::java_mirror_offset());
2408           if (tkls->offset() == mirror_field) {
2409             return adr2->in(AddPNode::Base);
2410           }
2411         }
2412       }
2413     }
2414   }
2415 
2416   return this;
2417 }

2591 //------------------------------Ideal------------------------------------------
2592 // Change back-to-back Store(, p, x) -> Store(m, p, y) to Store(m, p, x).
2593 // When a store immediately follows a relevant allocation/initialization,
2594 // try to capture it into the initialization, or hoist it above.
2595 Node *StoreNode::Ideal(PhaseGVN *phase, bool can_reshape) {
2596   Node* p = MemNode::Ideal_common(phase, can_reshape);
2597   if (p)  return (p == NodeSentinel) ? NULL : p;
2598 
2599   Node* mem     = in(MemNode::Memory);
2600   Node* address = in(MemNode::Address);
2601   // Back-to-back stores to same address?  Fold em up.  Generally
2602   // unsafe if I have intervening uses...  Also disallowed for StoreCM
2603   // since they must follow each StoreP operation.  Redundant StoreCMs
2604   // are eliminated just before matching in final_graph_reshape.
2605   if (phase->C->get_adr_type(phase->C->get_alias_index(adr_type())) != TypeAryPtr::INLINES) {
2606     Node* st = mem;
2607     // If Store 'st' has more than one use, we cannot fold 'st' away.
2608     // For example, 'st' might be the final state at a conditional
2609     // return.  Or, 'st' might be used by some node which is live at
2610     // the same time 'st' is live, which might be unschedulable.  So,
2611     // require exactly ONE user until such time as we clone 'mem' for
2612     // each of 'mem's uses (thus making the exactly-1-user-rule hold
2613     // true).
2614     while (st->is_Store() && st->outcnt() == 1 && st->Opcode() != Op_StoreCM) {
2615       // Looking at a dead closed cycle of memory?
2616       assert(st != st->in(MemNode::Memory), "dead loop in StoreNode::Ideal");
2617       assert(Opcode() == st->Opcode() ||
2618              st->Opcode() == Op_StoreVector ||
2619              Opcode() == Op_StoreVector ||
2620              phase->C->get_alias_index(adr_type()) == Compile::AliasIdxRaw ||
2621              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreI) || // expanded ClearArrayNode
2622              (Opcode() == Op_StoreI && st->Opcode() == Op_StoreL) || // initialization by arraycopy
2623              (Opcode() == Op_StoreL && st->Opcode() == Op_StoreN) ||
2624              (is_mismatched_access() || st->as_Store()->is_mismatched_access()),
2625              "no mismatched stores, except on raw memory: %s %s", NodeClassNames[Opcode()], NodeClassNames[st->Opcode()]);
2626 
2627       if (st->in(MemNode::Address)->eqv_uncast(address) &&
2628           st->as_Store()->memory_size() <= this->memory_size()) {
2629         Node* use = st->raw_out(0);
2630         phase->igvn_rehash_node_delayed(use);
2631         if (can_reshape) {
2632           use->set_req_X(MemNode::Memory, st->in(MemNode::Memory), phase->is_IterGVN());
2633         } else {
2634           // It's OK to do this in the parser, since DU info is always accurate,
2635           // and the parser always refers to nodes via SafePointNode maps.
2636           use->set_req(MemNode::Memory, st->in(MemNode::Memory));
2637         }
2638         return this;
2639       }
2640       st = st->in(MemNode::Memory);
2641     }
2642   }
2643 
2644 
2645   // Capture an unaliased, unconditional, simple store into an initializer.
2646   // Or, if it is independent of the allocation, hoist it above the allocation.
2647   if (ReduceFieldZeroing && /*can_reshape &&*/
2648       mem->is_Proj() && mem->in(0)->is_Initialize()) {
2649     InitializeNode* init = mem->in(0)->as_Initialize();
2650     intptr_t offset = init->can_capture_store(this, phase, can_reshape);
2651     if (offset > 0) {
2652       Node* moved = init->capture_store(this, offset, phase, can_reshape);
2653       // If the InitializeNode captured me, it made a raw copy of me,
2654       // and I need to disappear.
2655       if (moved != NULL) {
2656         // %%% hack to ensure that Ideal returns a new node:
2657         mem = MergeMemNode::make(mem);
2658         return mem;             // fold me away
2659       }
2660     }
2661   }
2662 
2663   return NULL;                  // No further progress
2664 }

2678 //------------------------------Identity---------------------------------------
2679 // Remove redundant stores:
2680 //   Store(m, p, Load(m, p)) changes to m.
2681 //   Store(, p, x) -> Store(m, p, x) changes to Store(m, p, x).
2682 Node* StoreNode::Identity(PhaseGVN* phase) {
2683   Node* mem = in(MemNode::Memory);
2684   Node* adr = in(MemNode::Address);
2685   Node* val = in(MemNode::ValueIn);
2686 
2687   Node* result = this;
2688 
2689   // Load then Store?  Then the Store is useless
2690   if (val->is_Load() &&
2691       val->in(MemNode::Address)->eqv_uncast(adr) &&
2692       val->in(MemNode::Memory )->eqv_uncast(mem) &&
2693       val->as_Load()->store_Opcode() == Opcode()) {
2694     result = mem;
2695   }
2696 
2697   // Two stores in a row of the same value?
2698   if (result == this &&
2699       mem->is_Store() &&
2700       mem->in(MemNode::Address)->eqv_uncast(adr) &&
2701       mem->in(MemNode::ValueIn)->eqv_uncast(val) &&
2702       mem->Opcode() == Opcode()) {
2703     result = mem;
2704   }
2705 
2706   // Store of zero anywhere into a freshly-allocated object?
2707   // Then the store is useless.
2708   // (It must already have been captured by the InitializeNode.)
2709   if (result == this && ReduceFieldZeroing) {
2710     // a newly allocated object is already all-zeroes everywhere
2711     if (mem->is_Proj() && mem->in(0)->is_Allocate() &&
2712         (phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == val)) {
2713       assert(!phase->type(val)->is_zero_type() || mem->in(0)->in(AllocateNode::DefaultValue) == NULL, "storing null to inline type array is forbidden");
2714       result = mem;
2715     }
2716 
2717     if (result == this) {
2718       // the store may also apply to zero-bits in an earlier object
2719       Node* prev_mem = find_previous_store(phase);
2720       // Steps (a), (b):  Walk past independent stores to find an exact match.
2721       if (prev_mem != NULL) {
2722         Node* prev_val = can_see_stored_value(prev_mem, phase);
2723         if (prev_val != NULL && phase->eqv(prev_val, val)) {
2724           // prev_val and val might differ by a cast; it would be good
2725           // to keep the more informative of the two.
2726           if (phase->type(val)->is_zero_type()) {
2727             result = mem;
2728           } else if (prev_mem->is_Proj() && prev_mem->in(0)->is_Initialize()) {
2729             InitializeNode* init = prev_mem->in(0)->as_Initialize();
2730             AllocateNode* alloc = init->allocation();
2731             if (alloc != NULL && alloc->in(AllocateNode::DefaultValue) == val) {
2732               result = mem;
2733             }
2734           }
2735         }
2736       }
2737     }
2738   }
2739 
2740   if (result != this && phase->is_IterGVN() != NULL) {
2741     MemBarNode* trailing = trailing_membar();
2742     if (trailing != NULL) {
2743 #ifdef ASSERT
2744       const TypeOopPtr* t_oop = phase->type(in(Address))->isa_oopptr();
2745       assert(t_oop == NULL || t_oop->is_known_instance_field(), "only for non escaping objects");
2746 #endif
2747       PhaseIterGVN* igvn = phase->is_IterGVN();
2748       trailing->remove(igvn);
2749     }
2750   }
2751 
2752   return result;
2753 }

3026   if (!IdealizeClearArrayNode || _is_large) return NULL;

3467 //===========================InitializeNode====================================
3468 // SUMMARY:
3469 // This node acts as a memory barrier on raw memory, after some raw stores.
3470 // The 'cooked' oop value feeds from the Initialize, not the Allocation.
3471 // The Initialize can 'capture' suitably constrained stores as raw inits.
3472 // It can coalesce related raw stores into larger units (called 'tiles').
3473 // It can avoid zeroing new storage for memory units which have raw inits.
3474 // At macro-expansion, it is marked 'complete', and does not optimize further.
3475 //
3476 // EXAMPLE:
3477 // The object 'new short[2]' occupies 16 bytes in a 32-bit machine.
3478 //   ctl = incoming control; mem* = incoming memory
3479 // (Note:  A star * on a memory edge denotes I/O and other standard edges.)
3480 // First allocate uninitialized memory and fill in the header:
3481 //   alloc = (Allocate ctl mem* 16 #short[].klass ...)
3482 //   ctl := alloc.Control; mem* := alloc.Memory*
3483 //   rawmem = alloc.Memory; rawoop = alloc.RawAddress
3484 // Then initialize to zero the non-header parts of the raw memory block:
3485 //   init = (Initialize alloc.Control alloc.Memory* alloc.RawAddress)
3486 //   ctl := init.Control; mem.SLICE(#short[*]) := init.Memory
3487 // After the initialize node executes, the object is ready for service:
3488 //   oop := (CheckCastPP init.Control alloc.RawAddress #short[])
3489 // Suppose its body is immediately initialized as {1,2}:
3490 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3491 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3492 //   mem.SLICE(#short[*]) := store2
3493 //
3494 // DETAILS:
3495 // An InitializeNode collects and isolates object initialization after
3496 // an AllocateNode and before the next possible safepoint.  As a
3497 // memory barrier (MemBarNode), it keeps critical stores from drifting
3498 // down past any safepoint or any publication of the allocation.
3499 // Before this barrier, a newly-allocated object may have uninitialized bits.
3500 // After this barrier, it may be treated as a real oop, and GC is allowed.
3501 //
3502 // The semantics of the InitializeNode include an implicit zeroing of
3503 // the new object from object header to the end of the object.
3504 // (The object header and end are determined by the AllocateNode.)
3505 //
3506 // Certain stores may be added as direct inputs to the InitializeNode.
3507 // These stores must update raw memory, and they must be to addresses
3508 // derived from the raw address produced by AllocateNode, and with
3509 // a constant offset.  They must be ordered by increasing offset.
3510 // The first one is at in(RawStores), the last at in(req()-1).
3511 // Unlike most memory operations, they are not linked in a chain,
3512 // but are displayed in parallel as users of the rawmem output of
3513 // the allocation.
3514 //
3515 // (See comments in InitializeNode::capture_store, which continue
3516 // the example given above.)
3517 //
3518 // When the associated Allocate is macro-expanded, the InitializeNode
3519 // may be rewritten to optimize collected stores.  A ClearArrayNode
3520 // may also be created at that point to represent any required zeroing.
3521 // The InitializeNode is then marked 'complete', prohibiting further
3522 // capturing of nearby memory operations.
3523 //
3524 // During macro-expansion, all captured initializations which store
3525 // constant values of 32 bits or smaller are coalesced (if advantageous)
3526 // into larger 'tiles' 32 or 64 bits.  This allows an object to be
3527 // initialized in fewer memory operations.  Memory words which are
3528 // covered by neither tiles nor non-constant stores are pre-zeroed
3529 // by explicit stores of zero.  (The code shape happens to do all
3530 // zeroing first, then all other stores, with both sequences occurring
3531 // in order of ascending offsets.)
3532 //
3533 // Alternatively, code may be inserted between an AllocateNode and its
3534 // InitializeNode, to perform arbitrary initialization of the new object.
3535 // E.g., the object copying intrinsics insert complex data transfers here.
3536 // The initialization must then be marked as 'complete' disable the
3537 // built-in zeroing semantics and the collection of initializing stores.
3538 //
3539 // While an InitializeNode is incomplete, reads from the memory state
3540 // produced by it are optimizable if they match the control edge and
3541 // new oop address associated with the allocation/initialization.
3542 // They return a stored value (if the offset matches) or else zero.
3543 // A write to the memory state, if it matches control and address,
3544 // and if it is to a constant offset, may be 'captured' by the
3545 // InitializeNode.  It is cloned as a raw memory operation and rewired
3546 // inside the initialization, to the raw oop produced by the allocation.
3547 // Operations on addresses which are provably distinct (e.g., to
3548 // other AllocateNodes) are allowed to bypass the initialization.
3549 //
3550 // The effect of all this is to consolidate object initialization
3551 // (both arrays and non-arrays, both piecewise and bulk) into a
3552 // single location, where it can be optimized as a unit.
3553 //
3554 // Only stores with an offset less than TrackedInitializationLimit words
3555 // will be considered for capture by an InitializeNode.  This puts a
3556 // reasonable limit on the complexity of optimized initializations.
3557

3656 bool InitializeNode::detect_init_independence(Node* value, PhaseGVN* phase) {
3657   ResourceMark rm;
3658   Unique_Node_List worklist;
3659   worklist.push(value);
3660 
3661   uint complexity_limit = 20;
3662   for (uint j = 0; j < worklist.size(); j++) {
3663     if (j >= complexity_limit) {
3664       return false;  // Bail out if processed too many nodes
3665     }
3666 
3667     Node* n = worklist.at(j);
3668     if (n == NULL)      continue;   // (can this really happen?)
3669     if (n->is_Proj())   n = n->in(0);
3670     if (n == this)      return false;  // found a cycle
3671     if (n->is_Con())    continue;
3672     if (n->is_Start())  continue;   // params, etc., are OK
3673     if (n->is_Root())   continue;   // even better
3674 
3675     // There cannot be any dependency if 'n' is a CFG node that dominates the current allocation
3676     if (n->is_CFG() && phase->is_dominator(n, allocation())) {
3677       continue;
3678     }
3679 
3680     Node* ctl = n->in(0);
3681     if (ctl != NULL && !ctl->is_top()) {
3682       if (ctl->is_Proj())  ctl = ctl->in(0);
3683       if (ctl == this)  return false;
3684 
3685       // If we already know that the enclosing memory op is pinned right after
3686       // the init, then any control flow that the store has picked up
3687       // must have preceded the init, or else be equal to the init.
3688       // Even after loop optimizations (which might change control edges)
3689       // a store is never pinned *before* the availability of its inputs.
3690       if (!MemNode::all_controls_dominate(n, this))
3691         return false;                  // failed to prove a good control
3692     }
3693 
3694     // Check data edges for possible dependencies on 'this'.
3695     for (uint i = 1; i < n->req(); i++) {
3696       Node* m = n->in(i);
3697       if (m == NULL || m == n || m->is_top())  continue;
3698 
3699       // Only process data inputs once
3700       worklist.push(m);
3701     }
3702   }
3703 
3704   return true;
3705 }

3707 // Here are all the checks a Store must pass before it can be moved into
3708 // an initialization.  Returns zero if a check fails.
3709 // On success, returns the (constant) offset to which the store applies,
3710 // within the initialized memory.
3711 intptr_t InitializeNode::can_capture_store(StoreNode* st, PhaseGVN* phase, bool can_reshape) {
3712   const int FAIL = 0;
3713   if (st->req() != MemNode::ValueIn + 1)
3714     return FAIL;                // an inscrutable StoreNode (card mark?)
3715   Node* ctl = st->in(MemNode::Control);
3716   if (!(ctl != NULL && ctl->is_Proj() && ctl->in(0) == this))
3717     return FAIL;                // must be unconditional after the initialization
3718   Node* mem = st->in(MemNode::Memory);
3719   if (!(mem->is_Proj() && mem->in(0) == this))
3720     return FAIL;                // must not be preceded by other stores
3721   Node* adr = st->in(MemNode::Address);
3722   intptr_t offset;
3723   AllocateNode* alloc = AllocateNode::Ideal_allocation(adr, phase, offset);
3724   if (alloc == NULL)
3725     return FAIL;                // inscrutable address
3726   if (alloc != allocation())
3727     return FAIL;                // wrong allocation!  (store needs to float up)
3728   int size_in_bytes = st->memory_size();
3729   if ((size_in_bytes != 0) && (offset % size_in_bytes) != 0) {
3730     return FAIL;                // mismatched access
3731   }
3732   Node* val = st->in(MemNode::ValueIn);
3733 
3734   if (!detect_init_independence(val, phase))
3735     return FAIL;                // stored value must be 'simple enough'
3736 
3737   // The Store can be captured only if nothing after the allocation
3738   // and before the Store is using the memory location that the store
3739   // overwrites.
3740   bool failed = false;
3741   // If is_complete_with_arraycopy() is true the shape of the graph is
3742   // well defined and is safe so no need for extra checks.
3743   if (!is_complete_with_arraycopy()) {
3744     // We are going to look at each use of the memory state following
3745     // the allocation to make sure nothing reads the memory that the
3746     // Store writes.
3747     const TypePtr* t_adr = phase->type(adr)->isa_ptr();
3748     int alias_idx = phase->C->get_alias_index(t_adr);
3749     ResourceMark rm;
3750     Unique_Node_List mems;
3751     mems.push(mem);
3752     Node* unique_merge = NULL;
3753     for (uint next = 0; next < mems.size(); ++next) {
3754       Node *m  = mems.at(next);
3755       for (DUIterator_Fast jmax, j = m->fast_outs(jmax); j < jmax; j++) {
3756         Node *n = m->fast_out(j);
3757         if (n->outcnt() == 0) {
3758           continue;
3759         }
3760         if (n == st) {
3761           continue;
3762         } else if (n->in(0) != NULL && n->in(0) != ctl) {
3763           // If the control of this use is different from the control
3764           // of the Store which is right after the InitializeNode then
3765           // this node cannot be between the InitializeNode and the
3766           // Store.
3767           continue;
3768         } else if (n->is_MergeMem()) {
3769           if (n->as_MergeMem()->memory_at(alias_idx) == m) {
3770             // We can hit a MergeMemNode (that will likely go away
3771             // later) that is a direct use of the memory state
3772             // following the InitializeNode on the same slice as the
3773             // store node that we'd like to capture. We need to check
3774             // the uses of the MergeMemNode.
3775             mems.push(n);
3776           }
3777         } else if (n->is_Mem()) {
3778           Node* other_adr = n->in(MemNode::Address);
3779           if (other_adr == adr) {
3780             failed = true;
3781             break;
3782           } else {
3783             const TypePtr* other_t_adr = phase->type(other_adr)->isa_ptr();
3784             if (other_t_adr != NULL) {
3785               int other_alias_idx = phase->C->get_alias_index(other_t_adr);
3786               if (other_alias_idx == alias_idx) {
3787                 // A load from the same memory slice as the store right
3788                 // after the InitializeNode. We check the control of the
3789                 // object/array that is loaded from. If it's the same as
3790                 // the store control then we cannot capture the store.
3791                 assert(!n->is_Store(), "2 stores to same slice on same control?");
3792                 Node* base = other_adr;
3793                 assert(base->is_AddP(), "should be addp but is %s", base->Name());
3794                 base = base->in(AddPNode::Base);
3795                 if (base != NULL) {
3796                   base = base->uncast();
3797                   if (base->is_Proj() && base->in(0) == alloc) {
3798                     failed = true;
3799                     break;
3800                   }
3801                 }
3802               }
3803             }
3804           }
3805         } else {
3806           failed = true;
3807           break;
3808         }
3809       }
3810     }
3811   }
3812   if (failed) {
3813     if (!can_reshape) {
3814       // We decided we couldn't capture the store during parsing. We
3815       // should try again during the next IGVN once the graph is
3816       // cleaner.
3817       phase->C->record_for_igvn(st);
3818     }
3819     return FAIL;
3820   }
3821 
3822   return offset;                // success
3823 }

3825 // Find the captured store in(i) which corresponds to the range
3826 // [start..start+size) in the initialized object.
3827 // If there is one, return its index i.  If there isn't, return the
3828 // negative of the index where it should be inserted.
3829 // Return 0 if the queried range overlaps an initialization boundary
3830 // or if dead code is encountered.
3831 // If size_in_bytes is zero, do not bother with overlap checks.
3832 int InitializeNode::captured_store_insertion_point(intptr_t start,
3833                                                    int size_in_bytes,
3834                                                    PhaseTransform* phase) {
3835   const int FAIL = 0, MAX_STORE = BytesPerLong;
3836 
3837   if (is_complete())
3838     return FAIL;                // arraycopy got here first; punt
3839 
3840   assert(allocation() != NULL, "must be present");
3841 
3842   // no negatives, no header fields:
3843   if (start < (intptr_t) allocation()->minimum_header_size())  return FAIL;
3844 
3845   // after a certain size, we bail out on tracking all the stores:
3846   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
3847   if (start >= ti_limit)  return FAIL;
3848 
3849   for (uint i = InitializeNode::RawStores, limit = req(); ; ) {
3850     if (i >= limit)  return -(int)i; // not found; here is where to put it
3851 
3852     Node*    st     = in(i);
3853     intptr_t st_off = get_store_offset(st, phase);
3854     if (st_off < 0) {
3855       if (st != zero_memory()) {
3856         return FAIL;            // bail out if there is dead garbage
3857       }
3858     } else if (st_off > start) {
3859       // ...we are done, since stores are ordered
3860       if (st_off < start + size_in_bytes) {
3861         return FAIL;            // the next store overlaps
3862       }
3863       return -(int)i;           // not found; here is where to put it
3864     } else if (st_off < start) {
3865       if (size_in_bytes != 0 &&
3866           start < st_off + MAX_STORE &&
3867           start < st_off + st->as_Store()->memory_size()) {
3868         return FAIL;            // the previous store overlaps
3869       }
3870     } else {
3871       if (size_in_bytes != 0 &&
3872           st->as_Store()->memory_size() != size_in_bytes) {
3873         return FAIL;            // mismatched store size
3874       }
3875       return i;
3876     }
3877 
3878     ++i;
3879   }
3880 }

3913 // Clone the given store, converting it into a raw store
3914 // initializing a field or element of my new object.
3915 // Caller is responsible for retiring the original store,
3916 // with subsume_node or the like.
3917 //
3918 // From the example above InitializeNode::InitializeNode,
3919 // here are the old stores to be captured:
3920 //   store1 = (StoreC init.Control init.Memory (+ oop 12) 1)
3921 //   store2 = (StoreC init.Control store1      (+ oop 14) 2)
3922 //
3923 // Here is the changed code; note the extra edges on init:
3924 //   alloc = (Allocate ...)
3925 //   rawoop = alloc.RawAddress
3926 //   rawstore1 = (StoreC alloc.Control alloc.Memory (+ rawoop 12) 1)
3927 //   rawstore2 = (StoreC alloc.Control alloc.Memory (+ rawoop 14) 2)
3928 //   init = (Initialize alloc.Control alloc.Memory rawoop
3929 //                      rawstore1 rawstore2)
3930 //
3931 Node* InitializeNode::capture_store(StoreNode* st, intptr_t start,
3932                                     PhaseGVN* phase, bool can_reshape) {
3933   assert(stores_are_sane(phase), "");
3934 
3935   if (start < 0)  return NULL;
3936   assert(can_capture_store(st, phase, can_reshape) == start, "sanity");
3937 
3938   Compile* C = phase->C;
3939   int size_in_bytes = st->memory_size();
3940   int i = captured_store_insertion_point(start, size_in_bytes, phase);
3941   if (i == 0)  return NULL;     // bail out
3942   Node* prev_mem = NULL;        // raw memory for the captured store
3943   if (i > 0) {
3944     prev_mem = in(i);           // there is a pre-existing store under this one
3945     set_req(i, C->top());       // temporarily disconnect it
3946     // See StoreNode::Ideal 'st->outcnt() == 1' for the reason to disconnect.
3947   } else {
3948     i = -i;                     // no pre-existing store
3949     prev_mem = zero_memory();   // a slice of the newly allocated object
3950     if (i > InitializeNode::RawStores && in(i-1) == prev_mem)
3951       set_req(--i, C->top());   // reuse this edge; it has been folded away
3952     else
3953       ins_req(i, C->top());     // build a new edge
3954   }
3955   Node* new_st = st->clone();
3956   new_st->set_req(MemNode::Control, in(Control));
3957   new_st->set_req(MemNode::Memory,  prev_mem);
3958   new_st->set_req(MemNode::Address, make_raw_address(start, phase));
3959   new_st = phase->transform(new_st);
3960 
3961   // At this point, new_st might have swallowed a pre-existing store
3962   // at the same offset, or perhaps new_st might have disappeared,
3963   // if it redundantly stored the same value (or zero to fresh memory).
3964 
3965   // In any case, wire it in:
3966   phase->igvn_rehash_node_delayed(this);
3967   set_req(i, new_st);
3968 
3969   // The caller may now kill the old guy.
3970   DEBUG_ONLY(Node* check_st = find_captured_store(start, size_in_bytes, phase));
3971   assert(check_st == new_st || check_st == NULL, "must be findable");
3972   assert(!is_complete(), "");
3973   return new_st;
3974 }

3987   case sizeof(jlong):  *(jlong*) addr = (jlong) con; break;
3988   default: return false;        // strange store size (detect size!=2**N here)
3989   }
3990   return true;                  // return success to caller

3993 // Coalesce subword constants into int constants and possibly
3994 // into long constants.  The goal, if the CPU permits,
3995 // is to initialize the object with a small number of 64-bit tiles.
3996 // Also, convert floating-point constants to bit patterns.
3997 // Non-constants are not relevant to this pass.
3998 //
3999 // In terms of the running example on InitializeNode::InitializeNode
4000 // and InitializeNode::capture_store, here is the transformation
4001 // of rawstore1 and rawstore2 into rawstore12:
4002 //   alloc = (Allocate ...)
4003 //   rawoop = alloc.RawAddress
4004 //   tile12 = 0x00010002
4005 //   rawstore12 = (StoreI alloc.Control alloc.Memory (+ rawoop 12) tile12)
4006 //   init = (Initialize alloc.Control alloc.Memory rawoop rawstore12)
4007 //
4008 void
4009 InitializeNode::coalesce_subword_stores(intptr_t header_size,
4010                                         Node* size_in_bytes,
4011                                         PhaseGVN* phase) {
4012   Compile* C = phase->C;
4013 
4014   assert(stores_are_sane(phase), "");
4015   // Note:  After this pass, they are not completely sane,
4016   // since there may be some overlaps.
4017 
4018   int old_subword = 0, old_long = 0, new_int = 0, new_long = 0;
4019 
4020   intptr_t ti_limit = (TrackedInitializationLimit * HeapWordSize);
4021   intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, ti_limit);
4022   size_limit = MIN2(size_limit, ti_limit);
4023   size_limit = align_up(size_limit, BytesPerLong);
4024   int num_tiles = size_limit / BytesPerLong;
4025 
4026   // allocate space for the tile map:
4027   const int small_len = DEBUG_ONLY(true ? 3 :) 30; // keep stack frames small
4028   jlong  tiles_buf[small_len];
4029   Node*  nodes_buf[small_len];
4030   jlong  inits_buf[small_len];
4031   jlong* tiles = ((num_tiles <= small_len) ? &tiles_buf[0]
4032                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4033   Node** nodes = ((num_tiles <= small_len) ? &nodes_buf[0]
4034                   : NEW_RESOURCE_ARRAY(Node*, num_tiles));
4035   jlong* inits = ((num_tiles <= small_len) ? &inits_buf[0]
4036                   : NEW_RESOURCE_ARRAY(jlong, num_tiles));
4037   // tiles: exact bitwise model of all primitive constants
4038   // nodes: last constant-storing node subsumed into the tiles model
4039   // inits: which bytes (in each tile) are touched by any initializations
4040 
4041   //// Pass A: Fill in the tile model with any relevant stores.
4042 
4043   Copy::zero_to_bytes(tiles, sizeof(tiles[0]) * num_tiles);
4044   Copy::zero_to_bytes(nodes, sizeof(nodes[0]) * num_tiles);
4045   Copy::zero_to_bytes(inits, sizeof(inits[0]) * num_tiles);
4046   Node* zmem = zero_memory(); // initially zero memory state
4047   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4048     Node* st = in(i);
4049     intptr_t st_off = get_store_offset(st, phase);
4050 
4051     // Figure out the store's offset and constant value:
4052     if (st_off < header_size)             continue; //skip (ignore header)
4053     if (st->in(MemNode::Memory) != zmem)  continue; //skip (odd store chain)
4054     int st_size = st->as_Store()->memory_size();
4055     if (st_off + st_size > size_limit)    break;
4056 
4057     // Record which bytes are touched, whether by constant or not.
4058     if (!store_constant(inits, num_tiles, st_off, st_size, (jlong) -1))
4059       continue;                 // skip (strange store size)
4060 
4061     const Type* val = phase->type(st->in(MemNode::ValueIn));
4062     if (!val->singleton())                continue; //skip (non-con store)
4063     BasicType type = val->basic_type();
4064 
4065     jlong con = 0;
4066     switch (type) {
4067     case T_INT:    con = val->is_int()->get_con();  break;
4068     case T_LONG:   con = val->is_long()->get_con(); break;
4069     case T_FLOAT:  con = jint_cast(val->getf());    break;
4070     case T_DOUBLE: con = jlong_cast(val->getd());   break;
4071     default:                              continue; //skip (odd store type)
4072     }
4073 
4074     if (type == T_LONG && Matcher::isSimpleConstant64(con) &&
4075         st->Opcode() == Op_StoreL) {
4076       continue;                 // This StoreL is already optimal.
4077     }
4078 
4079     // Store down the constant.
4080     store_constant(tiles, num_tiles, st_off, st_size, con);
4081 
4082     intptr_t j = st_off >> LogBytesPerLong;
4083 
4084     if (type == T_INT && st_size == BytesPerInt
4085         && (st_off & BytesPerInt) == BytesPerInt) {
4086       jlong lcon = tiles[j];
4087       if (!Matcher::isSimpleConstant64(lcon) &&
4088           st->Opcode() == Op_StoreI) {
4089         // This StoreI is already optimal by itself.
4090         jint* intcon = (jint*) &tiles[j];
4091         intcon[1] = 0;  // undo the store_constant()
4092 
4093         // If the previous store is also optimal by itself, back up and
4094         // undo the action of the previous loop iteration... if we can.
4095         // But if we can't, just let the previous half take care of itself.
4096         st = nodes[j];
4097         st_off -= BytesPerInt;
4098         con = intcon[0];
4099         if (con != 0 && st != NULL && st->Opcode() == Op_StoreI) {
4100           assert(st_off >= header_size, "still ignoring header");
4101           assert(get_store_offset(st, phase) == st_off, "must be");
4102           assert(in(i-1) == zmem, "must be");
4103           DEBUG_ONLY(const Type* tcon = phase->type(st->in(MemNode::ValueIn)));
4104           assert(con == tcon->is_int()->get_con(), "must be");
4105           // Undo the effects of the previous loop trip, which swallowed st:
4106           intcon[0] = 0;        // undo store_constant()
4107           set_req(i-1, st);     // undo set_req(i, zmem)
4108           nodes[j] = NULL;      // undo nodes[j] = st
4109           --old_subword;        // undo ++old_subword
4110         }
4111         continue;               // This StoreI is already optimal.
4112       }
4113     }
4114 
4115     // This store is not needed.
4116     set_req(i, zmem);
4117     nodes[j] = st;              // record for the moment
4118     if (st_size < BytesPerLong) // something has changed
4119           ++old_subword;        // includes int/float, but who's counting...
4120     else  ++old_long;
4121   }
4122 
4123   if ((old_subword + old_long) == 0)
4124     return;                     // nothing more to do
4125 
4126   //// Pass B: Convert any non-zero tiles into optimal constant stores.
4127   // Be sure to insert them before overlapping non-constant stores.
4128   // (E.g., byte[] x = { 1,2,y,4 }  =>  x[int 0] = 0x01020004, x[2]=y.)
4129   for (int j = 0; j < num_tiles; j++) {
4130     jlong con  = tiles[j];
4131     jlong init = inits[j];
4132     if (con == 0)  continue;
4133     jint con0,  con1;           // split the constant, address-wise
4134     jint init0, init1;          // split the init map, address-wise
4135     { union { jlong con; jint intcon[2]; } u;
4136       u.con = con;
4137       con0  = u.intcon[0];
4138       con1  = u.intcon[1];
4139       u.con = init;
4140       init0 = u.intcon[0];
4141       init1 = u.intcon[1];
4142     }
4143 
4144     Node* old = nodes[j];
4145     assert(old != NULL, "need the prior store");
4146     intptr_t offset = (j * BytesPerLong);
4147 
4148     bool split = !Matcher::isSimpleConstant64(con);
4149 
4150     if (offset < header_size) {
4151       assert(offset + BytesPerInt >= header_size, "second int counts");
4152       assert(*(jint*)&tiles[j] == 0, "junk in header");
4153       split = true;             // only the second word counts
4154       // Example:  int a[] = { 42 ... }
4155     } else if (con0 == 0 && init0 == -1) {
4156       split = true;             // first word is covered by full inits
4157       // Example:  int a[] = { ... foo(), 42 ... }
4158     } else if (con1 == 0 && init1 == -1) {
4159       split = true;             // second word is covered by full inits
4160       // Example:  int a[] = { ... 42, foo() ... }
4161     }
4162 
4163     // Here's a case where init0 is neither 0 nor -1:
4164     //   byte a[] = { ... 0,0,foo(),0,  0,0,0,42 ... }
4165     // Assuming big-endian memory, init0, init1 are 0x0000FF00, 0x000000FF.
4166     // In this case the tile is not split; it is (jlong)42.
4167     // The big tile is stored down, and then the foo() value is inserted.
4168     // (If there were foo(),foo() instead of foo(),0, init0 would be -1.)
4169 
4170     Node* ctl = old->in(MemNode::Control);
4171     Node* adr = make_raw_address(offset, phase);

4230 // Explore forward from in(start) to find the first fully initialized
4231 // word, and return its offset.  Skip groups of subword stores which
4232 // together initialize full words.  If in(start) is itself part of a
4233 // fully initialized word, return the offset of in(start).  If there
4234 // are no following full-word stores, or if something is fishy, return
4235 // a negative value.
4236 intptr_t InitializeNode::find_next_fullword_store(uint start, PhaseGVN* phase) {
4237   int       int_map = 0;
4238   intptr_t  int_map_off = 0;
4239   const int FULL_MAP = right_n_bits(BytesPerInt);  // the int_map we hope for
4240 
4241   for (uint i = start, limit = req(); i < limit; i++) {
4242     Node* st = in(i);
4243 
4244     intptr_t st_off = get_store_offset(st, phase);
4245     if (st_off < 0)  break;  // return conservative answer
4246 
4247     int st_size = st->as_Store()->memory_size();
4248     if (st_size >= BytesPerInt && (st_off % BytesPerInt) == 0) {
4249       return st_off;            // we found a complete word init
4250     }
4251 
4252     // update the map:
4253 
4254     intptr_t this_int_off = align_down(st_off, BytesPerInt);
4255     if (this_int_off != int_map_off) {
4256       // reset the map:
4257       int_map = 0;
4258       int_map_off = this_int_off;
4259     }
4260 
4261     int subword_off = st_off - this_int_off;
4262     int_map |= right_n_bits(st_size) << subword_off;
4263     if ((int_map & FULL_MAP) == FULL_MAP) {
4264       return this_int_off;      // we found a complete word init
4265     }
4266 
4267     // Did this store hit or cross the word boundary?
4268     intptr_t next_int_off = align_down(st_off + st_size, BytesPerInt);
4269     if (next_int_off == this_int_off + BytesPerInt) {
4270       // We passed the current int, without fully initializing it.
4271       int_map_off = next_int_off;
4272       int_map >>= BytesPerInt;
4273     } else if (next_int_off > this_int_off + BytesPerInt) {
4274       // We passed the current and next int.
4275       return this_int_off + BytesPerInt;
4276     }
4277   }
4278 
4279   return -1;
4280 }

4283 // Called when the associated AllocateNode is expanded into CFG.
4284 // At this point, we may perform additional optimizations.
4285 // Linearize the stores by ascending offset, to make memory
4286 // activity as coherent as possible.
4287 Node* InitializeNode::complete_stores(Node* rawctl, Node* rawmem, Node* rawptr,
4288                                       intptr_t header_size,
4289                                       Node* size_in_bytes,
4290                                       PhaseIterGVN* phase) {
4291   assert(!is_complete(), "not already complete");
4292   assert(stores_are_sane(phase), "");
4293   assert(allocation() != NULL, "must be present");
4294 
4295   remove_extra_zeroes();
4296 
4297   if (ReduceFieldZeroing || ReduceBulkZeroing)
4298     // reduce instruction count for common initialization patterns
4299     coalesce_subword_stores(header_size, size_in_bytes, phase);
4300 
4301   Node* zmem = zero_memory();   // initially zero memory state
4302   Node* inits = zmem;           // accumulating a linearized chain of inits
4303   #ifdef ASSERT
4304   intptr_t first_offset = allocation()->minimum_header_size();
4305   intptr_t last_init_off = first_offset;  // previous init offset
4306   intptr_t last_init_end = first_offset;  // previous init offset+size
4307   intptr_t last_tile_end = first_offset;  // previous tile offset+size
4308   #endif
4309   intptr_t zeroes_done = header_size;
4310 
4311   bool do_zeroing = true;       // we might give up if inits are very sparse
4312   int  big_init_gaps = 0;       // how many large gaps have we seen?
4313 
4314   if (UseTLAB && ZeroTLAB)  do_zeroing = false;
4315   if (!ReduceFieldZeroing && !ReduceBulkZeroing)  do_zeroing = false;
4316 
4317   for (uint i = InitializeNode::RawStores, limit = req(); i < limit; i++) {
4318     Node* st = in(i);
4319     intptr_t st_off = get_store_offset(st, phase);
4320     if (st_off < 0)
4321       break;                    // unknown junk in the inits
4322     if (st->in(MemNode::Memory) != zmem)
4323       break;                    // complicated store chains somehow in list
4324 
4325     int st_size = st->as_Store()->memory_size();
4326     intptr_t next_init_off = st_off + st_size;
4327 
4328     if (do_zeroing && zeroes_done < next_init_off) {
4329       // See if this store needs a zero before it or under it.
4330       intptr_t zeroes_needed = st_off;
4331 
4332       if (st_size < BytesPerInt) {
4333         // Look for subword stores which only partially initialize words.
4334         // If we find some, we must lay down some word-level zeroes first,
4335         // underneath the subword stores.
4336         //
4337         // Examples:
4338         //   byte[] a = { p,q,r,s }  =>  a[0]=p,a[1]=q,a[2]=r,a[3]=s
4339         //   byte[] a = { x,y,0,0 }  =>  a[0..3] = 0, a[0]=x,a[1]=y
4340         //   byte[] a = { 0,0,z,0 }  =>  a[0..3] = 0, a[2]=z
4341         //
4342         // Note:  coalesce_subword_stores may have already done this,
4343         // if it was prompted by constant non-zero subword initializers.
4344         // But this case can still arise with non-constant stores.
4345 
4346         intptr_t next_full_store = find_next_fullword_store(i, phase);
4347 
4348         // In the examples above:
4349         //   in(i)          p   q   r   s     x   y     z
4350         //   st_off        12  13  14  15    12  13    14
4351         //   st_size        1   1   1   1     1   1     1
4352         //   next_full_s.  12  16  16  16    16  16    16
4353         //   z's_done      12  16  16  16    12  16    12
4354         //   z's_needed    12  16  16  16    16  16    16
4355         //   zsize          0   0   0   0     4   0     4
4356         if (next_full_store < 0) {
4357           // Conservative tack:  Zero to end of current word.
4358           zeroes_needed = align_up(zeroes_needed, BytesPerInt);
4359         } else {
4360           // Zero to beginning of next fully initialized word.
4361           // Or, don't zero at all, if we are already in that word.
4362           assert(next_full_store >= zeroes_needed, "must go forward");
4363           assert((next_full_store & (BytesPerInt-1)) == 0, "even boundary");
4364           zeroes_needed = next_full_store;
4365         }
4366       }
4367 
4368       if (zeroes_needed > zeroes_done) {
4369         intptr_t zsize = zeroes_needed - zeroes_done;
4370         // Do some incremental zeroing on rawmem, in parallel with inits.
4371         zeroes_done = align_down(zeroes_done, BytesPerInt);
4372         rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4373                                               allocation()->in(AllocateNode::DefaultValue),
4374                                               allocation()->in(AllocateNode::RawDefaultValue),
4375                                               zeroes_done, zeroes_needed,
4376                                               phase);
4377         zeroes_done = zeroes_needed;
4378         if (zsize > InitArrayShortSize && ++big_init_gaps > 2)
4379           do_zeroing = false;   // leave the hole, next time
4380       }
4381     }
4382 
4383     // Collect the store and move on:
4384     phase->replace_input_of(st, MemNode::Memory, inits);
4385     inits = st;                 // put it on the linearized chain
4386     set_req(i, zmem);           // unhook from previous position
4387 
4388     if (zeroes_done == st_off)
4389       zeroes_done = next_init_off;
4390 
4391     assert(!do_zeroing || zeroes_done >= next_init_off, "don't miss any");
4392 
4393     #ifdef ASSERT
4394     // Various order invariants.  Weaker than stores_are_sane because
4395     // a large constant tile can be filled in by smaller non-constant stores.
4396     assert(st_off >= last_init_off, "inits do not reverse");
4397     last_init_off = st_off;
4398     const Type* val = NULL;
4399     if (st_size >= BytesPerInt &&
4400         (val = phase->type(st->in(MemNode::ValueIn)))->singleton() &&
4401         (int)val->basic_type() < (int)T_OBJECT) {
4402       assert(st_off >= last_tile_end, "tiles do not overlap");
4403       assert(st_off >= last_init_end, "tiles do not overwrite inits");
4404       last_tile_end = MAX2(last_tile_end, next_init_off);
4405     } else {
4406       intptr_t st_tile_end = align_up(next_init_off, BytesPerLong);
4407       assert(st_tile_end >= last_tile_end, "inits stay with tiles");
4408       assert(st_off      >= last_init_end, "inits do not overlap");
4409       last_init_end = next_init_off;  // it's a non-tile
4410     }
4411     #endif //ASSERT
4412   }
4413 
4414   remove_extra_zeroes();        // clear out all the zmems left over
4415   add_req(inits);
4416 
4417   if (!(UseTLAB && ZeroTLAB)) {
4418     // If anything remains to be zeroed, zero it all now.
4419     zeroes_done = align_down(zeroes_done, BytesPerInt);
4420     // if it is the last unused 4 bytes of an instance, forget about it
4421     intptr_t size_limit = phase->find_intptr_t_con(size_in_bytes, max_jint);
4422     if (zeroes_done + BytesPerLong >= size_limit) {
4423       AllocateNode* alloc = allocation();
4424       assert(alloc != NULL, "must be present");
4425       if (alloc != NULL && alloc->Opcode() == Op_Allocate) {
4426         Node* klass_node = alloc->in(AllocateNode::KlassNode);
4427         ciKlass* k = phase->type(klass_node)->is_klassptr()->klass();
4428         if (zeroes_done == k->layout_helper())
4429           zeroes_done = size_limit;
4430       }
4431     }
4432     if (zeroes_done < size_limit) {
4433       rawmem = ClearArrayNode::clear_memory(rawctl, rawmem, rawptr,
4434                                             allocation()->in(AllocateNode::DefaultValue),
4435                                             allocation()->in(AllocateNode::RawDefaultValue),
4436                                             zeroes_done, size_in_bytes, phase);
4437     }
4438   }
4439 
4440   set_complete(phase);
4441   return rawmem;
4442 }

4470 //============================MergeMemNode=====================================
4471 //
4472 // SEMANTICS OF MEMORY MERGES:  A MergeMem is a memory state assembled from several
4473 // contributing store or call operations.  Each contributor provides the memory
4474 // state for a particular "alias type" (see Compile::alias_type).  For example,
4475 // if a MergeMem has an input X for alias category #6, then any memory reference
4476 // to alias category #6 may use X as its memory state input, as an exact equivalent
4477 // to using the MergeMem as a whole.
4478 //   Load<6>( MergeMem(<6>: X, ...), p ) <==> Load<6>(X,p)
4479 //
4480 // (Here, the <N> notation gives the index of the relevant adr_type.)
4481 //
4482 // In one special case (and more cases in the future), alias categories overlap.
4483 // The special alias category "Bot" (Compile::AliasIdxBot) includes all memory
4484 // states.  Therefore, if a MergeMem has only one contributing input W for Bot,
4485 // it is exactly equivalent to that state W:
4486 //   MergeMem(<Bot>: W) <==> W
4487 //
4488 // Usually, the merge has more than one input.  In that case, where inputs
4489 // overlap (i.e., one is Bot), the narrower alias type determines the memory
4490 // state for that type, and the wider alias type (Bot) fills in everywhere else:
4491 //   Load<5>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<5>(W,p)
4492 //   Load<6>( MergeMem(<Bot>: W, <6>: X), p ) <==> Load<6>(X,p)
4493 //
4494 // A merge can take a "wide" memory state as one of its narrow inputs.
4495 // This simply means that the merge observes out only the relevant parts of
4496 // the wide input.  That is, wide memory states arriving at narrow merge inputs
4497 // are implicitly "filtered" or "sliced" as necessary.  (This is rare.)
4498 //
4499 // These rules imply that MergeMem nodes may cascade (via their <Bot> links),
4500 // and that memory slices "leak through":
4501 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y)) <==> MergeMem(<Bot>: W, <7>: Y)
4502 //
4503 // But, in such a cascade, repeated memory slices can "block the leak":
4504 //   MergeMem(<Bot>: MergeMem(<Bot>: W, <7>: Y), <7>: Y') <==> MergeMem(<Bot>: W, <7>: Y')
4505 //
4506 // In the last example, Y is not part of the combined memory state of the
4507 // outermost MergeMem.  The system must, of course, prevent unschedulable
4508 // memory states from arising, so you can be sure that the state Y is somehow
4509 // a precursor to state Y'.
4510 //
4511 //
4512 // REPRESENTATION OF MEMORY MERGES: The indexes used to address the Node::in array
4513 // of each MergeMemNode array are exactly the numerical alias indexes, including
4514 // but not limited to AliasIdxTop, AliasIdxBot, and AliasIdxRaw.  The functions
4515 // Compile::alias_type (and kin) produce and manage these indexes.
4516 //
4517 // By convention, the value of in(AliasIdxTop) (i.e., in(1)) is always the top node.
4518 // (Note that this provides quick access to the top node inside MergeMem methods,
4519 // without the need to reach out via TLS to Compile::current.)
4520 //
4521 // As a consequence of what was just described, a MergeMem that represents a full
4522 // memory state has an edge in(AliasIdxBot) which is a "wide" memory state,
4523 // containing all alias categories.
4524 //
4525 // MergeMem nodes never (?) have control inputs, so in(0) is NULL.
4526 //
4527 // All other edges in(N) (including in(AliasIdxRaw), which is in(3)) are either
4528 // a memory state for the alias type <N>, or else the top node, meaning that
4529 // there is no particular input for that alias type.  Note that the length of
4530 // a MergeMem is variable, and may be extended at any time to accommodate new
4531 // memory states at larger alias indexes.  When merges grow, they are of course
4532 // filled with "top" in the unused in() positions.
4533 //
4534 // This use of top is named "empty_memory()", or "empty_mem" (no-memory) as a variable.
4535 // (Top was chosen because it works smoothly with passes like GCM.)
4536 //
4537 // For convenience, we hardwire the alias index for TypeRawPtr::BOTTOM.  (It is
4538 // the type of random VM bits like TLS references.)  Since it is always the
4539 // first non-Bot memory slice, some low-level loops use it to initialize an
4540 // index variable:  for (i = AliasIdxRaw; i < req(); i++).
4541 //
4542 //
4543 // ACCESSORS:  There is a special accessor MergeMemNode::base_memory which returns
4544 // the distinguished "wide" state.  The accessor MergeMemNode::memory_at(N) returns
4545 // the memory state for alias type <N>, or (if there is no particular slice at <N>,
4546 // it returns the base memory.  To prevent bugs, memory_at does not accept <Top>
4547 // or <Bot> indexes.  The iterator MergeMemStream provides robust iteration over
4548 // MergeMem nodes or pairs of such nodes, ensuring that the non-top edges are visited.
4549 //
4550 // %%%% We may get rid of base_memory as a separate accessor at some point; it isn't
4551 // really that different from the other memory inputs.  An abbreviation called
4552 // "bot_memory()" for "memory_at(AliasIdxBot)" would keep code tidy.
4553 //
4554 //
4555 // PARTIAL MEMORY STATES:  During optimization, MergeMem nodes may arise that represent
4556 // partial memory states.  When a Phi splits through a MergeMem, the copy of the Phi
4557 // that "emerges though" the base memory will be marked as excluding the alias types
4558 // of the other (narrow-memory) copies which "emerged through" the narrow edges:
4559 //
4560 //   Phi<Bot>(U, MergeMem(<Bot>: W, <8>: Y))
4561 //     ==Ideal=>  MergeMem(<Bot>: Phi<Bot-8>(U, W), Phi<8>(U, Y))
4562 //
4563 // This strange "subtraction" effect is necessary to ensure IGVN convergence.
4564 // (It is currently unimplemented.)  As you can see, the resulting merge is
4565 // actually a disjoint union of memory states, rather than an overlay.
4566 //

4628 //------------------------------Ideal------------------------------------------
4629 // This method is invoked recursively on chains of MergeMem nodes
4630 Node *MergeMemNode::Ideal(PhaseGVN *phase, bool can_reshape) {
4631   // Remove chain'd MergeMems
4632   //
4633   // This is delicate, because the each "in(i)" (i >= Raw) is interpreted
4634   // relative to the "in(Bot)".  Since we are patching both at the same time,
4635   // we have to be careful to read each "in(i)" relative to the old "in(Bot)",
4636   // but rewrite each "in(i)" relative to the new "in(Bot)".
4637   Node *progress = NULL;
4638 
4639 
4640   Node* old_base = base_memory();
4641   Node* empty_mem = empty_memory();
4642   if (old_base == empty_mem)
4643     return NULL; // Dead memory path.
4644 
4645   MergeMemNode* old_mbase;
4646   if (old_base != NULL && old_base->is_MergeMem())
4647     old_mbase = old_base->as_MergeMem();
4648   else
4649     old_mbase = NULL;
4650   Node* new_base = old_base;
4651 
4652   // simplify stacked MergeMems in base memory
4653   if (old_mbase)  new_base = old_mbase->base_memory();
4654 
4655   // the base memory might contribute new slices beyond my req()
4656   if (old_mbase)  grow_to_match(old_mbase);
4657 
4658   // Look carefully at the base node if it is a phi.
4659   PhiNode* phi_base;
4660   if (new_base != NULL && new_base->is_Phi())
4661     phi_base = new_base->as_Phi();
4662   else
4663     phi_base = NULL;
4664 
4665   Node*    phi_reg = NULL;
4666   uint     phi_len = (uint)-1;
4667   if (phi_base != NULL && !phi_base->is_copy()) {
4668     // do not examine phi if degraded to a copy
4669     phi_reg = phi_base->region();
4670     phi_len = phi_base->req();
4671     // see if the phi is unfinished
4672     for (uint i = 1; i < phi_len; i++) {
4673       if (phi_base->in(i) == NULL) {
4674         // incomplete phi; do not look at it yet!
4675         phi_reg = NULL;
4676         phi_len = (uint)-1;
4677         break;
4678       }
4679     }
4680   }
4681 
4682   // Note:  We do not call verify_sparse on entry, because inputs
4683   // can normalize to the base_memory via subsume_node or similar
4684   // mechanisms.  This method repairs that damage.
4685 
4686   assert(!old_mbase || old_mbase->is_empty_memory(empty_mem), "consistent sentinels");
4687 
4688   // Look at each slice.
4689   for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4690     Node* old_in = in(i);
4691     // calculate the old memory value
4692     Node* old_mem = old_in;
4693     if (old_mem == empty_mem)  old_mem = old_base;
4694     assert(old_mem == memory_at(i), "");
4695 
4696     // maybe update (reslice) the old memory value
4697 
4698     // simplify stacked MergeMems
4699     Node* new_mem = old_mem;
4700     MergeMemNode* old_mmem;
4701     if (old_mem != NULL && old_mem->is_MergeMem())
4702       old_mmem = old_mem->as_MergeMem();
4703     else
4704       old_mmem = NULL;
4705     if (old_mmem == this) {
4706       // This can happen if loops break up and safepoints disappear.
4707       // A merge of BotPtr (default) with a RawPtr memory derived from a
4708       // safepoint can be rewritten to a merge of the same BotPtr with
4709       // the BotPtr phi coming into the loop.  If that phi disappears
4710       // also, we can end up with a self-loop of the mergemem.
4711       // In general, if loops degenerate and memory effects disappear,
4712       // a mergemem can be left looking at itself.  This simply means
4713       // that the mergemem's default should be used, since there is
4714       // no longer any apparent effect on this slice.
4715       // Note: If a memory slice is a MergeMem cycle, it is unreachable
4716       //       from start.  Update the input to TOP.
4717       new_mem = (new_base == this || new_base == empty_mem)? empty_mem : new_base;
4718     }
4719     else if (old_mmem != NULL) {
4720       new_mem = old_mmem->memory_at(i);
4721     }
4722     // else preceding memory was not a MergeMem
4723 
4724     // replace equivalent phis (unfortunately, they do not GVN together)
4725     if (new_mem != NULL && new_mem != new_base &&
4726         new_mem->req() == phi_len && new_mem->in(0) == phi_reg) {
4727       if (new_mem->is_Phi()) {
4728         PhiNode* phi_mem = new_mem->as_Phi();
4729         for (uint i = 1; i < phi_len; i++) {
4730           if (phi_base->in(i) != phi_mem->in(i)) {
4731             phi_mem = NULL;
4732             break;
4733           }
4734         }
4735         if (phi_mem != NULL) {
4736           // equivalent phi nodes; revert to the def
4737           new_mem = new_base;
4738         }
4739       }
4740     }
4741 
4742     // maybe store down a new value
4743     Node* new_in = new_mem;
4744     if (new_in == new_base)  new_in = empty_mem;
4745 
4746     if (new_in != old_in) {
4747       // Warning:  Do not combine this "if" with the previous "if"
4748       // A memory slice might have be be rewritten even if it is semantically
4749       // unchanged, if the base_memory value has changed.
4750       set_req(i, new_in);
4751       progress = this;          // Report progress
4752     }
4753   }
4754 
4755   if (new_base != old_base) {
4756     set_req(Compile::AliasIdxBot, new_base);
4757     // Don't use set_base_memory(new_base), because we need to update du.
4758     assert(base_memory() == new_base, "");
4759     progress = this;
4760   }
4761 
4762   if( base_memory() == this ) {
4763     // a self cycle indicates this memory path is dead
4764     set_req(Compile::AliasIdxBot, empty_mem);
4765   }
4766 
4767   // Resolve external cycles by calling Ideal on a MergeMem base_memory
4768   // Recursion must occur after the self cycle check above
4769   if( base_memory()->is_MergeMem() ) {
4770     MergeMemNode *new_mbase = base_memory()->as_MergeMem();
4771     Node *m = phase->transform(new_mbase);  // Rollup any cycles
4772     if( m != NULL &&
4773         (m->is_top() ||
4774          (m->is_MergeMem() && m->as_MergeMem()->base_memory() == empty_mem)) ) {
4775       // propagate rollup of dead cycle to self
4776       set_req(Compile::AliasIdxBot, empty_mem);
4777     }
4778   }
4779 
4780   if( base_memory() == empty_mem ) {
4781     progress = this;
4782     // Cut inputs during Parse phase only.
4783     // During Optimize phase a dead MergeMem node will be subsumed by Top.
4784     if( !can_reshape ) {
4785       for (uint i = Compile::AliasIdxRaw; i < req(); i++) {
4786         if( in(i) != empty_mem ) { set_req(i, empty_mem); }
4787       }
4788     }
4789   }
4790 
4791   if( !progress && base_memory()->is_Phi() && can_reshape ) {
4792     // Check if PhiNode::Ideal's "Split phis through memory merges"
4793     // transform should be attempted. Look for this->phi->this cycle.
4794     uint merge_width = req();
4795     if (merge_width > Compile::AliasIdxRaw) {
4796       PhiNode* phi = base_memory()->as_Phi();
4797       for( uint i = 1; i < phi->req(); ++i ) {// For all paths in
4798         if (phi->in(i) == this) {
4799           phase->is_IterGVN()->_worklist.push(phi);
4800           break;
4801         }
4802       }
4803     }
4804   }
4805 
4806   assert(progress || verify_sparse(), "please, no dups of base");
4807   return progress;
4808 }

4853 static void verify_memory_slice(const MergeMemNode* m, int alias_idx, Node* n) {
4854   if (!VerifyAliases)                return;  // don't bother to verify unless requested
4855   if (VMError::is_error_reported())  return;  // muzzle asserts when debugging an error
4856   if (Node::in_dump())               return;  // muzzle asserts when printing
4857   assert(alias_idx >= Compile::AliasIdxRaw, "must not disturb base_memory or sentinel");
4858   assert(n != NULL, "");
4859   // Elide intervening MergeMem's
4860   while (n->is_MergeMem()) {
4861     n = n->as_MergeMem()->memory_at(alias_idx);
4862   }
4863   Compile* C = Compile::current();
4864   const TypePtr* n_adr_type = n->adr_type();
4865   if (n == m->empty_memory()) {
4866     // Implicit copy of base_memory()
4867   } else if (n_adr_type != TypePtr::BOTTOM) {
4868     assert(n_adr_type != NULL, "new memory must have a well-defined adr_type");
4869     assert(C->must_alias(n_adr_type, alias_idx), "new memory must match selected slice");
4870   } else {
4871     // A few places like make_runtime_call "know" that VM calls are narrow,
4872     // and can be used to update only the VM bits stored as TypeRawPtr::BOTTOM.
4873     bool expected_wide_mem = false;
4874     if (n == m->base_memory()) {
4875       expected_wide_mem = true;
4876     } else if (alias_idx == Compile::AliasIdxRaw ||
4877                n == m->memory_at(Compile::AliasIdxRaw)) {
4878       expected_wide_mem = true;
4879     } else if (!C->alias_type(alias_idx)->is_rewritable()) {
4880       // memory can "leak through" calls on channels that
4881       // are write-once.  Allow this also.
4882       expected_wide_mem = true;
4883     }
4884     assert(expected_wide_mem, "expected narrow slice replacement");
4885   }
4886 }

4893 Node* MergeMemNode::memory_at(uint alias_idx) const {
4894   assert(alias_idx >= Compile::AliasIdxRaw ||
4895          alias_idx == Compile::AliasIdxBot && Compile::current()->AliasLevel() == 0,
4896          "must avoid base_memory and AliasIdxTop");
4897 
4898   // Otherwise, it is a narrow slice.
4899   Node* n = alias_idx < req() ? in(alias_idx) : empty_memory();
4900   Compile *C = Compile::current();
4901   if (is_empty_memory(n)) {
4902     // the array is sparse; empty slots are the "top" node
4903     n = base_memory();
4904     assert(Node::in_dump()
4905            || n == NULL || n->bottom_type() == Type::TOP
4906            || n->adr_type() == NULL // address is TOP
4907            || n->adr_type() == TypePtr::BOTTOM
4908            || n->adr_type() == TypeRawPtr::BOTTOM
4909            || Compile::current()->AliasLevel() == 0,
4910            "must be a wide memory");

4912     // See verify_memory_slice for comments on TypeRawPtr::BOTTOM.

4914     // make sure the stored slice is sane
4915     #ifdef ASSERT
4916     if (VMError::is_error_reported() || Node::in_dump()) {
4917     } else if (might_be_same(n, base_memory())) {
4918       // Give it a pass:  It is a mostly harmless repetition of the base.
4919       // This can arise normally from node subsumption during optimization.
4920     } else {

4922     }
