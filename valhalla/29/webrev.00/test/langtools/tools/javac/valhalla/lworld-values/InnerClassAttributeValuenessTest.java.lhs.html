24 /*
25  * @test
26  * @bug 8221330
27  * @summary Javac adds InnerClass attribute missing value flag
28  * @modules jdk.jdeps/com.sun.tools.classfile
29  * @run main/othervm InnerClassAttributeValuenessTest
30  */
31 
32 import com.sun.tools.classfile.AccessFlags;
33 
34 public class InnerClassAttributeValuenessTest {
35 
36     static inline class Inner {
37         int f;
38         private Inner() { f=0; }
39         private Inner(int v) { f=v; }
40 
41         public static Inner create(int v) {
42             return new Inner(v);
43         }
44 
45         // Uncomment the next line, and Inner ceases to be a value type
46         public static final Inner? ZERO = Inner.create(0);
47         public static final Inner? ZERO2 = Inner.create(0);
48     }
49 
50     public static void main(String[] args) {
51         if ((Inner.class.getModifiers() & AccessFlags.ACC_VALUE) == 0)
52             throw new AssertionError("Value flag missing");
53     }
54 }
