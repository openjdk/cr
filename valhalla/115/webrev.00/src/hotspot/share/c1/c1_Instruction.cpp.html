  40 int Instruction::dominator_depth() {
  41   int result = -1;
  42   if (block()) {
  43     result = block()->dominator_depth();
  44   }
  45   assert(result != -1 || this->as_Local(), "Only locals have dominator depth -1");
  46   return result;
  47 }

  49 Instruction::Condition Instruction::mirror(Condition cond) {
  50   switch (cond) {
  51     case eql: return eql;
  52     case neq: return neq;
  53     case lss: return gtr;
  54     case leq: return geq;
  55     case gtr: return lss;
  56     case geq: return leq;
  57     case aeq: return beq;
  58     case beq: return aeq;
  59   }
  60   ShouldNotReachHere();
  61   return eql;
  62 }

  64 
  65 Instruction::Condition Instruction::negate(Condition cond) {
  66   switch (cond) {
  67     case eql: return neq;
  68     case neq: return eql;
  69     case lss: return geq;
  70     case leq: return gtr;
  71     case gtr: return leq;
  72     case geq: return lss;
  73     case aeq: assert(false, "Above equal cannot be negated");
  74     case beq: assert(false, "Below equal cannot be negated");
  75   }
  76   ShouldNotReachHere();
  77   return eql;
  78 }

  80 void Instruction::update_exception_state(ValueStack* state) {
  81   if (state != NULL && (state->kind() == ValueStack::EmptyExceptionState || state->kind() == ValueStack::ExceptionState)) {
  82     assert(state->kind() == ValueStack::EmptyExceptionState || Compilation::current()->env()->should_retain_local_variables(), "unexpected state kind");
  83     _exception_state = state;
  84   } else {
  85     _exception_state = NULL;
  86   }
  87 }

  89 // Prev without need to have BlockBegin
  90 Instruction* Instruction::prev() {
  91   Instruction* p = NULL;
  92   Instruction* q = block();
  93   while (q != this) {
  94     assert(q != NULL, "this is not in the block's instruction list");
  95     p = q; q = q->next();
  96   }
  97   return p;
  98 }

 137 bool Instruction::maybe_flattened_array() {
 138   if (FlatArrayFlatten) {
 139     ciType* type = declared_type();
 140     if (type != NULL) {
 141       if (type->is_obj_array_klass()) {
 142         // Due to array covariance, the runtime type might be a flattened array.
 143         ciKlass* element_klass = type->as_obj_array_klass()->element_klass();
 144         if (element_klass->can_be_value_klass() && (!element_klass->is_valuetype() || element_klass->as_value_klass()->flatten_array())) {
 145           return true;
 146         }
 147       } else if (type->is_value_array_klass()) {
 148         ciKlass* element_klass = type->as_value_array_klass()->element_klass();
 149         assert(!element_klass->is_loaded() || element_klass->as_value_klass()->flatten_array(), "must be flattened");
 150         return true;
 151       } else if (type->is_klass() && type->as_klass()->is_java_lang_Object()) {
 152         // This can happen as a parameter to System.arraycopy()
 153         return true;
 154       }
 155     } else {
 156       // Type info gets lost during Phi merging (Phi, IfOp, etc), but we might be storing into a
 157       // flattened array, so we should do a runtime check.
 158       return true;
 159     }
 160   }
 161   return false;
 162 }
 163 
 164 bool Instruction::maybe_null_free_array() {
 165   ciType* type = declared_type();
 166   if (type != NULL) {
 167     if (type->is_obj_array_klass()) {
 168       // Due to array covariance, the runtime type might be a null-free array.
 169       ciKlass* element_klass = type->as_obj_array_klass()->element_klass();
 170       if (element_klass->can_be_value_klass()) {
 171         return true;
 172       }
 173     }
 174   } else {
 175     // Type info gets lost during Phi merging (Phi, IfOp, etc), but we might be storing into a
 176     // null-free array, so we should do a runtime check.
 177     return true;
 178   }
 179   return false;
 180 }

 210 // perform constant and interval tests on index value
 211 bool AccessIndexed::compute_needs_range_check() {
 212   if (length()) {
 213     Constant* clength = length()->as_Constant();
 214     Constant* cindex = index()->as_Constant();
 215     if (clength && cindex) {
 216       IntConstant* l = clength->type()->as_IntConstant();
 217       IntConstant* i = cindex->type()->as_IntConstant();
 218       if (l && i && i->value() < l->value() && i->value() >= 0) {
 219         return false;
 220       }
 221     }
 222   }
 223 
 224   if (!this->check_flag(NeedsRangeCheckFlag)) {
 225     return false;
 226   }
 227 
 228   return true;
 229 }

 266 bool StoreIndexed::is_exact_flattened_array_store() const {
 267   if (array()->is_loaded_flattened_array() && value()->as_Constant() == NULL && value()->declared_type() != NULL) {
 268     ciKlass* element_klass = array()->declared_type()->as_value_array_klass()->element_klass();
 269     ciKlass* actual_klass = value()->declared_type()->as_klass();
 270 
 271     // The following check can fail with inlining:
 272     //     void test45_inline(Object[] oa, Object o, int index) { oa[index] = o; }
 273     //     void test45(MyValue1[] va, int index, MyValue2 v) { test45_inline(va, v, index); }
 274     if (element_klass == actual_klass) {
 275       return true;
 276     }
 277   }
 278   return false;
 279 }

 331 // Implementation of ArithmeticOp
 332 
 333 bool ArithmeticOp::is_commutative() const {
 334   switch (op()) {
 335     case Bytecodes::_iadd: // fall through
 336     case Bytecodes::_ladd: // fall through
 337     case Bytecodes::_fadd: // fall through
 338     case Bytecodes::_dadd: // fall through
 339     case Bytecodes::_imul: // fall through
 340     case Bytecodes::_lmul: // fall through
 341     case Bytecodes::_fmul: // fall through
 342     case Bytecodes::_dmul: return true;
 343     default              : return false;
 344   }
 345 }

 348 bool ArithmeticOp::can_trap() const {
 349   switch (op()) {
 350     case Bytecodes::_idiv: // fall through
 351     case Bytecodes::_ldiv: // fall through
 352     case Bytecodes::_irem: // fall through
 353     case Bytecodes::_lrem: return true;
 354     default              : return false;
 355   }
 356 }

 359 // Implementation of LogicOp
 360 
 361 bool LogicOp::is_commutative() const {
 362 #ifdef ASSERT
 363   switch (op()) {
 364     case Bytecodes::_iand: // fall through
 365     case Bytecodes::_land: // fall through
 366     case Bytecodes::_ior : // fall through
 367     case Bytecodes::_lor : // fall through
 368     case Bytecodes::_ixor: // fall through
 369     case Bytecodes::_lxor: break;
 370     default              : ShouldNotReachHere(); break;
 371   }
 372 #endif
 373   // all LogicOps are commutative
 374   return true;
 375 }

 385 // Implementation of StateSplit
 386 
 387 void StateSplit::substitute(BlockList& list, BlockBegin* old_block, BlockBegin* new_block) {
 388   NOT_PRODUCT(bool assigned = false;)
 389   for (int i = 0; i < list.length(); i++) {
 390     BlockBegin** b = list.adr_at(i);
 391     if (*b == old_block) {
 392       *b = new_block;
 393       NOT_PRODUCT(assigned = true;)
 394     }
 395   }
 396   assert(assigned == true, "should have assigned at least once");
 397 }

 423 
 424 
 425 Invoke::Invoke(Bytecodes::Code code, ValueType* result_type, Value recv, Values* args,
 426                int vtable_index, ciMethod* target, ValueStack* state_before, bool never_null)
 427   : StateSplit(result_type, state_before)
 428   , _code(code)
 429   , _recv(recv)
 430   , _args(args)
 431   , _vtable_index(vtable_index)
 432   , _target(target)
 433 {
 434   set_flag(TargetIsLoadedFlag,   target->is_loaded());
 435   set_flag(TargetIsFinalFlag,    target_is_loaded() && target->is_final_method());
 436   set_flag(TargetIsStrictfpFlag, target_is_loaded() && target->is_strict());
 437   set_never_null(never_null);
 438 
 439   assert(args != NULL, "args must exist");
 440 #ifdef ASSERT
 441   AssertValues assert_value;
 442   values_do(&assert_value);
 443 #endif
 444 
 445   // provide an initial guess of signature size.
 446   _signature = new BasicTypeList(number_of_arguments() + (has_receiver() ? 1 : 0));
 447   if (has_receiver()) {
 448     _signature->append(as_BasicType(receiver()->type()));
 449   }
 450   for (int i = 0; i < number_of_arguments(); i++) {
 451     ValueType* t = argument_at(i)->type();
 452     BasicType bt = as_BasicType(t);
 453     _signature->append(bt);
 454   }
 455 }

 471 // Implementation of Contant
 472 intx Constant::hash() const {
 473   if (state_before() == NULL) {
 474     switch (type()->tag()) {
 475     case intTag:
 476       return HASH2(name(), type()->as_IntConstant()->value());
 477     case addressTag:
 478       return HASH2(name(), type()->as_AddressConstant()->value());
 479     case longTag:
 480       {
 481         jlong temp = type()->as_LongConstant()->value();
 482         return HASH3(name(), high(temp), low(temp));
 483       }
 484     case floatTag:
 485       return HASH2(name(), jint_cast(type()->as_FloatConstant()->value()));
 486     case doubleTag:
 487       {
 488         jlong temp = jlong_cast(type()->as_DoubleConstant()->value());
 489         return HASH3(name(), high(temp), low(temp));
 490       }
 491     case objectTag:
 492       assert(type()->as_ObjectType()->is_loaded(), "can't handle unloaded values");
 493       return HASH2(name(), type()->as_ObjectType()->constant_value());
 494     case metaDataTag:
 495       assert(type()->as_MetadataType()->is_loaded(), "can't handle unloaded values");
 496       return HASH2(name(), type()->as_MetadataType()->constant_value());
 497     default:
 498       ShouldNotReachHere();
 499     }
 500   }
 501   return 0;
 502 }

 504 bool Constant::is_equal(Value v) const {
 505   if (v->as_Constant() == NULL) return false;
 506 
 507   switch (type()->tag()) {
 508     case intTag:
 509       {
 510         IntConstant* t1 =    type()->as_IntConstant();
 511         IntConstant* t2 = v->type()->as_IntConstant();
 512         return (t1 != NULL && t2 != NULL &&
 513                 t1->value() == t2->value());
 514       }
 515     case longTag:
 516       {
 517         LongConstant* t1 =    type()->as_LongConstant();
 518         LongConstant* t2 = v->type()->as_LongConstant();
 519         return (t1 != NULL && t2 != NULL &&
 520                 t1->value() == t2->value());
 521       }
 522     case floatTag:
 523       {
 524         FloatConstant* t1 =    type()->as_FloatConstant();
 525         FloatConstant* t2 = v->type()->as_FloatConstant();
 526         return (t1 != NULL && t2 != NULL &&
 527                 jint_cast(t1->value()) == jint_cast(t2->value()));
 528       }
 529     case doubleTag:
 530       {
 531         DoubleConstant* t1 =    type()->as_DoubleConstant();
 532         DoubleConstant* t2 = v->type()->as_DoubleConstant();
 533         return (t1 != NULL && t2 != NULL &&
 534                 jlong_cast(t1->value()) == jlong_cast(t2->value()));
 535       }
 536     case objectTag:
 537       {
 538         ObjectType* t1 =    type()->as_ObjectType();
 539         ObjectType* t2 = v->type()->as_ObjectType();
 540         return (t1 != NULL && t2 != NULL &&
 541                 t1->is_loaded() && t2->is_loaded() &&
 542                 t1->constant_value() == t2->constant_value());
 543       }
 544     case metaDataTag:
 545       {
 546         MetadataType* t1 =    type()->as_MetadataType();
 547         MetadataType* t2 = v->type()->as_MetadataType();
 548         return (t1 != NULL && t2 != NULL &&
 549                 t1->is_loaded() && t2->is_loaded() &&
 550                 t1->constant_value() == t2->constant_value());
 551       }
 552     default:
 553       return false;
 554   }
 555 }

 557 Constant::CompareResult Constant::compare(Instruction::Condition cond, Value right) const {
 558   Constant* rc = right->as_Constant();
 559   // other is not a constant
 560   if (rc == NULL) return not_comparable;
 561 
 562   ValueType* lt = type();
 563   ValueType* rt = rc->type();
 564   // different types
 565   if (lt->base() != rt->base()) return not_comparable;
 566   switch (lt->tag()) {
 567   case intTag: {
 568     int x = lt->as_IntConstant()->value();
 569     int y = rt->as_IntConstant()->value();
 570     switch (cond) {
 571     case If::eql: return x == y ? cond_true : cond_false;
 572     case If::neq: return x != y ? cond_true : cond_false;
 573     case If::lss: return x <  y ? cond_true : cond_false;
 574     case If::leq: return x <= y ? cond_true : cond_false;
 575     case If::gtr: return x >  y ? cond_true : cond_false;
 576     case If::geq: return x >= y ? cond_true : cond_false;
 577     default     : break;
 578     }
 579     break;
 580   }
 581   case longTag: {
 582     jlong x = lt->as_LongConstant()->value();
 583     jlong y = rt->as_LongConstant()->value();
 584     switch (cond) {
 585     case If::eql: return x == y ? cond_true : cond_false;
 586     case If::neq: return x != y ? cond_true : cond_false;
 587     case If::lss: return x <  y ? cond_true : cond_false;
 588     case If::leq: return x <= y ? cond_true : cond_false;
 589     case If::gtr: return x >  y ? cond_true : cond_false;
 590     case If::geq: return x >= y ? cond_true : cond_false;
 591     default     : break;
 592     }
 593     break;
 594   }
 595   case objectTag: {
 596     ciObject* xvalue = lt->as_ObjectType()->constant_value();
 597     ciObject* yvalue = rt->as_ObjectType()->constant_value();
 598     assert(xvalue != NULL && yvalue != NULL, "not constants");
 599     if (xvalue->is_loaded() && yvalue->is_loaded()) {
 600       switch (cond) {
 601       case If::eql: return xvalue == yvalue ? cond_true : cond_false;
 602       case If::neq: return xvalue != yvalue ? cond_true : cond_false;
 603       default     : break;
 604       }
 605     }
 606     break;
 607   }
 608   case metaDataTag: {
 609     ciMetadata* xvalue = lt->as_MetadataType()->constant_value();
 610     ciMetadata* yvalue = rt->as_MetadataType()->constant_value();
 611     assert(xvalue != NULL && yvalue != NULL, "not constants");
 612     if (xvalue->is_loaded() && yvalue->is_loaded()) {
 613       switch (cond) {
 614       case If::eql: return xvalue == yvalue ? cond_true : cond_false;
 615       case If::neq: return xvalue != yvalue ? cond_true : cond_false;
 616       default     : break;
 617       }
 618     }
 619     break;
 620   }
 621   default:
 622     break;
 623   }
 624   return not_comparable;
 625 }

 628 // Implementation of BlockBegin
 629 
 630 void BlockBegin::set_end(BlockEnd* end) {
 631   assert(end != NULL, "should not reset block end to NULL");
 632   if (end == _end) {
 633     return;
 634   }
 635   clear_end();
 636 
 637   // Set the new end
 638   _end = end;
 639 
 640   _successors.clear();
 641   // Now reset successors list based on BlockEnd
 642   for (int i = 0; i < end->number_of_sux(); i++) {
 643     BlockBegin* sux = end->sux_at(i);
 644     _successors.append(sux);
 645     sux->_predecessors.append(this);
 646   }
 647   _end->set_begin(this);
 648 }
 649 
 650 
 651 void BlockBegin::clear_end() {
 652   // Must make the predecessors/successors match up with the
 653   // BlockEnd's notion.
 654   if (_end != NULL) {
 655     // disconnect from the old end
 656     _end->set_begin(NULL);
 657 
 658     // disconnect this block from it's current successors
 659     for (int i = 0; i < _successors.length(); i++) {
 660       _successors.at(i)->remove_predecessor(this);
 661     }
 662     _end = NULL;
 663   }
 664 }

 667 void BlockBegin::disconnect_edge(BlockBegin* from, BlockBegin* to) {
 668   // disconnect any edges between from and to
 669 #ifndef PRODUCT
 670   if (PrintIR && Verbose) {
 671     tty->print_cr("Disconnected edge B%d -> B%d", from->block_id(), to->block_id());
 672   }
 673 #endif
 674   for (int s = 0; s < from->number_of_sux();) {
 675     BlockBegin* sux = from->sux_at(s);
 676     if (sux == to) {
 677       int index = sux->_predecessors.find(from);
 678       if (index >= 0) {
 679         sux->_predecessors.remove_at(index);
 680       }
 681       from->_successors.remove_at(s);
 682     } else {
 683       s++;
 684     }
 685   }
 686 }

 689 void BlockBegin::disconnect_from_graph() {
 690   // disconnect this block from all other blocks
 691   for (int p = 0; p < number_of_preds(); p++) {
 692     pred_at(p)->remove_successor(this);
 693   }
 694   for (int s = 0; s < number_of_sux(); s++) {
 695     sux_at(s)->remove_predecessor(this);
 696   }

 699 void BlockBegin::substitute_sux(BlockBegin* old_sux, BlockBegin* new_sux) {
 700   // modify predecessors before substituting successors
 701   for (int i = 0; i < number_of_sux(); i++) {
 702     if (sux_at(i) == old_sux) {
 703       // remove old predecessor before adding new predecessor
 704       // otherwise there is a dead predecessor in the list
 705       new_sux->remove_predecessor(old_sux);
 706       new_sux->add_predecessor(this);
 707     }
 708   }
 709   old_sux->remove_predecessor(this);
 710   end()->substitute_sux(old_sux, new_sux);
 711 }

 715 // In general it is not possible to calculate a value for the field "depth_first_number"
 716 // of the inserted block, without recomputing the values of the other blocks
 717 // in the CFG. Therefore the value of "depth_first_number" in BlockBegin becomes meaningless.
 718 BlockBegin* BlockBegin::insert_block_between(BlockBegin* sux) {
 719   int bci = sux->bci();
 720   // critical edge splitting may introduce a goto after a if and array
 721   // bound check elimination may insert a predicate between the if and
 722   // goto. The bci of the goto can't be the one of the if otherwise
 723   // the state and bci are inconsistent and a deoptimization triggered
 724   // by the predicate would lead to incorrect execution/a crash.
 725   BlockBegin* new_sux = new BlockBegin(bci);
 726 
 727   // mark this block (special treatment when block order is computed)
 728   new_sux->set(critical_edge_split_flag);
 729 
 730   // This goto is not a safepoint.
 731   Goto* e = new Goto(sux, false);
 732   new_sux->set_next(e, bci);
 733   new_sux->set_end(e);
 734   // setup states
 735   ValueStack* s = end()->state();
 736   new_sux->set_state(s->copy(s->kind(), bci));
 737   e->set_state(s->copy(s->kind(), bci));
 738   assert(new_sux->state()->locals_size() == s->locals_size(), "local size mismatch!");
 739   assert(new_sux->state()->stack_size() == s->stack_size(), "stack size mismatch!");
 740   assert(new_sux->state()->locks_size() == s->locks_size(), "locks size mismatch!");
 741 
 742   // link predecessor to new block
 743   end()->substitute_sux(sux, new_sux);
 744 
 745   // The ordering needs to be the same, so remove the link that the
 746   // set_end call above added and substitute the new_sux for this
 747   // block.
 748   sux->remove_predecessor(new_sux);
 749 
 750   // the successor could be the target of a switch so it might have
 751   // multiple copies of this predecessor, so substitute the new_sux
 752   // for the first and delete the rest.
 753   bool assigned = false;
 754   BlockList& list = sux->_predecessors;
 755   for (int i = 0; i < list.length(); i++) {
 756     BlockBegin** b = list.adr_at(i);
 757     if (*b == this) {
 758       if (assigned) {
 759         list.remove_at(i);
 760         // reprocess this index
 761         i--;
 762       } else {
 763         assigned = true;
 764         *b = new_sux;
 765       }
 766       // link the new block back to it's predecessors.
 767       new_sux->add_predecessor(this);
 768     }
 769   }
 770   assert(assigned == true, "should have assigned at least once");
 771   return new_sux;
 772 }

 860 bool BlockBegin::try_merge(ValueStack* new_state) {
 861   TRACE_PHI(tty->print_cr("********** try_merge for block B%d", block_id()));
 862 
 863   // local variables used for state iteration
 864   int index;
 865   Value new_value, existing_value;
 866 
 867   ValueStack* existing_state = state();
 868   if (existing_state == NULL) {
 869     TRACE_PHI(tty->print_cr("first call of try_merge for this block"));
 870 
 871     if (is_set(BlockBegin::was_visited_flag)) {
 872       // this actually happens for complicated jsr/ret structures
 873       return false; // BAILOUT in caller
 874     }
 875 
 876     // copy state because it is altered
 877     new_state = new_state->copy(ValueStack::BlockBeginState, bci());
 878 
 879     // Use method liveness to invalidate dead locals
 880     MethodLivenessResult liveness = new_state->scope()->method()->liveness_at_bci(bci());
 881     if (liveness.is_valid()) {
 882       assert((int)liveness.size() == new_state->locals_size(), "error in use of liveness");
 883 
 884       for_each_local_value(new_state, index, new_value) {
 885         if (!liveness.at(index) || new_value->type()->is_illegal()) {
 886           new_state->invalidate_local(index);
 887           TRACE_PHI(tty->print_cr("invalidating dead local %d", index));
 888         }
 889       }
 890     }
 891 
 892     if (is_set(BlockBegin::parser_loop_header_flag)) {
 893       TRACE_PHI(tty->print_cr("loop header block, initializing phi functions"));
 894 
 895       for_each_stack_value(new_state, index, new_value) {
 896         new_state->setup_phi_for_stack(this, index);
 897         TRACE_PHI(tty->print_cr("creating phi-function %c%d for stack %d", new_state->stack_at(index)->type()->tchar(), new_state->stack_at(index)->id(), index));
 898       }
 899 
 900       BitMap& requires_phi_function = new_state->scope()->requires_phi_function();
 901 
 902       for_each_local_value(new_state, index, new_value) {
 903         bool requires_phi = requires_phi_function.at(index) || (new_value->type()->is_double_word() && requires_phi_function.at(index + 1));
 904         if (requires_phi || !SelectivePhiFunctions) {
 905           new_state->setup_phi_for_local(this, index);
 906           TRACE_PHI(tty->print_cr("creating phi-function %c%d for local %d", new_state->local_at(index)->type()->tchar(), new_state->local_at(index)->id(), index));
 907         }
 908       }
 909     }
 910 
 911     // initialize state of block
 912     set_state(new_state);
 913 
 914   } else if (existing_state->is_same(new_state)) {
 915     TRACE_PHI(tty->print_cr("exisiting state found"));
 916 
 917     assert(existing_state->scope() == new_state->scope(), "not matching");
 918     assert(existing_state->locals_size() == new_state->locals_size(), "not matching");
 919     assert(existing_state->stack_size() == new_state->stack_size(), "not matching");
 920 
 921     if (is_set(BlockBegin::was_visited_flag)) {
 922       TRACE_PHI(tty->print_cr("loop header block, phis must be present"));
 923 
 924       if (!is_set(BlockBegin::parser_loop_header_flag)) {
 925         // this actually happens for complicated jsr/ret structures
 926         return false; // BAILOUT in caller
 927       }
 928 
 929       for_each_local_value(existing_state, index, existing_value) {
 930         Value new_value = new_state->local_at(index);
 931         if (new_value == NULL || new_value->type()->tag() != existing_value->type()->tag()) {
 932           Phi* existing_phi = existing_value->as_Phi();
 933           if (existing_phi == NULL) {
 934             return false; // BAILOUT in caller
 935           }
 936           // Invalidate the phi function here. This case is very rare except for
 937           // JVMTI capability "can_access_local_variables".
 938           // In really rare cases we will bail out in LIRGenerator::move_to_phi.
 939           existing_phi->make_illegal();
 940           existing_state->invalidate_local(index);
 941           TRACE_PHI(tty->print_cr("invalidating local %d because of type mismatch", index));
 942         }
 943       }
 944 
 945 #ifdef ASSERT
 946       // check that all necessary phi functions are present
 947       for_each_stack_value(existing_state, index, existing_value) {
 948         assert(existing_value->as_Phi() != NULL && existing_value->as_Phi()->block() == this, "phi function required");
 949       }
 950       for_each_local_value(existing_state, index, existing_value) {
 951         assert(existing_value == new_state->local_at(index) || (existing_value->as_Phi() != NULL && existing_value->as_Phi()->as_Phi()->block() == this), "phi function required");
 952       }
 953 #endif
 954 
 955     } else {
 956       TRACE_PHI(tty->print_cr("creating phi functions on demand"));
 957 
 958       // create necessary phi functions for stack
 959       for_each_stack_value(existing_state, index, existing_value) {
 960         Value new_value = new_state->stack_at(index);
 961         Phi* existing_phi = existing_value->as_Phi();
 962 
 963         if (new_value != existing_value && (existing_phi == NULL || existing_phi->block() != this)) {
 964           existing_state->setup_phi_for_stack(this, index);
 965           TRACE_PHI(tty->print_cr("creating phi-function %c%d for stack %d", existing_state->stack_at(index)->type()->tchar(), existing_state->stack_at(index)->id(), index));
 966         }
 967       }
 968 
 969       // create necessary phi functions for locals
 970       for_each_local_value(existing_state, index, existing_value) {
 971         Value new_value = new_state->local_at(index);
 972         Phi* existing_phi = existing_value->as_Phi();
 973 
 974         if (new_value == NULL || new_value->type()->tag() != existing_value->type()->tag()) {
 975           existing_state->invalidate_local(index);
 976           TRACE_PHI(tty->print_cr("invalidating local %d because of type mismatch", index));
 977         } else if (new_value != existing_value && (existing_phi == NULL || existing_phi->block() != this)) {
 978           existing_state->setup_phi_for_local(this, index);
 979           TRACE_PHI(tty->print_cr("creating phi-function %c%d for local %d", existing_state->local_at(index)->type()->tchar(), existing_state->local_at(index)->id(), index));
 980         }
 981       }
 982     }
 983 
 984     assert(existing_state->caller_state() == new_state->caller_state(), "caller states must be equal");
 985 
 986   } else {
 987     assert(false, "stack or locks not matching (invalid bytecodes)");
 988     return false;
 989   }
 990 
 991   TRACE_PHI(tty->print_cr("********** try_merge for block B%d successful", block_id()));
 992 
 993   return true;
 994 }
