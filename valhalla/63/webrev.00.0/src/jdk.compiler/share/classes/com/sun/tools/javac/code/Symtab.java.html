 75 /** A class that defines all predefined constants and operators
 76  *  as well as special classes such as java.lang.Object, which need
 77  *  to be known to the compiler. All symbols are held in instance
 78  *  fields. This makes it possible to work in multiple concurrent
 79  *  projects, which might use different class files for library classes.
 80  *
 81  *  <p><b>This is NOT part of any supported API.
 82  *  If you write code that depends on this, you do so at your own risk.
 83  *  This code and its internal interfaces are subject to change or
 84  *  deletion without notice.</b>
 85  */

 90     /** Get the symbol table instance. */
 91     public static Symtab instance(Context context) {
 92         Symtab instance = context.get(symtabKey);
 93         if (instance == null)
 94             instance = new Symtab(context);
 95         return instance;
 96     }

265     public VarSymbol getClassField(Type type, Types types) {
266         return classFields.computeIfAbsent(
267             new UniqueType(type, types), k -> {
268                 Type arg = null;
269                 if (type.getTag() == ARRAY || type.getTag() == CLASS) {
270                     /* Temporary treatment for inline class: Given an inline class V that implements
271                        I1, I2, ... In, V.class is typed to be Class<? extends Object & I1 & I2 .. & In>
272                     */
273                     if (type.isValue()) {
274                         List<Type> bounds = List.of(objectType).appendList(((ClassSymbol) type.tsym).getInterfaces());
275                         arg = new WildcardType(types.makeIntersectionType(bounds), BoundKind.EXTENDS, boundClass);
276                     } else {
277                         arg = types.erasure(type);
278                     }
279                 }
280                 else if (type.isPrimitiveOrVoid())
281                     arg = types.boxedClass(type).type;
282                 else
283                     throw new AssertionError(type);
284 
285                 Type t = new ClassType(
286                     classType.getEnclosingType(), List.of(arg), classType.tsym);
287                 return new VarSymbol(
288                     STATIC | PUBLIC | FINAL, names._class, t, type.tsym);
289             });
290     }

320     public void synthesizeEmptyInterfaceIfMissing(final Type type) {
321         final Completer completer = type.tsym.completer;
322         type.tsym.completer = new Completer() {
323             @Override
324             public void complete(Symbol sym) throws CompletionFailure {
325                 try {
326                     completer.complete(sym);
327                 } catch (CompletionFailure e) {
328                     sym.flags_field |= (PUBLIC | INTERFACE);
329                     ((ClassType) sym.type).supertype_field = objectType;
330                 }
331             }
332 
333             @Override
334             public boolean isTerminal() {
335                 return completer.isTerminal();
336             }
337         };
338     }

340     public void synthesizeBoxTypeIfMissing(final Type type) {
341         ClassSymbol sym = enterClass(java_base, boxedName[type.getTag().ordinal()]);
342         final Completer completer = sym.completer;
343         sym.completer = new Completer() {
344             @Override
345             public void complete(Symbol sym) throws CompletionFailure {
346                 try {
347                     completer.complete(sym);
348                 } catch (CompletionFailure e) {
349                     sym.flags_field |= PUBLIC;
350                     ((ClassType) sym.type).supertype_field = objectType;
351                     MethodSymbol boxMethod =
352                         new MethodSymbol(PUBLIC | STATIC, names.valueOf,
353                                          new MethodType(List.of(type), sym.type,
354                                 List.nil(), methodClass),
355                             sym);
356                     sym.members().enter(boxMethod);
357                     MethodSymbol unboxMethod =
358                         new MethodSymbol(PUBLIC,
359                             type.tsym.name.append(names.Value), // x.intValue()
360                             new MethodType(List.nil(), type,
361                                 List.nil(), methodClass),
362                             sym);
363                     sym.members().enter(unboxMethod);
364                 }
365             }
366 
367             @Override
368             public boolean isTerminal() {
369                 return completer.isTerminal();
370             }
371         };
372     }

374     // Enter a synthetic class that is used to mark classes in ct.sym.
375     // This class does not have a class file.
376     private Type enterSyntheticAnnotation(String name) {
377         // for now, leave the module null, to prevent problems from synthesizing the
378         // existence of a class in any specific module, including noModule
379         ClassType type = (ClassType)enterClass(java_base, names.fromString(name)).type;
380         ClassSymbol sym = (ClassSymbol)type.tsym;
381         sym.completer = Completer.NULL_COMPLETER;
382         sym.flags_field = PUBLIC|ACYCLIC|ANNOTATION|INTERFACE;
383         sym.erasure_field = type;
384         sym.members_field = WriteableScope.create(sym);
385         type.typarams_field = List.nil();
386         type.allparams_field = List.nil();
387         type.supertype_field = annotationType;
388         type.interfaces_field = List.nil();
389         return type;
390     }

399

404 
405         MissingInfoHandler missingInfoHandler = MissingInfoHandler.instance(context);
406 
407         Target target = Target.instance(context);
408         rootPackage = new RootPackageSymbol(names.empty, null,
409                                             missingInfoHandler,

412         // create the basic builtin symbols

414                 {
415                     directives = List.nil();

417                     provides = List.nil();
418                     uses = List.nil();

420                     com.sun.tools.javac.code.Directive.RequiresDirective d =
421                             new com.sun.tools.javac.code.Directive.RequiresDirective(java_base,
422                                     EnumSet.of(com.sun.tools.javac.code.Directive.RequiresFlag.MANDATED));
423                     requires = List.of(d);
424                 }
425                 @Override
426                 public String toString() {
427                     return messages.getLocalizedString("compiler.misc.unnamed.module");
428                 }
429             };
430         addRootPackageFor(unnamedModule);
431         unnamedModule.enclosedPackages = unnamedModule.enclosedPackages.prepend(unnamedModule.unnamedPackage);
432 
433         errModule = new ModuleSymbol(names.empty, null) {

461 
462         // create the error symbols
463         errSymbol = new ClassSymbol(PUBLIC|STATIC|ACYCLIC, names.any, null, rootPackage);

472         initType(shortType, "short", "Short");
473         initType(charType, "char", "Character");
474         initType(intType, "int", "Integer");
475         initType(longType, "long", "Long");
476         initType(floatType, "float", "Float");
477         initType(doubleType, "double", "Double");
478         initType(booleanType, "boolean", "Boolean");
479         initType(voidType, "void", "Void");
480         initType(botType, "<nulltype>");
481         initType(errType, errSymbol);
482         initType(unknownType, unknownSymbol);
483 
484         // the builtin class of all arrays
485         arrayClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Array, noSymbol);
486 
487         // VGJ
488         boundClass = new ClassSymbol(PUBLIC|ACYCLIC, names.Bound, noSymbol);
489         boundClass.members_field = new Scope.ErrorScope(boundClass);
490

617         proprietaryType = enterSyntheticAnnotation("sun.Proprietary+Annotation");
618 
619         // Enter a synthetic class that is used to provide profile info for
620         // classes in ct.sym.  This class does not have a class file.
621         profileType = enterSyntheticAnnotation("jdk.Profile+Annotation");
622         MethodSymbol m = new MethodSymbol(PUBLIC | ABSTRACT, names.value, intType, profileType.tsym);
623         profileType.tsym.members().enter(m);
624 
625         // Enter a class for arrays.
626         // The class implements java.lang.Cloneable and java.io.Serializable.
627         // It has a final length field and a clone method.
628         ClassType arrayClassType = (ClassType)arrayClass.type;
629         arrayClassType.supertype_field = objectType;
630         arrayClassType.interfaces_field =
631                 List.of(cloneableType, serializableType, identityObjectType);
632 
633         arrayClass.members_field = WriteableScope.create(arrayClass);
634         lengthVar = new VarSymbol(
635             PUBLIC | FINAL,
636             names.length,
637             intType,
638             arrayClass);
639         arrayClass.members().enter(lengthVar);
640         arrayCloneMethod = new MethodSymbol(
641             PUBLIC,
642             names.clone,
643             new MethodType(List.nil(), objectType,
644                            List.nil(), methodClass),
645             arrayClass);
646         arrayClass.members().enter(arrayCloneMethod);
647 
648         if (java_base != noModule)
649             java_base.completer = moduleCompleter::complete; //bootstrap issues
650 
651     }

661     /** Create a new toplevel or member class symbol with given name
662      *  and owner and enter in `classes' unless already there.
663      */
664     public ClassSymbol enterClass(ModuleSymbol msym, Name name, TypeSymbol owner) {
665         Assert.checkNonNull(msym);
666         Name flatname = TypeSymbol.formFlatName(name, owner);
667         ClassSymbol c = getClass(msym, flatname);
668         if (c == null) {
669             c = defineClass(name, owner);
670             doEnterClass(msym, c);
671         } else if ((c.name != name || c.owner != owner) && owner.kind == TYP && c.owner.kind == PCK) {
672             // reassign fields of classes that might have been loaded with
673             // their flat names.
674             c.owner.members().remove(c);
675             c.name = name;
676             c.owner = owner;
677             c.fullname = ClassSymbol.formFullName(name, owner);
678         }
679         return c;
680     }

687     public PackageSymbol lookupPackage(ModuleSymbol msym, Name flatName) {
688         Assert.checkNonNull(msym);
689 
690         if (flatName.isEmpty()) {
691             //unnamed packages only from the current module - visiblePackages contains *root* package, not unnamed package!
692             return msym.unnamedPackage;
693         }
694 
695         if (msym == noModule) {
696             return enterPackage(msym, flatName);
697         }
698 
699         msym.complete();
700 
701         PackageSymbol pack;
702 
703         pack = msym.visiblePackages.get(flatName);
704 
705         if (pack != null)
706             return pack;
707 
708         pack = getPackage(msym, flatName);
709 
710         if (pack != null && pack.exists())
711             return pack;
712 
713         boolean dependsOnUnnamed = msym.requires != null &&
714                                    msym.requires.stream()
715                                                 .map(rd -> rd.module)
716                                                 .anyMatch(mod -> mod == unnamedModule);
717 
718         if (dependsOnUnnamed) {
719             //msyms depends on the unnamed module, for which we generally don't know
720             //the list of packages it "exports" ahead of time. So try to lookup the package in the
721             //current module, and in the unnamed module and see if it exists in one of them
722             PackageSymbol unnamedPack = getPackage(unnamedModule, flatName);
723 
724             if (unnamedPack != null && unnamedPack.exists()) {
725                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
726                 return unnamedPack;
727             }
728 
729             pack = enterPackage(msym, flatName);
730             pack.complete();
731             if (pack.exists())
732                 return pack;
733 
734             unnamedPack = enterPackage(unnamedModule, flatName);
735             unnamedPack.complete();
736             if (unnamedPack.exists()) {
737                 msym.visiblePackages.put(unnamedPack.fullname, unnamedPack);
738                 return unnamedPack;
739             }
740 
741             return pack;
742         }
743 
744         return enterPackage(msym, flatName);
745     }

764     public ClassSymbol enterClass(ModuleSymbol msym, Name flatname) {

785     /** Make a package, given its fully qualified name.
786      */
787     public PackageSymbol enterPackage(ModuleSymbol currModule, Name fullname) {
788         Assert.checkNonNull(currModule);
789         PackageSymbol p = getPackage(currModule, fullname);
790         if (p == null) {
791             Assert.check(!fullname.isEmpty(), () -> "rootPackage missing!; currModule: " + currModule);
792             p = new PackageSymbol(
793                     Convert.shortName(fullname),
794                     enterPackage(currModule, Convert.packagePart(fullname)));
795             p.completer = initialCompleter;
796             p.modle = currModule;
797             doEnterPackage(currModule, p);
798         }
799         return p;
800     }

803         packages.computeIfAbsent(pack.fullname, n -> new HashMap<>()).put(msym, pack);
804         msym.enclosedPackages = msym.enclosedPackages.prepend(pack);

807     private void addRootPackageFor(ModuleSymbol module) {
808         doEnterPackage(module, rootPackage);
809         PackageSymbol unnamedPackage = new PackageSymbol(names.empty, rootPackage) {
810                 @Override

812                     return messages.getLocalizedString("compiler.misc.unnamed.package");
813                 }
814             };
815         unnamedPackage.modle = module;
816         //we cannot use a method reference below, as initialCompleter might be null now
817         unnamedPackage.completer = s -> initialCompleter.complete(s);
818         unnamedPackage.flags_field |= EXISTS;
819         module.unnamedPackage = unnamedPackage;
