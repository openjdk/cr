24 /**
25  * @test
26  * @bug 8224162
27  * @summary Profile counter for a call site may overflow.
28  * @run main/othervm -Xbatch -XX:-UseOnStackReplacement -XX:MaxTrivialSize=0 compiler.profiling.TestProfileCounterOverflow
29  */
30 
31 package compiler.profiling;
32 
33 public class TestProfileCounterOverflow {
34     public static void test(long iterations) {
35         for (long j = 0; j < iterations; j++) {
36             call();
37         }
38     }
39 
40     public static void call() {}
41 
42     public static void main(String[] args) {
43         // trigger profiling on tier3
44         for (int i = 0; i < 500; i++) {
45             test(1);
46         }
47 
48         test(Integer.MAX_VALUE + 10000L); // overflow call counter
49 
50         // trigger c2 compilation
51         for (int i = 0; i < 10_000; i++) {
52             test(1);
53         }
54         System.out.println("TEST PASSED");
55     }
56 }
