 793                                                  srcRect.getIntY(),
 794                                                  (int)Math.ceil(srcRect.getWidth()),
 795                                                  (int)Math.ceil(srcRect.getHeight()));
 796                     }






 797                     g.setPaint(new ImagePattern(
 798                                img,
 799                                adjustedX, adjustedY,
 800                                scaledTileWidth, scaledTileHeight,
 801                                false, false));
 802
