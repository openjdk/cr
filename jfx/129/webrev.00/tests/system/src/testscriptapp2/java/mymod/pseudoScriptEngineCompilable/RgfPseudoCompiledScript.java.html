26 package pseudoScriptEngineCompilable;
27 
28 import javax.script.Bindings;
29 import javax.script.CompiledScript;
30 import javax.script.ScriptContext;
31 import javax.script.ScriptEngine;
32 import javax.script.ScriptException;
33 
34 public class RgfPseudoCompiledScript extends CompiledScript
35 {
36     String code = null;
37     ScriptEngine scriptEngine = null;
38 
39     RgfPseudoCompiledScript(String code, ScriptEngine scriptEngine) {
40         this.code = code;
41         this.scriptEngine = scriptEngine;
42     }
43 
44     public Object eval(Bindings bindings) throws ScriptException {
45         return scriptEngine.eval("RgfPseudoCompiledScript.eval(Bindings bindings): " + code, bindings);
46     }
47 
48     public Object eval(ScriptContext context) throws ScriptException {
49         return scriptEngine.eval("RgfPseudoCompiledScript.eval(ScriptContext context): " + code, context);
50     }
51 
52     public Object eval() throws ScriptException {
53             return scriptEngine.eval("RgfPseudoCompiledScript.eval(): " + code );
54     }
55 
56     public ScriptEngine getEngine() {
57         return scriptEngine;
58     }
59 }
