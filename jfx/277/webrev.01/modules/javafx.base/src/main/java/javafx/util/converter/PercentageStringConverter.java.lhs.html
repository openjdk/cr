26 package javafx.util.converter;
27 
28 import java.text.NumberFormat;
29 import java.util.Locale;
30 import javafx.util.StringConverter;
31 
32 /**
33  * <p>{@link StringConverter} implementation for {@link Number} values
34  * that represent percentages.</p>
35  *
36  * @see CurrencyStringConverter
37  * @see NumberStringConverter
38  * @see StringConverter
39  * @since JavaFX 2.1
40  */
41 public class PercentageStringConverter extends NumberStringConverter {
42 
43 
44     // ------------------------------------------------------------ Constructors

45     public PercentageStringConverter() {
46         this(Locale.getDefault());
47     }
48 



49     public PercentageStringConverter(Locale locale) {
50         super(locale, null, null);
51     }
52 



53     public PercentageStringConverter(NumberFormat numberFormat) {
54         super(null, null, numberFormat);
55     }
56 
57     // ----------------------------------------------------------------- Methods
58 
59     /** {@inheritDoc} */
60     @Override public NumberFormat getNumberFormat() {
61         Locale _locale = locale == null ? Locale.getDefault() : locale;
62 
63         if (numberFormat != null) {
64             return numberFormat;
65         } else {
66             return NumberFormat.getPercentInstance(_locale);
67         }
68     }
69 }
