 171 char* Method::name_and_sig_as_C_string(Klass* klass, Symbol* method_name, Symbol* signature) {
 172   const char* klass_name = klass->external_name();
 173   int klass_name_len  = (int)strlen(klass_name);
 174   int method_name_len = method_name->utf8_length();
 175   int len             = klass_name_len + 1 + method_name_len + signature->utf8_length();
 176   char* dest          = NEW_RESOURCE_ARRAY(char, len + 1);
 177   strcpy(dest, klass_name);
 178   dest[klass_name_len] = '.';
 179   strcpy(&dest[klass_name_len + 1], method_name->as_C_string());
 180   strcpy(&dest[klass_name_len + 1 + method_name_len], signature->as_C_string());
 181   dest[len] = 0;
 182   return dest;
 183 }

 223 int Method::fast_exception_handler_bci_for(const methodHandle& mh, Klass* ex_klass, int throw_bci, TRAPS) {
 224   // exception table holds quadruple entries of the form (beg_bci, end_bci, handler_bci, klass_index)
 225   // access exception table
 226   ExceptionTable table(mh());
 227   int length = table.length();
 228   // iterate through all entries sequentially
 229   constantPoolHandle pool(THREAD, mh->constants());
 230   for (int i = 0; i < length; i ++) {
 231     //reacquire the table in case a GC happened
 232     ExceptionTable table(mh());
 233     int beg_bci = table.start_pc(i);
 234     int end_bci = table.end_pc(i);
 235     assert(beg_bci <= end_bci, "inconsistent exception table");
 236     if (beg_bci <= throw_bci && throw_bci < end_bci) {
 237       // exception handler bci range covers throw_bci => investigate further
 238       int handler_bci = table.handler_pc(i);
 239       int klass_index = table.catch_type_index(i);
 240       if (klass_index == 0) {
 241         return handler_bci;
 242       } else if (ex_klass == NULL) {
 243         return handler_bci;
 244       } else {
 245         // we know the exception class => get the constraint class
 246         // this may require loading of the constraint class; if verification
 247         // fails or some other exception occurs, return handler_bci
 248         Klass* k = pool->klass_at(klass_index, CHECK_(handler_bci));
 249         assert(k != NULL, "klass not loaded");
 250         if (ex_klass->is_subtype_of(k)) {
 251           return handler_bci;
 252         }
 253       }
 254     }
 255   }
 256 
 257   return -1;
 258 }

 260 void Method::mask_for(int bci, InterpreterOopMap* mask) {
 261   methodHandle h_this(Thread::current(), this);
 262   // Only GC uses the OopMapCache during thread stack root scanning
 263   // any other uses generate an oopmap but do not save it in the cache.
 264   if (Universe::heap()->is_gc_active()) {
 265     method_holder()->mask_for(h_this, bci, mask);
 266   } else {
 267     OopMapCache::compute_one_oop_map(h_this, bci, mask);
 268   }
 269   return;

 293 // Return bci if it appears to be a valid bcp
 294 // Return -1 otherwise.
 295 // Used by profiling code, when invalid data is a possibility.
 296 // The caller is responsible for validating the Method* itself.
 297 int Method::validate_bci_from_bcp(address bcp) const {
 298   // keep bci as -1 if not a valid bci
 299   int bci = -1;
 300   if (bcp == 0 || bcp == code_base()) {
 301     // code_size() may return 0 and we allow 0 here
 302     // the method may be native
 303     bci = 0;
 304   } else if (contains(bcp)) {
 305     bci = bcp - code_base();
 306   }
 307   // Assert that if we have dodged any asserts, bci is negative.
 308   assert(bci == -1 || bci == bci_from(bcp_from(bci)), "sane bci if >=0");
 309   return bci;
 310 }

 387 // The RegisterNatives call being attempted tried to register with a method that
 388 // is not native.  Ask JVM TI what prefixes have been specified.  Then check
 389 // to see if the native method is now wrapped with the prefixes.  See the
 390 // SetNativeMethodPrefix(es) functions in the JVM TI Spec for details.
 391 static Method* find_prefixed_native(Klass* k, Symbol* name, Symbol* signature, TRAPS) {
 392 #if INCLUDE_JVMTI
 393   ResourceMark rm(THREAD);
 394   Method* method;
 395   int name_len = name->utf8_length();
 396   char* name_str = name->as_utf8();
 397   int prefix_count;
 398   char** prefixes = JvmtiExport::get_all_native_method_prefixes(&prefix_count);
 399   for (int i = 0; i < prefix_count; i++) {
 400     char* prefix = prefixes[i];
 401     int prefix_len = (int)strlen(prefix);
 402 
 403     // try adding this prefix to the method name and see if it matches another method name
 404     int trial_len = name_len + prefix_len;
 405     char* trial_name_str = NEW_RESOURCE_ARRAY(char, trial_len + 1);
 406     strcpy(trial_name_str, prefix);
 407     strcat(trial_name_str, name_str);
 408     TempNewSymbol trial_name = SymbolTable::probe(trial_name_str, trial_len);
 409     if (trial_name == NULL) {
 410       continue; // no such symbol, so this prefix wasn't used, try the next prefix
 411     }
 412     method = k->lookup_method(trial_name, signature);
 413     if (method == NULL) {
 414       continue; // signature doesn't match, try the next prefix
 415     }
 416     if (method->is_native()) {
 417       method->set_is_prefixed_native();
 418       return method; // wahoo, we found a prefixed version of the method, return it
 419     }
 420     // found as non-native, so prefix is good, add it, probably just need more prefixes
 421     name_len = trial_len;
 422     name_str = trial_name_str;
 423   }
 424 #endif // INCLUDE_JVMTI
 425   return NULL; // not found
 426 }

 428 bool Method::register_native(Klass* k, Symbol* name, Symbol* signature, address entry, TRAPS) {
 429   Method* method = k->lookup_method(name, signature);
 430   if (method == NULL) {
 431     ResourceMark rm(THREAD);
 432     stringStream st;
 433     st.print("Method '");
 434     print_external_name(&st, k, name, signature);
 435     st.print("' name or signature does not match");
 436     THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 437   }
 438   if (!method->is_native()) {
 439     // trying to register to a non-native method, see if a JVM TI agent has added prefix(es)
 440     method = find_prefixed_native(k, name, signature, THREAD);
 441     if (method == NULL) {
 442       ResourceMark rm(THREAD);
 443       stringStream st;
 444       st.print("Method '");
 445       print_external_name(&st, k, name, signature);
 446       st.print("' is not declared as native");
 447       THROW_MSG_(vmSymbols::java_lang_NoSuchMethodError(), st.as_string(), false);
 448     }
 449   }
 450 
 451   if (entry != NULL) {
 452     method->set_native_function(entry, native_bind_event_is_interesting);
 453   } else {
 454     method->clear_native_function();
 455   }
 456   if (log_is_enabled(Debug, jni, resolve)) {
 457     ResourceMark rm(THREAD);
 458     log_debug(jni, resolve)("[Registering JNI native method %s.%s]",
 459                             method->method_holder()->external_name(),
 460                             method->name()->as_C_string());
 461   }
 462   return true;
 463 }

 465 bool Method::was_executed_more_than(int n) {
 466   // Invocation counter is reset when the Method* is compiled.
 467   // If the method has compiled code we therefore assume it has
 468   // be excuted more than n times.
 469   if (is_accessor() || is_empty_method() || (code() != NULL)) {
 470     // interpreter doesn't bump invocation counter of trivial methods
 471     // compiler does not bump invocation counter of compiled methods
 472     return true;
 473   }
 474   else if ((method_counters() != NULL &&
 475             method_counters()->invocation_counter()->carry()) ||
 476            (method_data() != NULL &&
 477             method_data()->invocation_counter()->carry())) {
 478     // The carry bit is set when the counter overflows and causes
 479     // a compilation to occur.  We don't know how many times
 480     // the counter has been reset, so we simply assume it has
 481     // been executed more than n times.
 482     return true;
 483   } else {
 484     return invocation_count() > n;
 485   }
 486 }

 490   if (is_final()) tty->print("final ");
 491   if (is_synchronized()) tty->print("synchronized ");

 509   }

 513 // Build a MethodData* object to hold information about this method
 514 // collected in the interpreter.
 515 void Method::build_interpreter_method_data(const methodHandle& method, TRAPS) {
 516   // Do not profile the method if metaspace has hit an OOM previously
 517   // allocating profiling data. Callers clear pending exception so don't
 518   // add one here.
 519   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 520     return;
 521   }
 522 
 523   // Grab a lock here to prevent multiple
 524   // MethodData*s from being created.
 525   MutexLocker ml(THREAD, MethodData_lock);
 526   if (method->method_data() == NULL) {
 527     ClassLoaderData* loader_data = method->method_holder()->class_loader_data();
 528     MethodData* method_data = MethodData::allocate(loader_data, method, THREAD);
 529     if (HAS_PENDING_EXCEPTION) {
 530       CompileBroker::log_metaspace_failure();
 531       ClassLoaderDataGraph::set_metaspace_oom(true);
 532       return;   // return the exception (which is cleared)
 533     }
 534 
 535     method->set_method_data(method_data);
 536     if (PrintMethodData && (Verbose || WizardMode)) {
 537       ResourceMark rm(THREAD);
 538       tty->print("build_interpreter_method_data for ");
 539       method->print_name(tty);
 540       tty->cr();
 541       // At the end of the run, the MDO, full of data, will be dumped.
 542     }
 543   }
 544 }

 546 MethodCounters* Method::build_method_counters(Method* m, TRAPS) {
 547   // Do not profile the method if metaspace has hit an OOM previously
 548   if (ClassLoaderDataGraph::has_metaspace_oom()) {
 549     return NULL;
 550   }
 551 
 552   methodHandle mh(THREAD, m);
 553   MethodCounters* counters = MethodCounters::allocate(mh, THREAD);
 554   if (HAS_PENDING_EXCEPTION) {
 555     CompileBroker::log_metaspace_failure();
 556     ClassLoaderDataGraph::set_metaspace_oom(true);
 557     return NULL;   // return the exception (which is cleared)
 558   }
 559   if (!mh->init_method_counters(counters)) {
 560     MetadataFactory::free_metadata(mh->method_holder()->class_loader_data(), counters);
 561   }
 562 
 563   if (LogTouchedMethods) {
 564     mh->log_touched(CHECK_NULL);
 565   }
 566 
 567   return mh->method_counters();
 568 }

 597 bool Method::is_vanilla_constructor() const {
 598   // Returns true if this method is a vanilla constructor, i.e. an "<init>" "()V" method
 599   // which only calls the superclass vanilla constructor and possibly does stores of
 600   // zero constants to local fields:
 601   //
 602   //   aload_0
 603   //   invokespecial
 604   //   indexbyte1
 605   //   indexbyte2
 606   //
 607   // followed by an (optional) sequence of:
 608   //
 609   //   aload_0
 610   //   aconst_null / iconst_0 / fconst_0 / dconst_0
 611   //   putfield
 612   //   indexbyte1
 613   //   indexbyte2
 614   //
 615   // followed by:
 616   //
 617   //   return
 618 
 619   assert(name() == vmSymbols::object_initializer_name(),    "Should only be called for default constructors");
 620   assert(signature() == vmSymbols::void_method_signature(), "Should only be called for default constructors");
 621   int size = code_size();
 622   // Check if size match
 623   if (size == 0 || size % 5 != 0) return false;
 624   address cb = code_base();
 625   int last = size - 1;
 626   if (cb[0] != Bytecodes::_aload_0 || cb[1] != Bytecodes::_invokespecial || cb[last] != Bytecodes::_return) {
 627     // Does not call superclass default constructor
 628     return false;
 629   }
 630   // Check optional sequence
 631   for (int i = 4; i < last; i += 5) {
 632     if (cb[i] != Bytecodes::_aload_0) return false;
 633     if (!Bytecodes::is_zero_const(Bytecodes::cast(cb[i+1]))) return false;
 634     if (cb[i+2] != Bytecodes::_putfield) return false;
 635   }
 636   return true;
 637 }

 640 bool Method::compute_has_loops_flag() {
 641   BytecodeStream bcs(methodHandle(Thread::current(), this));
 642   Bytecodes::Code bc;
 643 
 644   while ((bc = bcs.next()) >= 0) {
 645     switch( bc ) {
 646       case Bytecodes::_ifeq:
 647       case Bytecodes::_ifnull:
 648       case Bytecodes::_iflt:
 649       case Bytecodes::_ifle:
 650       case Bytecodes::_ifne:
 651       case Bytecodes::_ifnonnull:
 652       case Bytecodes::_ifgt:
 653       case Bytecodes::_ifge:
 654       case Bytecodes::_if_icmpeq:
 655       case Bytecodes::_if_icmpne:
 656       case Bytecodes::_if_icmplt:
 657       case Bytecodes::_if_icmpgt:
 658       case Bytecodes::_if_icmple:
 659       case Bytecodes::_if_icmpge:
 660       case Bytecodes::_if_acmpeq:
 661       case Bytecodes::_if_acmpne:
 662       case Bytecodes::_goto:
 663       case Bytecodes::_jsr:
 664         if( bcs.dest() < bcs.next_bci() ) _access_flags.set_has_loops();
 665         break;
 666 
 667       case Bytecodes::_goto_w:
 668       case Bytecodes::_jsr_w:
 669         if( bcs.dest_w() < bcs.next_bci() ) _access_flags.set_has_loops();
 670         break;
 671 
 672       default:
 673         break;
 674     }
 675   }
 676   _access_flags.set_loops_flag_init();
 677   return _access_flags.has_loops();
 678 }

 803 objArrayHandle Method::resolved_checked_exceptions_impl(Method* method, TRAPS) {
 804   int length = method->checked_exceptions_length();
 805   if (length == 0) {  // common case
 806     return objArrayHandle(THREAD, Universe::the_empty_class_klass_array());
 807   } else {
 808     methodHandle h_this(THREAD, method);

 810     objArrayHandle mirrors (THREAD, m_oop);
 811     for (int i = 0; i < length; i++) {

 813       Klass* k = h_this->constants()->klass_at(table[i].class_cp_index, CHECK_(objArrayHandle()));

 815           !k->is_subclass_of(SystemDictionary::Throwable_klass())) {

 818           "Class %s in throws clause of method %s is not a subtype of class java.lang.Throwable",
 819           k->external_name(), method->external_name());

 823     return mirrors;
 824   }

 828 int Method::line_number_from_bci(int bci) const {
 829   int best_bci  =  0;
 830   int best_line = -1;
 831   if (bci == SynchronizationEntryBCI) bci = 0;
 832   if (0 <= bci && bci < code_size() && has_linenumber_table()) {
 833     // The line numbers are a short array of 2-tuples [start_pc, line_number].
 834     // Not necessarily sorted and not necessarily one-to-one.
 835     CompressedLineNumberReadStream stream(compressed_linenumber_table());
 836     while (stream.read_pair()) {
 837       if (stream.bci() == bci) {
 838         // perfect match
 839         return stream.line();
 840       } else {
 841         // update best_bci/line
 842         if (stream.bci() < bci && stream.bci() >= best_bci) {
 843           best_bci  = stream.bci();
 844           best_line = stream.line();
 845         }
 846       }
 847     }
 848   }
 849   return best_line;
 850 }

 876 void Method::set_native_function(address function, bool post_event_flag) {
 877   assert(function != NULL, "use clear_native_function to unregister natives");
 878   assert(!is_method_handle_intrinsic() || function == SharedRuntime::native_method_throw_unsatisfied_link_error_entry(), "");
 879   address* native_function = native_function_addr();
 880 
 881   // We can see racers trying to place the same native function into place. Once
 882   // is plenty.
 883   address current = *native_function;
 884   if (current == function) return;
 885   if (post_event_flag && JvmtiExport::should_post_native_method_bind() &&
 886       function != NULL) {
 887     // native_method_throw_unsatisfied_link_error_entry() should only
 888     // be passed when post_event_flag is false.
 889     assert(function !=
 890       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
 891       "post_event_flag mis-match");
 892 
 893     // post the bind event, and possible change the bind function
 894     JvmtiExport::post_native_method_bind(this, &function);
 895   }
 896   *native_function = function;
 897   // This function can be called more than once. We must make sure that we always
 898   // use the latest registered method -> check if a stub already has been generated.
 899   // If so, we have to make it not_entrant.
 900   CompiledMethod* nm = code(); // Put it into local variable to guard against concurrent updates
 901   if (nm != NULL) {
 902     nm->make_not_entrant();
 903   }
 904 }

1101 /****************************************************************************
1102 // The following illustrates how the entries work for CDS shared Methods:
1103 //
1104 // Our goal is to delay writing into a shared Method until it's compiled.
1105 // Hence, we want to determine the initial values for _i2i_entry,
1106 // _from_interpreted_entry and _from_compiled_entry during CDS dump time.
1107 //
1108 // In this example, both Methods A and B have the _i2i_entry of "zero_locals".
1109 // They also have similar signatures so that they will share the same
1110 // AdapterHandlerEntry.
1111 //
1112 // _adapter_trampoline points to a fixed location in the RW section of
1113 // the CDS archive. This location initially contains a NULL pointer. When the
1114 // first of method A or B is linked, an AdapterHandlerEntry is allocated
1115 // dynamically, and its c2i/i2c entries are generated.
1116 //
1117 // _i2i_entry and _from_interpreted_entry initially points to the same
1118 // (fixed) location in the CODE section of the CDS archive. This contains
1119 // an unconditional branch to the actual entry for "zero_locals", which is
1120 // generated at run time and may be on an arbitrary address. Thus, the
1121 // unconditional branch is also generated at run time to jump to the correct
1122 // address.
1123 //
1124 // Similarly, _from_compiled_entry points to a fixed address in the CODE
1125 // section. This address has enough space for an unconditional branch
1126 // instruction, and is initially zero-filled. After the AdapterHandlerEntry is
1127 // initialized, and the address for the actual c2i_entry is known, we emit a
1128 // branch instruction here to branch to the actual c2i_entry.
1129 //
1130 // The effect of the extra branch on the i2i and c2i entries is negligible.
1131 //
1132 // The reason for putting _adapter_trampoline in RO is many shared Methods
1133 // share the same AdapterHandlerEntry, so we can save space in the RW section
1134 // by having the extra indirection.
1135 
1136 
1137 [Method A: RW]
1138   _constMethod ----> [ConstMethod: RO]
1139                        _adapter_trampoline -----------+
1140                                                       |
1141   _i2i_entry              (same value as method B)    |
1142   _from_interpreted_entry (same value as method B)    |
1143   _from_compiled_entry    (same value as method B)    |
1144                                                       |
1145                                                       |
1146 [Method B: RW]                               +--------+
1147   _constMethod ----> [ConstMethod: RO]       |
1148                        _adapter_trampoline --+--->(AdapterHandlerEntry* ptr: RW)-+
1149                                                                                  |
1150                                                  +-------------------------------+
1151                                                  |
1152                                                  +----> [AdapterHandlerEntry] (allocated at run time)
1153                                                               _fingerprint
1154                                                               _c2i_entry ---------------------------------+->[c2i entry..]
1155  _i2i_entry  -------------+                                   _i2c_entry ---------------+-> [i2c entry..] |
1156  _from_interpreted_entry  |                                   _c2i_unverified_entry     |                 |
1157          |                |                                   _c2i_no_clinit_check_entry|                 |
1158          |                |  (_cds_entry_table: CODE)                                   |                 |
1159          |                +->[0]: jmp _entry_table[0] --> (i2i_entry_for "zero_locals") |                 |

1172 // Called when the method_holder is getting linked. Setup entrypoints so the method
1173 // is ready to be called from interpreter, compiler, and vtables.
1174 void Method::link_method(const methodHandle& h_method, TRAPS) {
1175   // If the code cache is full, we may reenter this function for the
1176   // leftover methods that weren't linked.
1177   if (is_shared()) {
1178 #ifdef ASSERT
1179     address entry = Interpreter::entry_for_cds_method(h_method);
1180     assert(entry != NULL && entry == _i2i_entry,
1181            "should be correctly set during dump time");
1182 #endif
1183     if (adapter() != NULL) {
1184       return;
1185     }
1186     assert(entry == _from_interpreted_entry,
1187            "should be correctly set during dump time");
1188   } else if (_i2i_entry != NULL) {
1189     return;
1190   }
1191   assert( _code == NULL, "nothing compiled yet" );
1192 
1193   // Setup interpreter entrypoint
1194   assert(this == h_method(), "wrong h_method()" );
1195 
1196   if (!is_shared()) {
1197     assert(adapter() == NULL, "init'd to NULL");
1198     address entry = Interpreter::entry_for_method(h_method);
1199     assert(entry != NULL, "interpreter entry must be non-null");
1200     // Sets both _i2i_entry and _from_interpreted_entry
1201     set_interpreter_entry(entry);
1202   }
1203 
1204   // Don't overwrite already registered native entries.
1205   if (is_native() && !has_native_function()) {
1206     set_native_function(
1207       SharedRuntime::native_method_throw_unsatisfied_link_error_entry(),
1208       !native_bind_event_is_interesting);
1209   }
1210 
1211   // Setup compiler entrypoint.  This is made eagerly, so we do not need
1212   // special handling of vtables.  An alternative is to make adapters more
1213   // lazily by calling make_adapter() from from_compiled_entry() for the
1214   // normal calls.  For vtable calls life gets more complicated.  When a
1215   // call-site goes mega-morphic we need adapters in all methods which can be
1216   // called from the vtable.  We need adapters on such methods that get loaded
1217   // later.  Ditto for mega-morphic itable calls.  If this proves to be a
1218   // problem we'll make these lazily later.
1219   (void) make_adapters(h_method, CHECK);
1220 
1221   // ONLY USE the h_method now as make_adapter may have blocked
1222 
1223 }

1225 address Method::make_adapters(const methodHandle& mh, TRAPS) {
1226   // Adapters for compiled code are made eagerly here.  They are fairly
1227   // small (generally < 100 bytes) and quick to make (and cached and shared)
1228   // so making them eagerly shouldn't be too expensive.
1229   AdapterHandlerEntry* adapter = AdapterHandlerLibrary::get_adapter(mh);
1230   if (adapter == NULL ) {
1231     if (!is_init_completed()) {
1232       // Don't throw exceptions during VM initialization because java.lang.* classes
1233       // might not have been initialized, causing problems when constructing the
1234       // Java exception object.
1235       vm_exit_during_initialization("Out of space in CodeCache for adapters");
1236     } else {
1237       THROW_MSG_NULL(vmSymbols::java_lang_VirtualMachineError(), "Out of space in CodeCache for adapters");
1238     }
1239   }
1240 
1241   if (mh->is_shared()) {
1242     assert(mh->adapter() == adapter, "must be");
1243     assert(mh->_from_compiled_entry != NULL, "must be");
1244   } else {
1245     mh->set_adapter_entry(adapter);
1246     mh->_from_compiled_entry = adapter->get_c2i_entry();
1247   }
1248   return adapter->get_c2i_entry();
1249 }

1293 // Install compiled code.  Instantly it can execute.
1294 void Method::set_code(const methodHandle& mh, CompiledMethod *code) {
1295   assert_lock_strong(CompiledMethod_lock);
1296   assert( code, "use clear_code to remove code" );
1297   assert( mh->check_code(), "" );
1298 
1299   guarantee(mh->adapter() != NULL, "Adapter blob must already exist!");
1300 
1301   // These writes must happen in this order, because the interpreter will
1302   // directly jump to from_interpreted_entry which jumps to an i2c adapter
1303   // which jumps to _from_compiled_entry.
1304   mh->_code = code;             // Assign before allowing compiled code to exec
1305 
1306   int comp_level = code->comp_level();
1307   // In theory there could be a race here. In practice it is unlikely
1308   // and not worth worrying about.
1309   if (comp_level > mh->highest_comp_level()) {
1310     mh->set_highest_comp_level(comp_level);
1311   }
1312 
1313   OrderAccess::storestore();
1314   mh->_from_compiled_entry = code->verified_entry_point();
1315   OrderAccess::storestore();
1316   // Instantly compiled code can execute.
1317   if (!mh->is_method_handle_intrinsic())
1318     mh->_from_interpreted_entry = mh->get_i2c_entry();
1319 }

1410 // Make an instance of a signature-polymorphic internal MH primitive.
1411 methodHandle Method::make_method_handle_intrinsic(vmIntrinsics::ID iid,
1412                                                          Symbol* signature,
1413                                                          TRAPS) {
1414   ResourceMark rm(THREAD);
1415   methodHandle empty;
1416 
1417   InstanceKlass* holder = SystemDictionary::MethodHandle_klass();
1418   Symbol* name = MethodHandles::signature_polymorphic_intrinsic_name(iid);
1419   assert(iid == MethodHandles::signature_polymorphic_name_id(name), "");
1420   if (TraceMethodHandles) {
1421     tty->print_cr("make_method_handle_intrinsic MH.%s%s", name->as_C_string(), signature->as_C_string());
1422   }
1423 
1424   // invariant:   cp->symbol_at_put is preceded by a refcount increment (more usually a lookup)
1425   name->increment_refcount();
1426   signature->increment_refcount();
1427 
1428   int cp_length = _imcp_limit;
1429   ClassLoaderData* loader_data = holder->class_loader_data();
1430   constantPoolHandle cp;
1431   {
1432     ConstantPool* cp_oop = ConstantPool::allocate(loader_data, cp_length, CHECK_(empty));
1433     cp = constantPoolHandle(THREAD, cp_oop);
1434   }
1435   cp->set_pool_holder(holder);
1436   cp->symbol_at_put(_imcp_invoke_name,       name);
1437   cp->symbol_at_put(_imcp_invoke_signature,  signature);
1438   cp->set_has_preresolution();
1439 
1440   // decide on access bits:  public or not?
1441   int flags_bits = (JVM_ACC_NATIVE | JVM_ACC_SYNTHETIC | JVM_ACC_FINAL);
1442   bool must_be_static = MethodHandles::is_signature_polymorphic_static(iid);
1443   if (must_be_static)  flags_bits |= JVM_ACC_STATIC;
1444   assert((flags_bits & JVM_ACC_PUBLIC) == 0, "do not expose these methods");
1445 
1446   methodHandle m;
1447   {
1448     InlineTableSizes sizes;
1449     Method* m_oop = Method::allocate(loader_data, 0,
1450                                      accessFlags_from(flags_bits), &sizes,
1451                                      ConstMethod::NORMAL, CHECK_(empty));
1452     m = methodHandle(THREAD, m_oop);
1453   }
1454   m->set_constants(cp());
1455   m->set_name_index(_imcp_invoke_name);
1456   m->set_signature_index(_imcp_invoke_signature);
1457   assert(MethodHandles::is_signature_polymorphic_name(m->name()), "");
1458   assert(m->signature() == signature, "");
1459   m->compute_from_signature(signature);
1460   m->init_intrinsic_id();
1461   assert(m->is_method_handle_intrinsic(), "");
1462 #ifdef ASSERT
1463   if (!MethodHandles::is_signature_polymorphic(m->intrinsic_id()))  m->print();
1464   assert(MethodHandles::is_signature_polymorphic(m->intrinsic_id()), "must be an invoker");
1465   assert(m->intrinsic_id() == iid, "correctly predicted iid");
1466 #endif //ASSERT
1467 
1468   // Finally, set up its entry points.
1469   assert(m->can_be_statically_bound(), "");
1470   m->set_vtable_index(Method::nonvirtual_vtable_index);
1471   m->link_method(m, CHECK_(empty));
1472 
1473   if (TraceMethodHandles && (Verbose || WizardMode)) {
1474     ttyLocker ttyl;
1475     m->print_on(tty);
1476   }
1477 
1478   return m;
1479 }

1492                                                 u_char* new_compressed_linenumber_table, int new_compressed_linenumber_size, TRAPS) {
1493   // Code below does not work for native methods - they should never get rewritten anyway
1494   assert(!m->is_native(), "cannot rewrite native methods");
1495   // Allocate new Method*
1496   AccessFlags flags = m->access_flags();
1497 
1498   ConstMethod* cm = m->constMethod();
1499   int checked_exceptions_len = cm->checked_exceptions_length();
1500   int localvariable_len = cm->localvariable_table_length();
1501   int exception_table_len = cm->exception_table_length();
1502   int method_parameters_len = cm->method_parameters_length();
1503   int method_annotations_len = cm->method_annotations_length();
1504   int parameter_annotations_len = cm->parameter_annotations_length();
1505   int type_annotations_len = cm->type_annotations_length();
1506   int default_annotations_len = cm->default_annotations_length();
1507 
1508   InlineTableSizes sizes(
1509       localvariable_len,
1510       new_compressed_linenumber_size,
1511       exception_table_len,
1512       checked_exceptions_len,
1513       method_parameters_len,
1514       cm->generic_signature_index(),
1515       method_annotations_len,
1516       parameter_annotations_len,
1517       type_annotations_len,
1518       default_annotations_len,
1519       0);
1520 
1521   ClassLoaderData* loader_data = m->method_holder()->class_loader_data();
1522   Method* newm_oop = Method::allocate(loader_data,
1523                                       new_code_length,
1524                                       flags,
1525                                       &sizes,
1526                                       m->method_type(),
1527                                       CHECK_(methodHandle()));
1528   methodHandle newm (THREAD, newm_oop);
1529 
1530   // Create a shallow copy of Method part, but be careful to preserve the new ConstMethod*
1531   ConstMethod* newcm = newm->constMethod();
1532   int new_const_method_size = newm->constMethod()->size();
1533 
1534   // This works because the source and target are both Methods. Some compilers
1535   // (e.g., clang) complain that the target vtable pointer will be stomped,
1536   // so cast away newm()'s and m()'s Methodness.
1537   memcpy((void*)newm(), (void*)m(), sizeof(Method));
1538 
1539   // Create shallow copy of ConstMethod.
1540   memcpy(newcm, m->constMethod(), sizeof(ConstMethod));
1541 
1542   // Reset correct method/const method, method size, and parameter info
1543   newm->set_constMethod(newcm);
1544   newm->constMethod()->set_code_size(new_code_length);
1545   newm->constMethod()->set_constMethod_size(new_const_method_size);
1546   assert(newm->code_size() == new_code_length, "check");
1547   assert(newm->method_parameters_length() == method_parameters_len, "check");
1548   assert(newm->checked_exceptions_length() == checked_exceptions_len, "check");
1549   assert(newm->exception_table_length() == exception_table_len, "check");
1550   assert(newm->localvariable_table_length() == localvariable_len, "check");
1551   // Copy new byte codes
1552   memcpy(newm->code_base(), new_code, new_code_length);
1553   // Copy line number table
1554   if (new_compressed_linenumber_size > 0) {
1555     memcpy(newm->compressed_linenumber_table(),
1556            new_compressed_linenumber_table,
1557            new_compressed_linenumber_size);
1558   }
1559   // Copy method_parameters
1560   if (method_parameters_len > 0) {
1561     memcpy(newm->method_parameters_start(),
1562            m->method_parameters_start(),
1563            method_parameters_len * sizeof(MethodParametersElement));
1564   }
1565   // Copy checked_exceptions
1566   if (checked_exceptions_len > 0) {
1567     memcpy(newm->checked_exceptions_start(),
1568            m->checked_exceptions_start(),
1569            checked_exceptions_len * sizeof(CheckedExceptionElement));
1570   }
1571   // Copy exception table
1572   if (exception_table_len > 0) {
1573     memcpy(newm->exception_table_start(),
1574            m->exception_table_start(),
1575            exception_table_len * sizeof(ExceptionTableElement));
1576   }
1577   // Copy local variable number table
1578   if (localvariable_len > 0) {
1579     memcpy(newm->localvariable_table_start(),
1580            m->localvariable_table_start(),
1581            localvariable_len * sizeof(LocalVariableTableElement));
1582   }
1583   // Copy stackmap table
1584   if (m->has_stackmap_table()) {
1585     int code_attribute_length = m->stackmap_data()->length();
1586     Array<u1>* stackmap_data =
1587       MetadataFactory::new_array<u1>(loader_data, code_attribute_length, 0, CHECK_(methodHandle()));
1588     memcpy((void*)stackmap_data->adr_at(0),
1589            (void*)m->stackmap_data()->adr_at(0), code_attribute_length);
1590     newm->set_stackmap_data(stackmap_data);
1591   }
1592 
1593   // copy annotations over to new method
1594   newcm->copy_annotations_from(loader_data, cm, CHECK_(methodHandle()));
1595   return newm;
1596 }

1613 void Method::init_intrinsic_id() {
1614   assert(_intrinsic_id == vmIntrinsics::_none, "do this just once");
1615   const uintptr_t max_id_uint = right_n_bits((int)(sizeof(_intrinsic_id) * BitsPerByte));
1616   assert((uintptr_t)vmIntrinsics::ID_LIMIT <= max_id_uint, "else fix size");
1617   assert(intrinsic_id_size_in_bytes() == sizeof(_intrinsic_id), "");
1618 
1619   // the klass name is well-known:
1620   vmSymbols::SID klass_id = klass_id_for_intrinsics(method_holder());
1621   assert(klass_id != vmSymbols::NO_SID, "caller responsibility");
1622 
1623   // ditto for method and signature:
1624   vmSymbols::SID  name_id = vmSymbols::find_sid(name());
1625   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1626       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1627       && name_id == vmSymbols::NO_SID) {
1628     return;
1629   }
1630   vmSymbols::SID   sig_id = vmSymbols::find_sid(signature());
1631   if (klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle)
1632       && klass_id != vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle)
1633       && sig_id == vmSymbols::NO_SID) {
1634     return;
1635   }
1636   jshort flags = access_flags().as_short();
1637 
1638   vmIntrinsics::ID id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1639   if (id != vmIntrinsics::_none) {
1640     set_intrinsic_id(id);
1641     if (id == vmIntrinsics::_Class_cast) {
1642       // Even if the intrinsic is rejected, we want to inline this simple method.
1643       set_force_inline(true);
1644     }
1645     return;
1646   }
1647 
1648   // A few slightly irregular cases:
1649   switch (klass_id) {
1650   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_StrictMath):
1651     // Second chance: check in regular Math.
1652     switch (name_id) {
1653     case vmSymbols::VM_SYMBOL_ENUM_NAME(min_name):
1654     case vmSymbols::VM_SYMBOL_ENUM_NAME(max_name):
1655     case vmSymbols::VM_SYMBOL_ENUM_NAME(sqrt_name):
1656       // pretend it is the corresponding method in the non-strict class:
1657       klass_id = vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_Math);
1658       id = vmIntrinsics::find_id(klass_id, name_id, sig_id, flags);
1659       break;
1660     default:
1661       break;
1662     }
1663     break;
1664 
1665   // Signature-polymorphic methods: MethodHandle.invoke*, InvokeDynamic.*., VarHandle
1666   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_MethodHandle):
1667   case vmSymbols::VM_SYMBOL_ENUM_NAME(java_lang_invoke_VarHandle):
1668     if (!is_native())  break;
1669     id = MethodHandles::signature_polymorphic_name_id(method_holder(), name());
1670     if (is_static() != MethodHandles::is_signature_polymorphic_static(id))
1671       id = vmIntrinsics::_none;
1672     break;
1673 
1674   default:
1675     break;
1676   }
1677 
1678   if (id != vmIntrinsics::_none) {
1679     // Set up its iid.  It is an alias method.
1680     set_intrinsic_id(id);
1681     return;
1682   }
1683 }

1685 bool Method::load_signature_classes(const methodHandle& m, TRAPS) {
1686   if (!THREAD->can_call_java()) {
1687     // There is nothing useful this routine can do from within the Compile thread.
1688     // Hopefully, the signature contains only well-known classes.
1689     // We could scan for this and return true/false, but the caller won't care.
1690     return false;
1691   }
1692   bool sig_is_loaded = true;


1693   ResourceMark rm(THREAD);
1694   for (ResolvingSignatureStream ss(m()); !ss.is_done(); ss.next()) {

1695     if (ss.is_reference()) {
1696       // load everything, including arrays "[Lfoo;"
1697       Klass* klass = ss.as_klass(SignatureStream::ReturnNull, THREAD);


1698       // We are loading classes eagerly. If a ClassNotFoundException or
1699       // a LinkageError was generated, be sure to ignore it.

1889 static void clear_matches(Method* m, int bci) {
1890   InstanceKlass* ik = m->method_holder();
1891   BreakpointInfo* prev_bp = NULL;
1892   BreakpointInfo* next_bp;
1893   for (BreakpointInfo* bp = ik->breakpoints(); bp != NULL; bp = next_bp) {
1894     next_bp = bp->next();
1895     // bci value of -1 is used to delete all breakpoints in method m (ex: clear_all_breakpoint).
1896     if (bci >= 0 ? bp->match(m, bci) : bp->match(m)) {
1897       // do this first:
1898       bp->clear(m);
1899       // unhook it
1900       if (prev_bp != NULL)
1901         prev_bp->set_next(next_bp);
1902       else
1903         ik->set_breakpoints(next_bp);
1904       delete bp;
1905       // When class is redefined JVMTI sets breakpoint in all versions of EMCP methods
1906       // at same location. So we have multiple matching (method_index and bci)
1907       // BreakpointInfo nodes in BreakpointInfo list. We should just delete one
1908       // breakpoint for clear_breakpoint request and keep all other method versions
1909       // BreakpointInfo for future clear_breakpoint request.
1910       // bcivalue of -1 is used to clear all breakpoints (see clear_all_breakpoints)
1911       // which is being called when class is unloaded. We delete all the Breakpoint
1912       // information for all versions of method. We may not correctly restore the original
1913       // bytecode in all method versions, but that is ok. Because the class is being unloaded
1914       // so these methods won't be used anymore.
1915       if (bci >= 0) {
1916         break;
1917       }
1918     } else {
1919       // This one is a keeper.
1920       prev_bp = bp;
1921     }
1922   }
1923 }

2042 // This is a block allocating object, sort of like JNIHandleBlock, only a
2043 // lot simpler.
2044 // It's allocated on the CHeap because once we allocate a jmethodID, we can
2045 // never get rid of it.
2046 
2047 static const int min_block_size = 8;
2048 
2049 class JNIMethodBlockNode : public CHeapObj<mtClass> {
2050   friend class JNIMethodBlock;
2051   Method**        _methods;
2052   int             _number_of_methods;
2053   int             _top;
2054   JNIMethodBlockNode* _next;
2055 
2056  public:
2057 
2058   JNIMethodBlockNode(int num_methods = min_block_size);
2059 
2060   ~JNIMethodBlockNode() { FREE_C_HEAP_ARRAY(Method*, _methods); }
2061 
2062   void ensure_methods(int num_addl_methods) {
2063     if (_top < _number_of_methods) {
2064       num_addl_methods -= _number_of_methods - _top;
2065       if (num_addl_methods <= 0) {
2066         return;
2067       }
2068     }
2069     if (_next == NULL) {
2070       _next = new JNIMethodBlockNode(MAX2(num_addl_methods, min_block_size));
2071     } else {
2072       _next->ensure_methods(num_addl_methods);
2073     }
2074   }
2075 };
2076 
2077 class JNIMethodBlock : public CHeapObj<mtClass> {
2078   JNIMethodBlockNode _head;
2079   JNIMethodBlockNode *_last_free;
2080  public:
2081   static Method* const _free_method;
2082 
2083   JNIMethodBlock(int initial_capacity = min_block_size)
2084       : _head(initial_capacity), _last_free(&_head) {}
2085 
2086   void ensure_methods(int num_addl_methods) {
2087     _last_free->ensure_methods(num_addl_methods);
2088   }
2089 
2090   Method** add_method(Method* m) {
2091     for (JNIMethodBlockNode* b = _last_free; b != NULL; b = b->_next) {
2092       if (b->_top < b->_number_of_methods) {
2093         // top points to the next free entry.
2094         int i = b->_top;
2095         b->_methods[i] = m;
2096         b->_top++;
2097         _last_free = b;
2098         return &(b->_methods[i]);
2099       } else if (b->_top == b->_number_of_methods) {
2100         // if the next free entry ran off the block see if there's a free entry
2101         for (int i = 0; i < b->_number_of_methods; i++) {
2102           if (b->_methods[i] == _free_method) {
2103             b->_methods[i] = m;
2104             _last_free = b;
2105             return &(b->_methods[i]);
2106           }
2107         }
2108         // Only check each block once for frees.  They're very unlikely.
2109         // Increment top past the end of the block.
2110         b->_top++;
2111       }
2112       // need to allocate a next block.
2113       if (b->_next == NULL) {
2114         b->_next = _last_free = new JNIMethodBlockNode();
2115       }
2116     }
2117     guarantee(false, "Should always allocate a free block");
2118     return NULL;
2119   }
2120 
2121   bool contains(Method** m) {
2122     if (m == NULL) return false;
2123     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2124       if (b->_methods <= m && m < b->_methods + b->_number_of_methods) {
2125         // This is a bit of extra checking, for two reasons.  One is
2126         // that contains() deals with pointers that are passed in by
2127         // JNI code, so making sure that the pointer is aligned
2128         // correctly is valuable.  The other is that <= and > are
2129         // technically not defined on pointers, so the if guard can
2130         // pass spuriously; no modern compiler is likely to make that
2131         // a problem, though (and if one did, the guard could also
2132         // fail spuriously, which would be bad).
2133         ptrdiff_t idx = m - b->_methods;
2134         if (b->_methods + idx == m) {
2135           return true;
2136         }
2137       }
2138     }
2139     return false;  // not found
2140   }
2141 
2142   // Doesn't really destroy it, just marks it as free so it can be reused.
2143   void destroy_method(Method** m) {
2144 #ifdef ASSERT
2145     assert(contains(m), "should be a methodID");
2146 #endif // ASSERT
2147     *m = _free_method;
2148   }
2149 
2150   // During class unloading the methods are cleared, which is different
2151   // than freed.
2152   void clear_all_methods() {
2153     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2154       for (int i = 0; i< b->_number_of_methods; i++) {
2155         b->_methods[i] = NULL;
2156       }
2157     }
2158   }
2159 #ifndef PRODUCT
2160   int count_methods() {
2161     // count all allocated methods
2162     int count = 0;
2163     for (JNIMethodBlockNode* b = &_head; b != NULL; b = b->_next) {
2164       for (int i = 0; i< b->_number_of_methods; i++) {
2165         if (b->_methods[i] != _free_method) count++;
2166       }
2167     }
2168     return count;
2169   }
2170 #endif // PRODUCT
2171 };

2311   }

2434 // TouchedMethodRecord -- we can't use a HashtableEntry<Method*> because
2435 // the Method may be garbage collected. Let's roll our own hash table.
2436 class TouchedMethodRecord : CHeapObj<mtTracing> {
2437 public:
2438   // It's OK to store Symbols here because they will NOT be GC'ed if
2439   // LogTouchedMethods is enabled.
2440   TouchedMethodRecord* _next;
2441   Symbol* _class_name;
2442   Symbol* _method_name;
2443   Symbol* _method_signature;
2444 };
2445 
2446 static const int TOUCHED_METHOD_TABLE_SIZE = 20011;
2447 static TouchedMethodRecord** _touched_method_table = NULL;
2448 
2449 void Method::log_touched(TRAPS) {
2450 
2451   const int table_size = TOUCHED_METHOD_TABLE_SIZE;
2452   Symbol* my_class = klass_name();
2453   Symbol* my_name  = name();
2454   Symbol* my_sig   = signature();
2455 
2456   unsigned int hash = my_class->identity_hash() +
2457                       my_name->identity_hash() +
2458                       my_sig->identity_hash();
2459   juint index = juint(hash) % table_size;
2460 
2461   MutexLocker ml(THREAD, TouchedMethodLog_lock);
2462   if (_touched_method_table == NULL) {
2463     _touched_method_table = NEW_C_HEAP_ARRAY2(TouchedMethodRecord*, table_size,
2464                                               mtTracing, CURRENT_PC);
2465     memset(_touched_method_table, 0, sizeof(TouchedMethodRecord*)*table_size);
2466   }
2467 
2468   TouchedMethodRecord* ptr = _touched_method_table[index];
2469   while (ptr) {
2470     if (ptr->_class_name       == my_class &&
2471         ptr->_method_name      == my_name &&
2472         ptr->_method_signature == my_sig) {
2473       return;
2474     }
2475     if (ptr->_next == NULL) break;
2476     ptr = ptr->_next;
2477   }
2478   TouchedMethodRecord* nptr = NEW_C_HEAP_OBJ(TouchedMethodRecord, mtTracing);
2479   my_class->increment_refcount();
2480   my_name->increment_refcount();
2481   my_sig->increment_refcount();
2482   nptr->_class_name         = my_class;
2483   nptr->_method_name        = my_name;
2484   nptr->_method_signature   = my_sig;
2485   nptr->_next               = NULL;
2486 
2487   if (ptr == NULL) {
2488     // first
2489     _touched_method_table[index] = nptr;
2490   } else {
2491     ptr->_next = nptr;
2492   }
2493 }
